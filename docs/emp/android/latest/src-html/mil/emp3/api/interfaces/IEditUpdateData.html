001package mil.emp3.api.interfaces;
002
003import mil.emp3.api.enums.FeatureEditUpdateTypeEnum;
004import mil.emp3.api.enums.FeaturePropertyChangedEnum;
005
006import org.cmapi.primitives.IGeoAirControlMeasure;
007import org.cmapi.primitives.IGeoMilSymbol;
008
009/**
010 * This interface allows access to the edit update event data.
011 */
012public interface IEditUpdateData {
013
014    /**
015     * This method return the type of update the EDIT_UPDATE event is indicating.
016     * @return {@link FeatureEditUpdateTypeEnum}
017     */
018    FeatureEditUpdateTypeEnum getUpdateType();
019    
020    /**
021     * This method returns an array of indexes of coordinates affected by the update operation.
022     * @return An array of indexes if getUpdateType return  FeatureEditUpdateTypeEnum.COORDINATE_ADDED, COORDINATE_MOVED, or COORDINATE_DELETED. See @see FeatureEditUpdateTypeEnum.
023     */
024    int[] getCoordinateIndexes();
025
026    /**
027     * This method return the MilStd modifier that was updated. This only applies to MilStd features.
028     * @return IGeoMilSymbol.Modifier enumerated value if getUpdateType return {@link FeatureEditUpdateTypeEnum}.MILSTD_MODIFIER_UPDATED}, null otherwise.
029     */
030    IGeoMilSymbol.Modifier getChangedModifier();
031
032    /**
033     * This method return the ACM attribute that was updated. This only applies to ACM features.
034     * @return IGeoAirControlMeasure.Attribute enumerated value if getUpdateType return FeatureEditUpdateTypeEnum.ACM_ATTRIBUTE_UPDATED, null otherwise.
035     */
036    IGeoAirControlMeasure.Attribute getChangedAttribute();
037
038    /**
039     * This method returns the feature property that was updated. This only applies to feature update events where the getUpdateType returns FeatureEditUpdateTypeEnum.FEATURE_PROPERTY_UPDATED.
040     * @return FeaturePropertyChangedEnum value if getUpdateType returns FeatureEditUpdateTypeEnum.FEATURE_PROPERTY_UPDATED, null otherwise.
041     */
042    public FeaturePropertyChangedEnum getChangedProperty();
043}
