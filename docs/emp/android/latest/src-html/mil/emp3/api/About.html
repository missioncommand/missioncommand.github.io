001package mil.emp3.api;
002
003import android.util.Log;
004
005import java.lang.reflect.Field;
006import java.util.ArrayList;
007import java.util.List;
008
009import mil.emp3.api.interfaces.IMap;
010import mil.emp3.api.interfaces.core.IStorageManager;
011import mil.emp3.api.utils.ManagerFactory;
012import mil.emp3.mapengine.interfaces.IMapInstance;
013
014/**
015 * Retrieves version information from all available BuildConfig classes. BuildConfig class is generated by the build process
016 * for each Android Library and Android APK. emp3-android-common has been modified to included the fields listed in buildConfigFields
017 * listed below. Map Engines have one additional field which shows the map engine version and Id. IMapInstance is modified
018 * to print the engine related version information.
019 *
020 * emp3-android-common was also updated to set versionCode and versionName which are standard android fields in the application
021 * manifest. Since that information will be lost when Application using EMP API is built (manifests are merged) we don't use that
022 * information here. That information is visible for the sdk-apk and mapengine-apk when they are installed on the device.
023 */
024public class About {
025    private static String TAG = About.class.getSimpleName();
026
027    static final private IStorageManager storageManager = ManagerFactory.getInstance().getStorageManager();
028
029    /*
030        If you add any other android libraries or APKs please update the following array. Not that it doesn't include
031        map engines. We will invoke a method on MapInstance to get map engine information.
032     */
033    private static String[] buildConfigClasses = {
034            "mil.emp3.apk.BuildConfig",
035            "mil.emp3.view.BuildConfig",
036            "mil.emp3.mirrorcache.api.BuildConfig",
037            "mil.emp3.mirrorcache.mirrorables.BuildConfig"
038    };
039
040    /*
041        Following information will be printed per component. If you want to change this then you will need to visit
042        emp3-android-common and each of the map engine build scripts.
043     */
044    private static List<String> buildConfigFields = new ArrayList<>();
045    static
046    {
047        buildConfigFields.add("Implementation_Id");
048        buildConfigFields.add("Implementation_Title");
049        buildConfigFields.add("Implementation_Vendor");
050        buildConfigFields.add("Implementation_Version");
051        buildConfigFields.add("Implementation_Map_Engine_Id");
052        buildConfigFields.add("Implementation_Sec_Renderer_Id");
053        buildConfigFields.add("Implementation_Sec_Cmapi_Id");
054    }
055
056    /**
057     * For each available BuildConfig object print out the above properties.
058     * Get the MapInstance associated with the client map and print its properties.
059     * Note that all BuildConfig classes may not be available in each application as it depend on how application has
060     * chosen to compile/link these packages (static load, dynamic load etc).
061     * @param clientmap
062     * @return
063     */
064    public static String getVersionInformation(IMap clientmap) {
065        StringBuilder builder = new StringBuilder();
066
067        for(String bc: buildConfigClasses) {
068            builder.append("\n");
069            try {
070                Class<?> clazz = null;
071
072                /**
073                 * Adjusted for PluggableMap and WorldWindPlugin (a plugin for PluggableMap) loading
074                 */
075                try {
076                    clazz = About.class.getClassLoader().loadClass(bc);
077                } catch (ClassNotFoundException e) {
078                    Log.e(TAG, "getVersionInformation " + bc + " not available via About class loader");
079                }
080
081                if(null == clazz) {
082                    if(null == clientmap) {
083                        throw new ClassNotFoundException("bc");
084                    }
085                    clazz = clientmap.getClass().getClassLoader().loadClass(bc);
086                }
087
088                Field[] declaredFields = clazz.getDeclaredFields();
089                for (Field field : declaredFields) {
090                    if (java.lang.reflect.Modifier.isStatic(field.getModifiers())) {
091                        if(buildConfigFields.contains(field.getName())) {
092                            try {
093                                builder.append("\n\t\t" + (String)field.get(null));
094                            } catch (IllegalAccessException e) {
095                                Log.e(TAG, "getVersionInformation " + field.getName(), e);
096                            }
097                        }
098                    }
099                }
100            } catch (ClassNotFoundException e) {
101                Log.e(TAG, "getVersionInformation " + bc + " not available");
102            }
103        }
104
105        IMapInstance mapInstance = storageManager.getMapInstance(clientmap);
106        if(null != mapInstance) {
107            builder.append("\n");
108            mapInstance.getVersionInformation(builder, buildConfigFields);
109        }
110
111        builder.append("\n\n");
112        Log.d(TAG, builder.toString());
113
114        return builder.toString();
115    }
116}
