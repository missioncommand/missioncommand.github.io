001package mil.emp3.api.utils;
002
003import android.util.Log;
004
005import org.cmapi.primitives.GeoPosition;
006import org.cmapi.primitives.IGeoPosition;
007
008import mil.emp3.api.global;
009
010/**
011 * Convenience class for for creating a GeoPosition obkect.
012 */
013
014public class EmpGeoPosition extends GeoPosition {
015    private final static String TAG = EmpGeoPosition.class.getSimpleName();
016    /**
017     * Creates EmpGeoPosition object from latitude and longitude. Sets altitude to 0.
018     * @param latitude
019     * @param longitude
020     */
021    public EmpGeoPosition(double latitude, double longitude) {
022        setLatitude(latitude);
023        setLongitude(longitude);
024        setAltitude(0);
025    }
026
027    /**
028     * Creates EmpGeoPosition object from latitude, longitude, and altitude.
029     * @param latitude
030     * @param longitude
031     * @param altitude
032     */
033    public EmpGeoPosition(double latitude, double longitude, double altitude) {
034        setLatitude(latitude);
035        setLongitude(longitude);
036        setAltitude(altitude);
037    }
038
039    /**
040     * Copies latitude, longitude and altitude.
041     * @param geoPosition
042     */
043    public EmpGeoPosition(IGeoPosition geoPosition) {
044        setLatitude(geoPosition.getLatitude());
045        setLongitude(geoPosition.getLongitude());
046        setAltitude(geoPosition.getAltitude());
047    }
048
049    public EmpGeoPosition() {
050
051    }
052
053    public boolean isValid() {
054        return validate(getLatitude(), getLongitude(), getAltitude());
055    }
056
057    public static boolean validate(double latitude, double longitude, double altitude) {
058        if(Double.isNaN(latitude) || latitude < global.LATITUDE_MINIMUM || latitude > global.LATITUDE_MAXIMUM) {
059            Log.e(TAG, "Invalid Latitude " + latitude);
060            return false;
061        }
062        if(Double.isNaN(longitude) || longitude < global.LONGITUDE_MINIMUM || longitude > global.LONGITUDE_MAXIMUM) {
063            Log.e(TAG, "Invalid Longitude " + longitude);
064            return false;
065        }
066        if(Double.isNaN(altitude)) {
067            Log.e(TAG, "Invalid altitude " + altitude);
068            return false;
069        }
070        return true;
071    }
072
073    public static boolean validate(IGeoPosition position) {
074        return validate(position.getLatitude(), position.getLongitude(), position.getAltitude());
075    }
076}
