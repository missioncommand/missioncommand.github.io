001package mil.emp3.api.enums;
002
003/**
004 * Configuration property keys used by properties parameter of various API calls. Each enumerated
005 * value requires a specific value type.
006 */
007public enum Property {
008    /**
009     * Fully qualified class name for a map engine. Property value must be of String type
010     */
011    ENGINE_CLASSNAME ("engine.className"),
012    /**
013     * Fully qualified android APK name that contains an implementation of engine.className. Property value must be of String type.
014     */
015    ENGINE_APKNAME   ("engine.apkName")  ,
016    /**
017     * Stroke color used to highlight selected feature(s) (Features selected using selectFeature method of IMap). The property value must be of
018     * type IGeoColor. The default color is yellow.
019     */
020    DEFAULT_SELECT_STROKE_COLOR ("default.stroke.color"),
021    /**
022     * Scale factor to use when an icon type feature is selected. Property value must be of type double. The default value is 1.2.
023     */
024    DEFAULT_SELECT_ICON_SCALE   ("default.icon.scale"),
025    /**
026     * Debug setting for the API and implementation code. Property value must be of type boolean. The default is
027     * false. Setting this value to true will generate additional log statements.
028     */
029    DEBUG                       ("debug"),
030    /**
031     * Proxy url to be used when a feature or map service set useProxy to true. The default value is null. Property value must be of type string.
032     */
033    PROXY_URL                   ("proxy.url"),
034    /**
035     * MirrorCache mode to be used. The default value is {@link mil.emp3.api.enums.MirrorCacheModeEnum#DISABLED}. This property requires a
036     * {@link mil.emp3.api.enums.MirrorCacheModeEnum} value.
037     */
038    MIRROR_CACHE_MODE           ("mirror.cache.mode"),
039    ;
040
041    final private String value;
042    Property(String value) {
043        this.value = value;
044    }
045
046    public String getValue() {
047        return value;
048    }
049
050    static public Property fromValue(String v) {
051        for (Property c : Property.values()) {
052            if (c.value.equals(v)) {
053                return c;
054            }
055        }
056        throw new IllegalArgumentException(v);
057    }
058}
