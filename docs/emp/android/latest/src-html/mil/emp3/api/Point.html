001package mil.emp3.api;
002
003
004import org.cmapi.primitives.GeoPoint;
005import org.cmapi.primitives.IGeoIconStyle;
006import org.cmapi.primitives.IGeoPoint;
007import org.cmapi.primitives.IGeoPosition;
008
009import mil.emp3.api.enums.FeatureTypeEnum;
010import mil.emp3.api.abstracts.Feature;
011
012import org.cmapi.primitives.GeoPosition;
013
014/**
015 * This class implements the Point feature that encapsulates a GeoPoint object.
016 */
017public class Point extends Feature<IGeoPoint> implements IGeoPoint {
018    private double dIconScale = 1.0;
019    private int resourceId = 0;
020
021    /**
022     * this is the default constructor.
023     */
024    public Point() {
025        super(new GeoPoint(), FeatureTypeEnum.GEO_POINT);
026    }
027
028    /**
029     * This constructor creates a Point feature wherer the icon is accessed via the URL provided. The caller must
030     * provide the GeoIconStyle to correctly position the icon.
031     * @param sURL A valid URL
032     */
033    public Point(String sURL) {
034        super(new GeoPoint(), FeatureTypeEnum.GEO_POINT);
035        this.setIconURI(sURL);
036    }
037
038    /**
039     * This constructor creates a point feature positioned at the coordinates provided.
040     * @param dLat
041     * @param dLong
042     */
043    public Point(double dLat, double dLong) {
044        super(new GeoPoint(), FeatureTypeEnum.GEO_POINT);
045        
046        IGeoPosition oPos = new GeoPosition();
047        oPos.setLatitude(dLat);
048        oPos.setLongitude(dLong);
049        this.setPosition(oPos);
050    }
051
052    /**
053     * This constructor creates a point feature from a geo point object.
054     * @param oRenderable
055     */
056    public Point(IGeoPoint oRenderable) {
057        super(oRenderable, FeatureTypeEnum.GEO_POINT);
058    }
059
060    /**
061     * This method set the image style.
062     * @param oStyle See {@link IGeoIconStyle}
063     */
064    @Override
065    public void setIconStyle(IGeoIconStyle oStyle) {
066        ((IGeoPoint) this.getRenderable()).setIconStyle(oStyle);
067    }
068
069    /**
070     * This method retrieves the icon style of the point if one has been set.
071     * @return An {@link IGeoIconStyle} or null
072     */
073    @Override
074    public IGeoIconStyle getIconStyle() {
075        return this.getRenderable().getIconStyle();
076    }
077
078    /**
079     * This method set the image URL of a point feature. It will override the default icon.
080     * @param sURL The URL of the image.
081     */
082    @Override
083    public void setIconURI(String sURL) {
084        ((IGeoPoint) this.getRenderable()).setIconURI(sURL);
085    }
086
087    /**
088     * This method retrieves the URL of the image
089     * @return URL string.
090     */
091    @Override
092    public String getIconURI() {
093        return ((IGeoPoint) this.getRenderable()).getIconURI();
094    }
095
096    /**
097     * This method sets the icon scale. The size of the icon will be modified by this factor when rendered.
098     * @param dScale A value smaller than 1.0 decreases the size. A value larger than 1.0 increases the size.
099     */
100    public void setIconScale(double dScale) {
101        this.dIconScale = Math.abs(dScale);
102    }
103
104    /**
105     * This method returns the scale value.
106     * @return
107     */
108    public double getIconScale() {
109        return this.dIconScale;
110    }
111
112    /**
113     * This method is an ANDROID ONLY method. It allows the developer to us an android drawable resource
114     * as an icon for a Point feature. The resource must be accessible to the EMP Core. If a
115     * resource Id is set the icon style offset is interpreted as a fraction.
116     * @param resId The resource ID of the resource.
117     */
118    public void setResourceId(int resId) {
119        this.resourceId = resId;
120    }
121
122    /**
123     * This method is an ANDROID ONLY method. It returns the resource Id.
124     * @return The resojurce Id or 0 if one has not been set.
125     */
126    public int getResourceId() {
127        return this.resourceId;
128    }
129}
