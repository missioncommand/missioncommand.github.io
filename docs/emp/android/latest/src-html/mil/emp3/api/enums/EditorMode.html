001package mil.emp3.api.enums;
002
003/**
004 * {@link mil.emp3.api.abstracts.Map} is normally in INACTIVE mode but will transition to one of the Editor Modes as described below.
005 * Mode transitions are controlled by following Map methods
006 * <ul>
007 *     <li>editFeature, cancelEdit, completeEdit - EDIT_MODE</li>
008 *     <li>drawFeature, cancelDraw, completeDraw - DRAW_MODE</li>
009 *     <li>drawFreehand, drawFreehandExit - FREEHAND_MODE</li>
010 * </ul>
011 */
012public enum EditorMode {
013    /**
014     * Map is in normal mode and can transition to any other based on application actions and current value of {@link MapMotionLockEnum}
015     */
016    INACTIVE,
017
018    /**
019     * Map is in editFeature operation and will return to INACTIVE on cancelEdit or completeEdit
020     */
021    EDIT_MODE,
022
023    /**
024     * Map is in drawFeature operation and will return to INACTIVE on cancelDraw or completeDraw
025     */
026    DRAW_MODE,
027
028    /**
029     * Map is within drawFreehand operation and will return to INACTIVE on drawFreehandExit
030     */
031    FREEHAND_MODE
032}
