001package mil.emp3.api.events;
002
003import mil.emp3.api.enums.MapFeatureEventEnum;
004import mil.emp3.api.interfaces.IFeature;
005import mil.emp3.api.interfaces.IMap;
006
007public abstract class MapFeatureAddedEvent extends Event<MapFeatureEventEnum, IMap> {
008    final private IFeature feature;
009
010    protected MapFeatureAddedEvent(MapFeatureEventEnum eEvent, IMap map, IFeature feature) {
011        super(eEvent, map);
012        this.feature = feature;
013    }
014
015    /**
016     * Returns the feature for which the event was generated.
017     * @return the feature added
018     */
019    public IFeature getFeature() {
020        return feature;
021    }
022}
