001package mil.emp3.api.events;
002
003import mil.emp3.api.interfaces.IContainer;
004import mil.emp3.api.interfaces.IFeature;
005import mil.emp3.api.interfaces.IMap;
006import mil.emp3.api.interfaces.IOverlay;
007import mil.emp3.api.enums.ContainerEventEnum;
008import org.cmapi.primitives.IGeoBase;
009
010import java.util.List;
011
012/**
013 * This class implements an abstract Container event. It provides access to all event properties.
014 */
015public abstract class ContainerEvent extends Event<ContainerEventEnum, IContainer> {
016    /**
017     * This constructor is used by the EMP core to create the event.
018     * @param eEvent {@link ContainerEventEnum} This parameter indicates the type of container event.
019     * @param oTarget this parameter indicates the actual container in which the events has occurred.
020     */
021    protected ContainerEvent(ContainerEventEnum eEvent, IContainer oTarget) {
022        super(eEvent, oTarget);
023    }
024
025    /**
026     * This method retrieves a list of the children that the events refers to. If
027     * The event is an ADD, the list contains the children that were added. If its a REMOVE
028     * event the list of children that were removed.
029     * @return 
030     */
031    public abstract List<IGeoBase> getAffectedChildren();
032
033    /**
034     * This method is a convenience function that indicate if the target is an IMap
035     * @return True if the target object implements the IMap interface. False otherwise.
036     */
037    public boolean targetIsMap() {
038        return (this.getTarget() instanceof IMap);
039    }
040
041    /**
042     * This method is a convenience function that indicate if the target is an IOverlay
043     * @return True if the target object implements the IOverlay interface. False otherwise.
044     */
045    public boolean targetIsOverlay() {
046        return (this.getTarget() instanceof IOverlay);
047    }
048
049    /**
050     * This method is a convenience function that indicate if the target is an IFeature
051     * @return True if the target object implements the IFeature interface. False otherwise.
052     */
053    public boolean targetIsFeature() {
054        return (this.getTarget() instanceof IFeature);
055    }
056
057    /**
058     * This method retrieves the target as an IMap object.
059     * @return An IMap object if the target implements the IMap interface. null otherwise.
060     */
061    public IMap getTargetMap() {
062        if (this.targetIsMap()) {
063            return (IMap) this.getTarget();
064        }
065        return null;
066    }
067
068    /**
069     * This method retrieves the target as an IOverlay object.
070     * @return An IOverlay object if the target implements the IOverlay interface. null otherwise.
071     */
072    public IOverlay getTargetOverlay() {
073        if (this.targetIsOverlay()) {
074            return (IOverlay) this.getTarget();
075        }
076        return null;
077    }
078
079    /**
080     * This method retrieves the target as an IFeature object.
081     * @return An IFeature object if the target implements the IFeature interface. null otherwise.
082     */
083    public IFeature getTargetFeature() {
084        if (this.targetIsFeature()) {
085            return (IFeature) this.getTarget();
086        }
087        return null;
088    }
089}
