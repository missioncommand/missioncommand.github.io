001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoDocument;
004import org.cmapi.primitives.IGeoRenderable;
005
006import java.util.List;
007
008/**
009 * This class defines the interface to a KML Feature object.
010 */
011public interface IKML extends IFeature<IGeoRenderable>, IGeoDocument {
012    /**
013     * This method retrieves the list of feature generated from the KML document. This method is for
014     * internal use only. Client software manipulating the list or the features listed, will cause unforeseen results.
015     * @return The list of features.
016     */
017    List<IFeature> getFeatureList();
018
019    /**
020     * This method retrieves the list of image layers generated from the ground overlays in the KML document. This method is for
021     * internal use only. Client software manipulating the list or the image layers listed, will cause unforeseen results.
022     * @return The list of image layers.
023     */
024    List<IImageLayer> getImageLayerList();
025
026    /**
027     * This method retrieves a list of features which are generated from a KML placemark with the specified KML id.
028     * @param kmlId The placemark Id.
029     * @return The list of features with the specified KML id.
030     */
031    List<IFeature> findKMLId(String kmlId);
032
033    /**
034     * This method retrieves a KML string representing all contained features and image layers. It is important
035     * to understand that the string returned is a valid KML document, but may not match the initial document.
036     * @return
037     */
038    String exportToKML();
039}
