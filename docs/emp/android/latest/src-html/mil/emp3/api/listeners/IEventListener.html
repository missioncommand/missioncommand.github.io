001package mil.emp3.api.listeners;
002
003
004import mil.emp3.api.interfaces.IEvent;
005
006/**
007 * This interface defines the base of all event listener interfaces.
008 * @param <T> An object the is derived from {@link IEvent}.<br/>
009 * <br/>
010 * See the
011 * <a href="package-summary.html#package.description">
012 *    package overview
013 * </a> for more information.
014 */
015public interface IEventListener<T extends IEvent> {
016
017  /** 
018   * This method must be implemented by an object class to handle T type events.
019   * @param event An event class that implements the {@link IEvent} interface.
020   */
021  void onEvent(T event);
022}
