001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004/**
005 * Types user interaction events that can be generated by a map based on gestures.
006 */
007public enum UserInteractionEventEnum implements IEventEnum {
008    /**
009     * Object was clicked.
010     */
011    CLICKED,
012
013    /**
014     * Object was double clicked.
015     */
016    DOUBLE_CLICKED,
017
018    /**
019     * Object was long pressed.
020     */
021    LONG_PRESS,
022    //PRESS_RELEASE,
023
024    /**
025     * Feature or map was dragged.
026     */
027    DRAG,
028    /**
029     * Drag operation has completed, i.e. user lifted the finger.
030     */
031    DRAG_COMPLETE
032}
