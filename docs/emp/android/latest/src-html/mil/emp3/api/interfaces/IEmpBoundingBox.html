001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoBounds;
004import org.cmapi.primitives.IGeoPosition;
005
006import mil.emp3.api.utils.EmpBoundingBox;
007
008/**
009 * This class defines the interface to an EMP bounding box.
010 */
011
012public interface IEmpBoundingBox extends IGeoBounds {
013    /**
014     * This method returns the northern latitude.
015     * @return The southern latitude in degrees.
016     */
017    double north();
018
019    /**
020     * This method returns the southern latitude.
021     * @return The southern latitude in degrees.
022     */
023    double south();
024
025    /**
026     * This method returns the eastern longitude.
027     * @return The eastern longitude in degrees.
028     */
029    double east();
030
031    /**
032     * This method returns the western longitude.
033     * @return The western logintude in degrees.
034     */
035    double west();
036
037    /**
038     * This method returns the delta of the north and south latitudes.
039     * @return Degrees
040     */
041    double deltaLatitude();
042
043    /**
044     * This method returns the delta between the west and east longitudes.
045     * @return degrees.
046     */
047    double deltaLongitude();
048
049    /**
050     * This method check if the international date line is within the bounding box.
051     * @return True if the IDL is contained in the bounding box, false if not.
052     */
053    boolean containsIDL();
054
055    void copyFrom(IGeoBounds geoBounds);
056
057    void copyFrom(IEmpBoundingBox empBounds);
058
059    boolean intersects(double northDeg, double southDeg, double eastDeg, double westDeg);
060
061    boolean intersects(IGeoBounds bounds);
062
063    boolean intersects(IEmpBoundingBox bounds);
064
065    /**
066     * This method returns the coordinate of the middle of the west edge of the boundary.
067     * @return
068     */
069    IGeoPosition centerWest();
070
071    /**
072     * This method returns the coordinate of the middle of the east edge of the boundary.
073     * @return
074     */
075    IGeoPosition centerEast();
076
077    /**
078     * This method returns the coordinate of the middle of the north edge of the boundary.
079     * @return
080     */
081    IGeoPosition centerNorth();
082
083    /**
084     * This method returns the coordinate of the middle of the south edge of the boundary.
085     * @return
086     */
087    IGeoPosition centerSouth();
088
089    /**
090     * This method checks if the the given lat/long is contained in the bounding box.
091     * @param latitude
092     * @param longitude
093     * @return If if it is contained in the bounding box, false otherwise.
094     */
095    boolean contains(double latitude, double longitude);
096
097    /**
098     * This method returns the width in meters of the bounding box across the center.
099     * @return The Width in meters.
100     */
101    double widthAcrossCenter();
102
103    /**
104     * This method returns the height in meters of the bounding box across the center.
105     * @return The Height in meters.
106     */
107    double heightAcrossCenter();
108
109    /**
110     * This method returns the latitude of the center of the bounding area.
111     * @return Center latitude in degrees.
112     */
113    double centerLatitude();
114
115    /**
116     * This method returns the longitude of the center of the bounding area.
117     * @return Center longitude in degrees.
118     */
119    double centerLongitude();
120
121    /**
122     * This method returns the bounding box that is the intersection of this box and the box defined
123     * by the with parameter. The resulting box is returned in the result parameter.
124     * @param with The bounding box to intersect with.
125     * @param result A bounding box to place the result.
126     * @return The result parameter with the intersecting bounds or null if the two bounding boxes do not intersect.
127     */
128    IEmpBoundingBox intersection(IEmpBoundingBox with, IEmpBoundingBox result);
129
130    /**
131     * This method returns the bounding box that is the intersection of this box and the box defined
132     * by the with parameter.
133     * @param with The bounding box to intersect with.
134     * @return The resulting bounding box. Or null if the two bounding boxes do not intersect.
135     */
136    IEmpBoundingBox intersection(IEmpBoundingBox with);
137
138    /**
139     * This method will ensure that the bounding box includes the position provided.
140     * @param latitude     The latitude in degrees.
141     * @param longitude    The longitude in degrees.
142     */
143    void includePosition(double latitude, double longitude);
144}
