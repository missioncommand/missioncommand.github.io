019/**
020 * When map view changes i.e. pan/zoom/tilt/rotate/heading we recalculate the quadrilateral that approximates visible geographical area
021 * of the map view. When the entire view is occupied by the map we can get the exact geographical coordinates of the four corners. When the map
022 * shows sky/space then we provide the best possible geographical coordinates. SEC Military Symbol render-er is one know customer of this
023 * capability.
024 *
025 * The vertices represent a polygon/quadrilateral and they are stored in clockwise order. You cannot associate any specific vertex to any specific
026 * corner of the view. The class guarentees that all four vertices are non-null and have valid latitude/longitude values. Altitude is always set to zero.
027 *
028 * Class also provides a capability to to convert (approximately) these four vertices to GeoBounds object i.e. North, East, West and South
029 * boundaries.
030 *
031 * If you are interested in the logic used to generate this object then please refer to BoundsGeneration class in mapengine-worldwind-apk project.
032 */
033
034public class EmpBoundingArea extends GeoBounds implements IEmpBoundingArea {
035    private static String TAG = EmpBoundingArea.class.getSimpleName();
036    public final static int REQUIRED_VERTICES = 4;
037    private final IGeoPosition[] vertices;
038
039    private final ICamera camera;              // Camera when vertices were calculated, we will need this for
040                                               // adjusting the distance.
041    private final boolean cameraOnScreen;      // True if camera position is on screen.
042
043    private final IGeoPosition geometricCenter; // Position corresponding to the geometric center of the vertices.
044
045    // We will adjust vertices to restrict the distance from the camera. Purpose is to reduce the size of resources used
046    // by SEC Military Symbol renderer. We will do the adjustment only when user tries to retrieve anything related
047    // to the vertices.
048
049    private final static double MAX_DISTANCE_FROM_CENTER = 5000000; // Maximum distance from center of four vertices.
050
051    private IGeoPosition[] adjustedVertices = null;   // Approximate the vertices to keep longitude range to 180 degrees.
052
053    private String boundingAreaString = null;                // Stores bounding area String that was created in toString operation.
054
055    /**
056     * All parameters must be non-null and have valid values.
057     * @param currentCamera - Camera object when the vertices were calculated.
058     * @param v1
059     * @param v2
060     * @param v3
061     * @param v4
062     */
063    public EmpBoundingArea(ICamera currentCamera, boolean cameraOnScreen, IGeoPosition v1, IGeoPosition v2, IGeoPosition v3, IGeoPosition v4,
064                           IGeoBounds geoBounds, IGeoPosition geometricCenter) {
065
066        if((null == v1) || (null == v2) || (null == v3) || (null == v4)) {
067            throw new IllegalArgumentException("All vertices must be non-null");
068        }
069        vertices = new IGeoPosition[REQUIRED_VERTICES];
070        vertices[0] = v1;
071        vertices[1] = v2;
072        vertices[2] = v3;
073        vertices[3] = v4;
074
075        for(int ii = 0; ii < vertices.length; ii++) {
076            if (Double.isNaN(vertices[ii].getLatitude()) || (vertices[ii].getLatitude() < global.LATITUDE_MINIMUM) || (vertices[ii].getLatitude() > global.LATITUDE_MAXIMUM)) {
077                throw new InvalidParameterException("Latitude is Out Of Range for vertex " + (ii+1));
078            }
079
080            if (Double.isNaN(vertices[ii].getLongitude()) || (vertices[ii].getLongitude() < global.LONGITUDE_MINIMUM) || (vertices[ii].getLongitude() > global.LONGITUDE_MAXIMUM)) {
081                throw new InvalidParameterException("Longitude is Out Of Range for vertex " + (ii+1));
082            }
083
084            vertices[ii].setAltitude(0);
085        }
086
087        camera = new Camera();
088        camera.copySettingsFrom(currentCamera);
089        this.cameraOnScreen = cameraOnScreen;
090        this.geometricCenter = geometricCenter;
091
092        super.setEast(geoBounds.getEast());
093        super.setWest(geoBounds.getWest());
094        super.setNorth(geoBounds.getNorth());
095        super.setSouth(geoBounds.getSouth());
096    }
097
098    private void adjustVerticesByDistance() {
099        if(null != adjustedVertices) return;
100
101        IGeoPosition center;
102
103        if(!cameraOnScreen) {
104            // Find the center of the four vertices.
105            List<IGeoPosition> cornersFound = new ArrayList<>();
106            for (int ii = 0; ii < vertices.length; ii++) {
107                cornersFound.add(vertices[ii]);
108            }
109            center = GeoLibrary.getCenter(cornersFound);
110        } else {
111            center = new GeoPosition();
112            center.setLatitude(camera.getLatitude());
113            center.setLongitude(camera.getLongitude());
114        }
115
116        adjustedVertices = new IGeoPosition[vertices.length];
117        double distance;
118
119        // If distance from center is more than 5,000,000 meters then clip it.
120        for (int ii = 0; ii < vertices.length; ii++) {
121            try {
122                distance = GeoLibrary.computeDistanceBetween(center, vertices[ii]);
123                Log.d(TAG, "distance " + ii + " " + distance);
124                if (distance > MAX_DISTANCE_FROM_CENTER) {
125                    double bearing = GeoLibrary.computeBearing(center, vertices[ii]);
126                    adjustedVertices[ii] = GeoLibrary.computePositionAt(bearing, MAX_DISTANCE_FROM_CENTER, center);
127                } else {
128                    adjustedVertices[ii] = vertices[ii];
129                }
130            } catch(Exception e) {
131                Log.e(TAG, e.getMessage(), e);
132                adjustedVertices[ii] = vertices[ii];
133            }
134        }
135    }
136    /**
137     * Converts vertices to a String to a format required by Military Symbol render-er. It returns a string based on adjustedVertices that
138     * meets the SEC Render-er requirement of 180 degrees max span of longitude. This calculation is very much approximate and may need
139     * more work in future.
140     * @return
141     */
142
143    @Override
144    public String toString() {
145        if(null != boundingAreaString) {
146            return boundingAreaString;
147        }
148        adjustVerticesByDistance();
149        StringBuilder builder = new StringBuilder();
150        for(int ii = adjustedVertices.length - 1; ii >= 0 ; ii--) {
151            builder.append(adjustedVertices[ii].getLongitude() + "," + adjustedVertices[ii].getLatitude() + " ");
152        }
153        builder.append(adjustedVertices[adjustedVertices.length - 1].getLongitude() + "," + adjustedVertices[adjustedVertices.length - 1].getLatitude());
154        boundingAreaString = builder.toString();
155        return boundingAreaString;
156    }
157
158    /**
159     * Fetch the four vertices. We are cloning the vertices and returning. Application will not be allowed to change this object.
160     * @return
161     */
162    public IGeoPosition[] getBoundingVertices() {
163        IGeoPosition[] bounds = new IGeoPosition[REQUIRED_VERTICES];
164        for(int ii = 0; ii < vertices.length; ii++) {
165            bounds[ii] = new GeoPosition();
166            bounds[ii].setLatitude(vertices[ii].getLatitude());
167            bounds[ii].setLongitude(vertices[ii].getLongitude());
168        }
169        return bounds;
170    }

179        throw new IllegalStateException("Cannot set on " + this.getClass().getSimpleName());
180    }

189        throw new IllegalStateException("Cannot set on " + this.getClass().getSimpleName());
190    }

199        throw new IllegalStateException("Cannot set on " + this.getClass().getSimpleName());

209        throw new IllegalStateException("Cannot set on " + this.getClass().getSimpleName());

212    /**
213     * Builds a GeoBounds object using the calculated boundaries.
214     * @return
215     */
216    @Override
217    public IGeoBounds getGeoBounds() {
218        IGeoBounds geoBounds = new GeoBounds();
219        geoBounds.setNorth(getNorth());
220        geoBounds.setEast(getEast());
221        geoBounds.setWest(getWest());
222        geoBounds.setSouth(getSouth());
223        return geoBounds;
224    }
225
226    /**
227     * Builds a EmpBoundingBox using the calculated boundaries.
228     * @return
229     */
230    @Override
231    public EmpBoundingBox getEmpBoundingBox() {
232        EmpBoundingBox empBoundingBox = new EmpBoundingBox(getNorth(), getSouth(), getEast(), getWest());
233        return empBoundingBox;
234    }
235
236    @Override
237    public IGeoPosition getCenter() {
238        List<IGeoPosition> cornersFound = new ArrayList<>();
239
240        IGeoPosition nw = new GeoPosition();
241        nw.setLatitude(getNorth());
242        nw.setLongitude(getWest());
243        cornersFound.add(nw);
244
245        IGeoPosition ne = new GeoPosition();
246        ne.setLatitude(getNorth());
247        ne.setLongitude(getEast());
248        cornersFound.add(ne);
249
250        IGeoPosition se = new GeoPosition();
251        se.setLatitude(getSouth());
252        se.setLongitude(getEast());
253        cornersFound.add(se);
254
255        IGeoPosition sw = new GeoPosition();
256        sw.setLatitude(getSouth());
257        sw.setLongitude(getWest());
258        cornersFound.add(sw);
259
260        return GeoLibrary.getCenter(cornersFound);
261    }
262
263    @Override
264    public boolean cameraPositionIsVisible() {
265        return cameraOnScreen;
266    }
267
268    @Override
269    public IGeoPosition getGeometricCenter() { return new EmpGeoPosition(geometricCenter);}
