019/**
020 * This class implements an polygon. It requires 3 or more geographic coordinates.
021 */
022public class Polygon extends Feature<IGeoPolygon> implements IGeoPolygon {
023
024    /**
025     * This constructor creates a default geo polygon.
026     */
027    public Polygon() {
028        super(new GeoPolygon(), FeatureTypeEnum.GEO_POLYGON);
029    }
030
031    /**
032     * This constructor creates a default geo polygon with the coordinate list provided.
033     * @param oPositionList
034     */
035    public Polygon(List<IGeoPosition> oPositionList) {
036        super(new GeoPolygon(), FeatureTypeEnum.GEO_POLYGON);
037        if (null == oPositionList) {
038            throw new InvalidParameterException("The position list parameter can NOT be null");
039        }
040        this.getRenderable().getPositions().addAll(oPositionList);
041    }
042
043    /**
044     * This constructor creates a polygon feature from the geo polygon provided.
045     * @param oRenderable
046     */
047    public Polygon(IGeoPolygon oRenderable) {
048        super(oRenderable, FeatureTypeEnum.GEO_POLYGON);
049    }
050}
