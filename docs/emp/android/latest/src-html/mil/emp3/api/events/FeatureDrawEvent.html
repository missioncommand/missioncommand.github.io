001package mil.emp3.api.events;
002
003import java.util.List;
004
005import mil.emp3.api.enums.FeatureDrawEventEnum;
006import mil.emp3.api.interfaces.IEditUpdateData;
007import mil.emp3.api.interfaces.IFeature;
008import mil.emp3.api.interfaces.IMap;
009
010/**
011 * This event class is generated during the process of drawing a feature.
012 */
013public abstract class FeatureDrawEvent extends Event<FeatureDrawEventEnum, IFeature> {
014    protected FeatureDrawEvent(FeatureDrawEventEnum eEvent, IFeature oTarget) {
015        super(eEvent, oTarget);
016    }
017
018    /**
019     * This method return the map the event occurred on.
020     * @return An object that implements the {@link IMap} interface.
021     */
022    public abstract IMap getMap();
023
024    /**
025     * This method retrieves a list of object that identifies what was updated in the feature. It only applies to DRAW_UPDATE events.
026     * @return A list of {@link IEditUpdateData} object for DRAW_UPDATE events or null otherwise.
027     */
028    public abstract List<IEditUpdateData> getUpdateData();
029}
