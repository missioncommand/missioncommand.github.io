024/**
025 * This class implements an ellipse feature. It requires a single coordinate that indicates the
026 * geographic position of the center of the ellipse.
027 */
028public class Ellipse extends Feature<IGeoEllipse> implements IGeoEllipse {
029
030    public final static double MINIMUM_SEMI_MAJOR = global.MINIMUM_DISTANCE;
031    public final static double MINIMUM_SEMI_MINOR = global.MINIMUM_DISTANCE;
032
033    /**
034     * This constructor creates an ellipse with default dimensions.
035     */
036    public Ellipse() {
037        super(new GeoEllipse(), FeatureTypeEnum.GEO_ELLIPSE);
038        this.setFillStyle(null);
039    }
040
041    /**
042     * This constructor creates an ellipse with the given major and minor radius. Negative values are made positive. The absolute value of the values must be >= 1.0.
043     * @param dMajorRadius see {@link #setSemiMajor(double)}.
044     * @param dMinorRadius  see {@link #setSemiMinor(double)}.
045     */
046    public Ellipse(double dMajorRadius, double dMinorRadius) {
047        super(new GeoEllipse(), FeatureTypeEnum.GEO_ELLIPSE);
048        dMajorRadius = makePositive(dMajorRadius, "Invalid major radius. NaN");
049        dMinorRadius = makePositive(dMinorRadius, "Invalid minor radius. NaN");
050
051        if((dMajorRadius < MINIMUM_SEMI_MAJOR) || (dMinorRadius < MINIMUM_SEMI_MINOR)) {
052            throw new InvalidParameterException("Invalid Major or Minor. " + dMajorRadius + " " + dMinorRadius +
053                    " Minimum supported " + MINIMUM_SEMI_MAJOR + " " + MINIMUM_SEMI_MINOR);
054        }
055        this.setSemiMajor(dMajorRadius);
056        this.setSemiMinor(dMinorRadius);
057
058        this.setFillStyle(null);
059    }
060
061    /**
062     * This constructor creates an ellipse with the given major, minor radius and oriented at the azimuth angle from north.
063     * @param dMajorRadius see {@link #setSemiMajor(double)}.
064     * @param dMinorRadius  see {@link #setSemiMinor(double)}.
065     * @param dAzimuth see {@link Feature#setAzimuth(double)}.
066     */
067    public Ellipse(double dMajorRadius, double dMinorRadius, double dAzimuth) {
068        super(new GeoEllipse(), FeatureTypeEnum.GEO_ELLIPSE);
069        dMajorRadius = makePositive(dMajorRadius, "Invalid major radius. NaN");
070        dMinorRadius = makePositive(dMinorRadius, "Invalid minor radius. NaN");
071
072        if((dMajorRadius < MINIMUM_SEMI_MAJOR) || (dMinorRadius < MINIMUM_SEMI_MINOR)) {
073            throw new InvalidParameterException("Invalid Major or Minor. " + dMajorRadius + " " + dMinorRadius +
074                    " Minimum supported " + MINIMUM_SEMI_MAJOR + " " + MINIMUM_SEMI_MINOR);
075        }
076
077        this.setAzimuth(dAzimuth);
078        this.setSemiMajor(dMajorRadius);
079        this.setSemiMinor(dMinorRadius);
080        this.setFillStyle(null);
081    }
082
083    /**
084     * This constructor creates an ellipse that encapsulates the given IGeoEllipse.
085     * @param oRenderable See {@link IGeoEllipse}
086     */
087    public Ellipse(IGeoEllipse oRenderable) {
088        super(oRenderable, FeatureTypeEnum.GEO_ELLIPSE);
089
090        if(null == oRenderable) {
091            throw new InvalidParameterException("Encapsulated GeoEllipse must be non-null");
092        }
093        this.setSemiMajor(makePositive(this.getSemiMajor(), "Invalid Semi Major NaN"));
094        this.setSemiMinor(makePositive(this.getSemiMinor(), "Invalid Semi Minor NaN"));
095
096        if((this.getSemiMajor() < MINIMUM_SEMI_MAJOR) || (this.getSemiMinor() < MINIMUM_SEMI_MINOR)) {
097            throw new InvalidParameterException("Invalid Major or Minor. " + this.getSemiMajor() + " " + this.getSemiMinor() +
098                    " Minimum supported " + MINIMUM_SEMI_MAJOR + " " + MINIMUM_SEMI_MINOR);
099        }
100
101        this.setAzimuth(this.getAzimuth());  // For validation
102    }
103
104    /**
105     * This method sets the major radius of an elliptical feature.
106     * @param value The major radius in meters. The absolute value must be >= 1.0. Otherwise an InvalidParameter is raised.
107     */
108    @Override
109    public void setSemiMajor(double value) {
110        value = makePositive(value, "Semi Major is NaN");
111
112        if (value < MINIMUM_SEMI_MAJOR) {
113            throw new InvalidParameterException("Semi Major must be >= 1.0");
114        }
115        this.getRenderable().setSemiMajor(value);
116    }
117
118    /**
119     * This method retrieves the major radius value.
120     * @return The major radius value.
121     */
122    @Override
123    public double getSemiMajor() {
124        return this.getRenderable().getSemiMajor();
125    }
126
127    /**
128     * This method sets minor radius of an elliptical feature.
129     * @param value The minor radius in meters. An InvalidPaameterException is raised if the absolute value is less than 1.0.
130     */
131    @Override
132    public void setSemiMinor(double value) {
133        value = makePositive(value, "Semi Minor is NaN");
134
135        if (value < MINIMUM_SEMI_MINOR) {
136            throw new InvalidParameterException("The Semi Minor must be >= 1.0.");
137        }
138        this.getRenderable().setSemiMinor(value);
139    }
140
141    /**
142     * This method retrieves the minor radius of an elliptical feature
143     * @return The minor radius.
144     */
145    @Override
146    public double getSemiMinor() {
147        return this.getRenderable().getSemiMinor();
148    }

150    public IEmpBoundingBox getFeatureBoundingBox() {
151        IEmpBoundingBox bBox = null;
152        List<IGeoPosition> posList = getPositions();
153
154        if ((null != posList) && !posList.isEmpty()) {
155            double majorAxis = this.getSemiMajor();
156            double minorAxis = this.getSemiMinor();
157
158            if (!Double.isNaN(majorAxis) && !Double.isNaN(minorAxis)) {
159                double angle = this.getAzimuth();
160                bBox = new EmpBoundingBox();
161                IGeoPosition pos = new GeoPosition();
162
163                // Compute north.
164                GeoLibrary.computePositionAt(angle, minorAxis, posList.get(0), pos);
165                bBox.includePosition(pos.getLatitude(), pos.getLongitude());
166
167                // Compute east.
168                angle = this.getAzimuth() + 90.0;
169                angle = (((angle + 360.0) % 360.0) + 360.0) % 360.0;
170                GeoLibrary.computePositionAt(angle, majorAxis, posList.get(0), pos);
171                bBox.includePosition(pos.getLatitude(), pos.getLongitude());
172
173                // Compute south.
174                angle = this.getAzimuth() + 180.0;
175                angle = (((angle + 360.0) % 360.0) + 360.0) % 360.0;
176                GeoLibrary.computePositionAt(angle, minorAxis, posList.get(0), pos);
177                bBox.includePosition(pos.getLatitude(), pos.getLongitude());
178
179                // Compute west.
180                angle = this.getAzimuth() + 270.0;
181                angle = (((angle + 360.0) % 360.0) + 360.0) % 360.0;
182                GeoLibrary.computePositionAt(angle, majorAxis, posList.get(0), pos);
183                bBox.includePosition(pos.getLatitude(), pos.getLongitude());
184
185                // Now we need to extend the box by ~ 10%.
186                double deltaLat = bBox.deltaLatitude();
187                double deltaLong = bBox.deltaLongitude();
188
189                if (deltaLat == 0.0) {
190                    deltaLat = 0.05;
191                }
192                if (deltaLong == 0.0) {
193                    deltaLong = 0.05;
194                }
195
196                deltaLat *= 0.05;
197                deltaLong *= 0.05;
198
199                bBox.includePosition(bBox.getNorth() + deltaLat, bBox.getWest());
200                bBox.includePosition(bBox.getSouth() - deltaLat, bBox.getWest());
201                bBox.includePosition(bBox.getNorth(), bBox.getWest() - deltaLong);
202                bBox.includePosition(bBox.getNorth(), bBox.getEast() + deltaLong);
203            }
204        }
205        return bBox;
206    }
