033     * @param Latitude

036    public static double getEarthRadiusAt(double Latitude) {
037        int newLat = (int) (Math.round(Math.abs(Latitude)) % 90);

042    /**
043     * This method computes the ground (arc) distance between the two locations provided.
044     * @param oLatLon1
045     * @param oLatLon2
046     * @return The distance in meters.
047     * @throws InvalidParameterException
048     */
049    public static double computeDistanceBetween(IGeoPosition oLatLon1, IGeoPosition oLatLon2) throws InvalidParameterException {
050
051        if (oLatLon1 == null) {
052            throw new InvalidParameterException("oLatLon1 can not be null.");
053        }
054
055        if (oLatLon2 == null) {
056            throw new InvalidParameterException("oLatLon2 can not be null.");
057        }
058
059        double dLat1Rad = Math.toRadians(oLatLon1.getLatitude());
060        double dLon1Rad = Math.toRadians(oLatLon1.getLongitude());
061        double dLat2Rad = Math.toRadians(oLatLon2.getLatitude());
062        double dLon2Rad = Math.toRadians(oLatLon2.getLongitude());
063        double dDeltaLat = dLat2Rad - dLat1Rad;
064        double dDeltaLon = dLon2Rad - dLon1Rad;
065
066        double a = Math.sin(dDeltaLat / 2.0) * Math.sin(dDeltaLat / 2.0) +
067                Math.cos(dLat1Rad) * Math.cos(dLat2Rad) *
068                        Math.sin(dDeltaLon / 2.0) * Math.sin(dDeltaLon / 2.0);
069        double c = 2.0 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
070        return c * GeoLibrary.getEarthRadiusAt(oLatLon1.getLatitude());
071    }

073    /**
074     * This method return the bearing (from north) of a line from oFrom to oTo.
075     * @param oFrom
076     * @param oTo
077     * @return the bearing is return in degrees.
078     * @throws InvalidParameterException
079     */
080    public static double computeBearing(IGeoPosition oFrom, IGeoPosition oTo) throws InvalidParameterException {
081
082        if (oFrom == null) {
083            throw new InvalidParameterException("oFrom can not be null.");
084        }
085
086        if (oTo == null) {
087            throw new InvalidParameterException("oTo can not be null.");
088        }
089
090        double dLat1Rad = Math.toRadians(oFrom.getLatitude());
091        double dLat2Rad = Math.toRadians(oTo.getLatitude());
092        double dDeltaLon = Math.toRadians(oTo.getLongitude() - oFrom.getLongitude());
093        double y = Math.sin(dDeltaLon) * Math.cos(dLat2Rad);
094        double x = Math.cos(dLat1Rad) * Math.sin(dLat2Rad) - Math.sin(dLat1Rad) * Math.cos(dLat2Rad) * Math.cos(dDeltaLon);
095        double brng = Math.toDegrees(Math.atan2(y, x));
096
097        return (brng + 360.0) % 360.0;
098    }

116    /**
117     * This method computes a location which is at the specified bearing at the specified distance from the
118     * provided oFrom location.
119     * @param dBearing The bearing in degrees
120     * @param dDistance The distance in meters.
121     * @param oFrom The starting position
122     * @param result the object to place the resulting position.
123     * @throws InvalidParameterException
124     */
125    public static void computePositionAt(double dBearing, double dDistance, IGeoPosition oFrom, IGeoPosition result) throws InvalidParameterException {
126        if (oFrom == null) {
127            throw new InvalidParameterException("oFrom can not be null.");
128        }
129        if (result == null) {
130            throw new InvalidParameterException("result must be provided.");
131        }
132
133        double dAngularDistance = dDistance / GeoLibrary.getEarthRadiusAt(oFrom.getLatitude()); //DEFAULT_EARTH_RADIUS;
134        double dBearingRad = Math.toRadians(dBearing);
135        double dLat1Rad = Math.toRadians(oFrom.getLatitude());
136        double dLon1Rad = Math.toRadians(oFrom.getLongitude());
137        double dLat2Rad = Math.asin(Math.sin(dLat1Rad) * Math.cos(dAngularDistance) +
138                Math.cos(dLat1Rad) * Math.sin(dAngularDistance) * Math.cos(dBearingRad));
139        double dLon2Rad = dLon1Rad + Math.atan2(Math.sin(dBearingRad) * Math.sin(dAngularDistance) * Math.cos(dLat1Rad),
140                Math.cos(dAngularDistance) - Math.sin(dLat1Rad) * Math.sin(dLat2Rad));
141
142        dLon2Rad = (dLon2Rad + 3 * Math.PI) % (2 * Math.PI) - Math.PI;
143        result.setLatitude(Math.toDegrees(dLat2Rad));
144        result.setLongitude(Math.toDegrees(dLon2Rad));
145    }

147    public static IGeoPosition midPointBetween(IGeoPosition oLocation1, IGeoPosition oLocation2) throws InvalidParameterException {
148        if (oLocation1 == null) {
149            throw new InvalidParameterException("oLocation1 can not be null.");
150        }
151
152        if (oLocation2 == null) {
153            throw new InvalidParameterException("oLocation2 can not be null.");
154        }
155        IGeoPosition oNewPos = new GeoPosition();
156
157        double dLat1Rad = Math.toRadians(oLocation1.getLatitude());
158        double dLon1Rad = Math.toRadians(oLocation1.getLongitude());
159        double dLat2Rad = Math.toRadians(oLocation2.getLatitude());
160        double dDeltaLon = Math.toRadians(oLocation2.getLongitude() - oLocation1.getLongitude());
161
162        double Bx = Math.cos(dLat2Rad) * Math.cos(dDeltaLon);
163        double By = Math.cos(dLat2Rad) * Math.sin(dDeltaLon);
164
165        double dLat3Rad = Math.atan2(Math.sin(dLat1Rad) + Math.sin(dLat2Rad),
166                Math.sqrt((Math.cos(dLat1Rad) + Bx) * (Math.cos(dLat1Rad) + Bx) + By * By));
167        double dLon3Rad = dLon1Rad + Math.atan2(By, Math.cos(dLat1Rad) + Bx);
168        dLon3Rad = (dLon3Rad + 3 * Math.PI) % (2 * Math.PI) - Math.PI;
169
170        oNewPos.setLatitude(Math.toDegrees(dLat3Rad));
171        oNewPos.setLongitude(Math.toDegrees(dLon3Rad));
172
173        return oNewPos;
174    }

176    /**
177     * This method compute the rhumb distance between to points.
178     * @param oFrom
179     * @param oTo
180     * @return The distance in meters.
181     * @throws InvalidParameterException
182     */
183    public static double computeRhumbDistance(IGeoPosition oFrom, IGeoPosition oTo) throws InvalidParameterException {
184        if (oFrom == null) {
185            throw new InvalidParameterException("oFrom can not be null.");
186        }
187
188        if (oTo == null) {
189            throw new InvalidParameterException("oTo can not be null.");
190        }
191        double lat1 = Math.toRadians(oFrom.getLatitude()),
192               lat2 = Math.toRadians(oTo.getLatitude());
193        double dLat = Math.toRadians(oTo.getLatitude() - oFrom.getLatitude());
194        double dLon = Math.toRadians(Math.abs(oTo.getLongitude() - oFrom.getLongitude()));
195
196        double dPhi = Math.log(Math.tan(lat2 / 2 + Math.PI / 4) / Math.tan(lat1 / 2 + Math.PI / 4));
197        double q = (!Double.isNaN(dLat / dPhi) && (dPhi != 0)) ? dLat / dPhi : Math.cos(lat1); // E-W line gives dPhi=0
198        // if dLon over 180° take shorter rhumb across 180° meridian:
199        if (dLon > Math.PI) {
200            dLon = 2 * Math.PI - dLon;
201        }
202        double dist = Math.sqrt(dLat * dLat + q * q * dLon * dLon) * GeoLibrary.getEarthRadiusAt(oFrom.getLatitude()); //DEFAULT_EARTH_RADIUS;
203
204        return dist;
205    }

207    /**
208     * This method calculates the rhumb bearing of the line starting at oFrom and ending at oTo.
209     * @param oFrom
210     * @param oTo
211     * @return The bearing is return in degrees.
212     * @throws InvalidParameterException
213     */
214    public static double computeRhumbBearing(IGeoPosition oFrom, IGeoPosition oTo) throws InvalidParameterException {
215        if (oFrom == null) {
216            throw new InvalidParameterException("oFrom can not be null.");
217        }
218
219        if (oTo == null) {
220            throw new InvalidParameterException("oTo can not be null.");
221        }
222        double lat1 = Math.toRadians(oFrom.getLatitude()),
223               lat2 = Math.toRadians(oTo.getLatitude());
224        double dLon = Math.toRadians(oTo.getLongitude() - oFrom.getLongitude());
225
226        double dPhi = Math.log(Math.tan(lat2 / 2 + Math.PI / 4) / Math.tan(lat1 / 2 + Math.PI / 4));
227        if (Math.abs(dLon) > Math.PI) {
228            dLon = dLon > 0 ? -(2 * Math.PI - dLon) : (2 * Math.PI + dLon);
229        }
230        double brng = Math.atan2(dLon, dPhi);
231
232        return (Math.toDegrees(brng) + 360) % 360;
233    }

235    /**
236     * This method calculates a point at a specified distance at a specified angle from the specified position.
237     * @param dBearing The bearing in degrees
238     * @param dDistance The distance in meters.
239     * @param oFrom
240     * @return Geo position
241     * @throws InvalidParameterException
242     */
243    public static IGeoPosition calculateRhumbPositionAt(double dBearing, double dDistance, IGeoPosition oFrom) throws InvalidParameterException {
244        if (oFrom == null) {
245            throw new InvalidParameterException("oFrom can not be null.");
246        }
247        double d = dDistance / GeoLibrary.getEarthRadiusAt(oFrom.getLatitude()); // DEFAULT_EARTH_RADIUS; // d = angular distance covered on earth's surface
248        double lat1 = Math.toRadians(oFrom.getLatitude()),
249               lon1 = Math.toRadians(oFrom.getLongitude()),
250               lon2;
251        double brng = Math.toRadians(dBearing);
252
253        double lat2 = lat1 + d * Math.cos(brng);
254        double dLat = lat2 - lat1;
255        double dPhi = Math.log(Math.tan(lat2 / 2 + Math.PI / 4) / Math.tan(lat1 / 2 + Math.PI / 4));
256        double q = (!Double.isNaN(dLat / dPhi) && (dPhi != 0)) ? dLat / dPhi : Math.cos(lat1); // E-W line gives dPhi=0
257        double dLon = d * Math.sin(brng) / q;
258        // check for some daft bugger going past the pole
259        if (Math.abs(lat2) > Math.PI / 2) {
260            lat2 = lat2 > 0 ? Math.PI - lat2 : -(Math.PI - lat2);
261        }
262        lon2 = (lon1 + dLon + 3 * Math.PI) % (2 * Math.PI) - Math.PI;
263        IGeoPosition oNewPos = new GeoPosition();
264
265        oNewPos.setLatitude(Math.toDegrees(lat2));
266        oNewPos.setLongitude(Math.toDegrees(lon2));
267
268        return oNewPos;
269    }

298    /**
299     * Uses the cross product to determine if a point is on the left side of
300     * a line or not.
301     *
302     * @param p1 the first point of the line
303     * @param p2 the second point of the line
304     * @param p3 the point to test
305     *
306     * @return true if p3 is on the left side of the line, false otherwise
307     */
308    public static boolean isLeft(IGeoPosition p1, IGeoPosition p2, IGeoPosition p3) {
309        double lat1 = Math.toRadians(p1.getLatitude());
310        double lon1 = Math.toRadians(p1.getLongitude());
311
312        double lat2 = Math.toRadians(p2.getLatitude());
313        double lon2 = Math.toRadians(p2.getLongitude());
314
315        double lat3 = Math.toRadians(p3.getLatitude());
316        double lon3 = Math.toRadians(p3.getLongitude());
317
318        return ((lat2 - lat1) * (lon3 - lon1) - (lon2 - lon1) * (lat3 - lat1)) > 0;
319    }

321    /**
322     * http://www.movable-type.co.uk/scripts/latlong.html
323     * Returns the point of intersection of two paths defined by point and bearing.
324     *
325     * @param   {LatLon} p1 - First point.
326     * @param   {number} brng1 - Initial bearing from first point.
327     * @param   {LatLon} p2 - Second point.
328     * @param   {number} brng2 - Initial bearing from second point.
329     * @returns {LatLon|null} Destination point (null if no unique intersection defined).
330     *
331     * @example
332     *     var p1 = LatLon(51.8853, 0.2545), brng1 = 108.547;
333     *     var p2 = LatLon(49.0034, 2.5735), brng2 =  32.435;
334     *     var pInt = LatLon.intersection(p1, brng1, p2, brng2); // 50.9078°N, 004.5084°E
335     */
336    public static IGeoPosition intersection(IGeoPosition p1, double brng1, IGeoPosition p2, double brng2) {
337
338        // see http://williams.best.vwh.net/avform.htm#Intersection
339
340        double phi1 = Math.toRadians(p1.getLatitude());
341        double lambda1 = Math.toRadians(p1.getLongitude());
342        double phi2 = Math.toRadians(p2.getLatitude());
343        double lambda2 = Math.toRadians(p2.getLongitude());
344        
345        double theta13 = Math.toRadians(brng1);
346        double theta23 = Math.toRadians(brng2);
347        double deltaPhi = phi2-phi1;
348        double deltaLambda = lambda2-lambda1;
349
350        double daba12 = 2*Math.asin( Math.sqrt( Math.sin(deltaPhi/2)*Math.sin(deltaPhi/2)
351                + Math.cos(phi1)*Math.cos(phi2)*Math.sin(deltaLambda/2)*Math.sin(deltaLambda/2) ) );
352        if (daba12 == 0) return null;
353
354        // initial/final bearings between points
355        double thetaa = Math.acos( ( Math.sin(phi2) - Math.sin(phi1)*Math.cos(daba12) ) / ( Math.sin(daba12)*Math.cos(phi1) ) );
356        if (isNaN(thetaa)) thetaa = 0; // protect against rounding
357        double thetab = Math.acos( ( Math.sin(phi1) - Math.sin(phi2)*Math.cos(daba12) ) / ( Math.sin(daba12)*Math.cos(phi2) ) );
358
359        double theta12 = Math.sin(lambda2-lambda1)>0 ? thetaa : 2*Math.PI-thetaa;
360        double theta21 = Math.sin(lambda2-lambda1)>0 ? 2*Math.PI-thetab : thetab;
361
362        double alpha1 = (theta13 - theta12 + Math.PI) % (2*Math.PI) - Math.PI; // angle 2-1-3
363        double alpha2 = (theta21 - theta23 + Math.PI) % (2*Math.PI) - Math.PI; // angle 1-2-3
364
365        if (Math.sin(alpha1)==0 && Math.sin(alpha2)==0) return null; // infinite intersections
366        if (Math.sin(alpha1)*Math.sin(alpha2) < 0) return null;      // ambiguous intersection
367
368        //alpha1 = Math.abs(alpha1);
369        //alpha2 = Math.abs(alpha2);
370        // ... Ed Williams takes abs of alpha1/alpha2, but seems to break calculation?
371
372        double alpha3 = Math.acos( -Math.cos(alpha1)*Math.cos(alpha2) + Math.sin(alpha1)*Math.sin(alpha2)*Math.cos(daba12) );
373        double daba13 = Math.atan2( Math.sin(daba12)*Math.sin(alpha1)*Math.sin(alpha2), Math.cos(alpha2)+Math.cos(alpha1)*Math.cos(alpha3) );
374        double phi3 = Math.asin( Math.sin(phi1)*Math.cos(daba13) + Math.cos(phi1)*Math.sin(daba13)*Math.cos(theta13) );
375        double dabalambda13 = Math.atan2( Math.sin(theta13)*Math.sin(daba13)*Math.cos(phi1), Math.cos(daba13)-Math.sin(phi1)*Math.sin(phi3) );
376        double lambda3 = lambda1 + dabalambda13;
377
378        IGeoPosition geoPosition = new GeoPosition();
379        geoPosition.setLatitude(Math.toDegrees(phi3));
380        geoPosition.setLongitude((Math.toDegrees(lambda3)+540)%360-180);
381        // return new (phi3.toDegrees(), (lambda3.toDegrees()+540)%360-180); // normalise to −180..+180°
382        return geoPosition;
383    }

385    /**
386     * Procedure:
387     *     Convert each lat/long pair into a unit-length 3D vector.
388     *     Sum each of those vectors
389     *     Normalise the resulting vector
390     *     Convert back to spherical coordinates
391     * @param positionList
392     * @return
393     */
394    public static IGeoPosition getCenter(List<IGeoPosition> positionList) {
395
396        IGeoPosition center = null;
397        if((null != positionList) && (0 != positionList.size())) {
398            center = new GeoPosition();
399            if(1 == positionList.size()) {
400                center.setLongitude(positionList.get(0).getLongitude());
401                center.setLatitude(positionList.get(0).getLatitude());
402            } else {
403                double x = 0;
404                double y = 0;
405                double z = 0;
406
407                for(IGeoPosition position: positionList)
408                {
409                    double latitude = position.getLatitude() * Math.PI / 180;
410                    double longitude = position.getLongitude() * Math.PI / 180;
411
412
413                    x += Math.cos(latitude) * Math.cos(longitude);
414                    y += Math.cos(latitude) * Math.sin(longitude);
415                    z += Math.sin(latitude);
416                }
417
418                x = x / positionList.size();
419                y = y / positionList.size();
420                z = z / positionList.size();
421
422                double centralLongitude = Math.atan2(y, x);
423                double centralSquareRoot = Math.sqrt(x * x + y * y);
424                double centralLatitude = Math.atan2(z, centralSquareRoot);
425
426                center.setLongitude(centralLongitude * 180 / Math.PI);
427                center.setLatitude(centralLatitude * 180 / Math.PI);
428
429                Log.d(TAG, "Center Lat/Lon " + center.getLatitude() + "/" + center.getLongitude());
430            }
431        }
432
433        return center;
434    }
