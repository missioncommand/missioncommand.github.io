001package mil.emp3.api.utils;
002
003import org.cmapi.primitives.GeoColor;
004import org.cmapi.primitives.GeoStrokeStyle;
005import org.cmapi.primitives.IGeoColor;
006import org.cmapi.primitives.IGeoFillStyle;
007import org.cmapi.primitives.IGeoStrokeStyle;
008
009/**
010 * Underlying CMAPI supports multiple style objects that controle the appearance of the features rendered on the map,
011 *     GeoStrokeStyle
012 *     GeoFillStyle
013 *     GeiLabelStyle
014 *
015 * EmpStyles is a utility class that provided convenience methods for building these objects and providing a String representation
016 * of their members.
017 */
018public class EmpStyles {
019
020    /**
021     * Convert GeoFillStyle to String
022     * @param geoFillStyle
023     * @return
024     */
025    public static String toString(IGeoFillStyle geoFillStyle) {
026        if(null == geoFillStyle) {
027            return "FillStyle ";
028        }
029        String fillColor = "FillStyle " + toString(geoFillStyle.getFillColor());
030        if(null != geoFillStyle.getFillPattern()) {
031            return fillColor + " Pattern " + geoFillStyle.getFillPattern() + " ";
032        }
033        return fillColor + " Pattern ";
034    }
035
036    /**
037     * Convert GeoColor to String
038     * @param geoColor
039     * @return
040     */
041    public static String toString(IGeoColor geoColor) {
042        if(null == geoColor) {
043            return "Color ";
044        } else {
045            return "Color R:" + geoColor.getRed() + " G:" + geoColor.getGreen() + " B:" + geoColor.getBlue() + " A:" + geoColor.getAlpha() + " ";
046        }
047    }
048
049    /**
050     * Convert Stroke Style to String
051     * @param geoStrokeStyle
052     * @return
053     */
054    public static String toString(IGeoStrokeStyle geoStrokeStyle) {
055        if(null == geoStrokeStyle) {
056            return "StrokeStyle ";
057        }
058
059        String strokeColor = "StrokeStyle " + toString(geoStrokeStyle.getStrokeColor());
060        return strokeColor + "W:" + geoStrokeStyle.getStrokeWidth() + " SF:" + geoStrokeStyle.getStipplingFactor() +
061                " SP:" + Integer.toBinaryString(geoStrokeStyle.getStipplingPattern()) + " ";
062        // When interpreting the stippling pattern output consider negative numbers and 2s complement.
063    }
064
065    /**
066     * Build a Stroke Style object
067     * @param red
068     * @param green
069     * @param blue
070     * @param alpha
071     * @param width
072     * @param stipplingFactor
073     * @param stipplingPattern
074     * @return
075     */
076    public static IGeoStrokeStyle buildStrokeStyle(int red, int green, int blue, double alpha, double width, int stipplingFactor, short stipplingPattern) {
077        IGeoStrokeStyle geoStrokeStyle = new GeoStrokeStyle();
078        IGeoColor geoColor = new GeoColor();
079        geoColor.setRed(red);
080        geoColor.setGreen(green);
081        geoColor.setBlue(blue);
082        geoStrokeStyle.setStrokeColor(geoColor);
083
084        geoStrokeStyle.setStrokeWidth(width);
085        geoStrokeStyle.setStipplingFactor(stipplingFactor);
086        geoStrokeStyle.setStipplingPattern(stipplingPattern);
087        return geoStrokeStyle;
088    }
089}
