001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004
005/**
006 * Types of visibility actions that can be applied to an object.
007 */
008public enum VisibilityActionEnum implements IEventEnum {
009    /**
010     * Visibility state of the object and all its descendants is set to HIDDEN
011     */
012    HIDE_ALL,
013
014    /**
015     * Visibility state of the object and all its descendants is set to VISIBLE.
016     * <p>
017     * Internal:
018     * If the object has any parents that are HIDDEN, the parents must be toggled to VISIBLE all the way up the hierarchy.
019     * This should behave as defined in the TOGGLE_ON action
020     * </p>
021     */
022    SHOW_ALL,
023
024    /**
025     * Visibility state of the objects is set to VISIBLE.
026     * <p>
027     * Internal:
028     * For each descendant if the visibility state is VISIBLE_ANCESTOR_HIDDEN set state to VISIBLE, else if child state is HIDDEN leave as HIDDEN.
029     * If the object has any parents that are HIDDEN, the parents must be toggled to VISIBLE all the way up the hierarchy.
030     * This should behave as defined in the TOGGLE_ON action
031     * </p>
032    */
033    TOGGLE_ON,
034
035    /**
036     *  Visibility state of the object is set to HIDDEN.
037     *  <p>
038     * Internal:
039     * For each descendant if the visibility state is VISIBLE set state to VISIBLE_ANCESTOR_HIDDEN, else if child state is HIDDEN leave as HIDDEN.
040     * </p>
041    */
042    TOGGLE_OFF
043}
