025/**
026 * Container class serves as a base class for the Map objects including Map, Overlay and Feature. It implements some basic capabilities
027 * related to children, listeners and properties.
028 *
029 * A note on memory management: When an application passes a reference to a Container to the EMP via any of interface methods, the EMP Core doesn't
030 * clone the object, it uses and stores the reference to the object in its internal structures. Applications need to be careful when making changes
031 * to these objects. In many cases 'apply' methods are provided in the interface to allow the applications to control when a change is reflected
032 * on the map.
033 */
034public class Container implements IContainer{
035
036    final protected IStorageManager storageManager = ManagerFactory.getInstance().getStorageManager();
037    final protected IEventManager eventManager = ManagerFactory.getInstance().getEventManager();
038    final private IGeoContainer geoContainer;
039
040    protected IGeoContainer getGeoContainer() { return geoContainer; }
041
042    protected Container(IGeoContainer geoContainer) {
043        this.geoContainer = geoContainer;
044    }
045
046    /**
047     * Set or reset the flag that indicates if Container should be treated as read only.
048     * Application or EMP API may choose to mark this Container as read only. There is no mechanism for enforcing this read only restriction,
049     * it is just a hint.
050     * @param var1
051     */
052    @Override
053    public void setReadOnly(boolean var1) {
054        this.geoContainer.setReadOnly(var1);
055    }
056
057    /**
058     * Returns current value of read/write flag for the container.
059     * @return
060     */
061    @Override
062    public boolean getReadOnly() {
063        return this.geoContainer.getReadOnly();
064    }
065
066    @Override
067    public List<IContainer> getParents() {
068        return storageManager.getParents(this);
069    }
070
071    @Override
072    public void clearContainer()
073            throws EMP_Exception {
074        storageManager.removeChildren(this);
075    }
076
077    @Override
078    public boolean hasChildren() {
079        List<IGeoBase> oChildren = getChildren();
080        return !((oChildren == null) || (oChildren.size() == 0));
081    }
082
083    /**
084     * Returns immediate children of the Container.
085     * @return
086     */
087    @Override
088    public List<IGeoBase> getChildren() {
089        return storageManager.getImmediateChildren(this);
090    }
091
092    @Override
093    public EventListenerHandle addContainerEventListener(IContainerEventListener listener) throws EMP_Exception {
094        return eventManager.addEventHandler(EventListenerTypeEnum.CONTAINER_EVENT_LISTENER, this, listener);
095    }
096
097    @Override
098    public EventListenerHandle addVisibilityEventListener(IVisibilityEventListener listener) throws EMP_Exception {
099        return eventManager.addEventHandler(EventListenerTypeEnum.VISIBILITY_EVENT_LISTENER, this, listener);
100    }
101
102    @Override
103    public EventListenerHandle addFeatureEventListener(IFeatureEventListener listener) throws EMP_Exception {
104        return eventManager.addEventHandler(EventListenerTypeEnum.FEATURE_EVENT_LISTENER, this, listener);
105    }
106
107    @Override
108    public EventListenerHandle addFeatureInteractionEventListener(IFeatureInteractionEventListener listener) throws EMP_Exception {
109        return eventManager.addEventHandler(EventListenerTypeEnum.FEATURE_INTERACTION_EVENT_LISTENER, this, listener);
110    }
111
112    @Override
113    public EventListenerHandle addFeatureEditEventListener(IFeatureEditEventListener listener) throws EMP_Exception {
114        return eventManager.addEventHandler(EventListenerTypeEnum.FEATURE_EDIT_EVENT_LISTENER, this, listener);
115    }
116
117    @Override
118    public void removeEventListener(EventListenerHandle hListener) {
119        eventManager.removeEventHandler(hListener);
120    }
121
122    /**
123     * Each Container has user specified name. There is no check in the EMP Core for duplicate names or any sort of validation.
124     *
125     * For icon based MIL-STD-2525 point features the name will display if populated.  If the uniqueDesignation modifier is also populated
126     * and labels are turned on both will display with the name to the right of the uniqueDesignation.
127     * This is based on behavior we had to be compliant with legacy products.
128     * @param s
129     */
130    @Override
131    public void setName(String s) {
132        this.geoContainer.setName(s);
133    }
134
135    /**
136     * Returns previously assigned name of the container.
137     * @return
138     */
139    @Override
140    public String getName() {
141        return this.geoContainer.getName();
142    }
143
144    /**
145     * Sets GeoId of the Container. This will be removed once a constructor is created in GeoBase class. EMP-2995
146     * @param uuid
147     */
148    @Override
149    public void setGeoId(java.util.UUID uuid) {
150        this.geoContainer.setGeoId(uuid);
151        // throw new IllegalStateException("Application is not allowed to change the GeoId");
152    }
153
154    /**
155     * Returns GeoId of a container. GeoId is cloned before handing it off to the application.
156     * @return
157     */
158    @Override
159    public UUID getGeoId() {
160        UUID geoId = new UUID(this.geoContainer.getGeoId().getMostSignificantBits(), this.geoContainer.getGeoId().getLeastSignificantBits());
161        return geoId;
162    }
163
164    /**
165     * Application can set the data provider Id for its own use. This attribute is not processed by EMP
166     * @param s
167     */
168    @Override
169    public void setDataProviderId(String s) {
170        this.geoContainer.setDataProviderId(s);
171    }
172
173    /**
174     * Get the data provider id.
175     * @return
176     */
177    @Override
178    public String getDataProviderId() {
179        return this.geoContainer.getDataProviderId();
180    }
181
182    /**
183     * Sets the description of the Container.
184     * Description is typically used for things like the pop up window that may display when a feature is clicked on by user.
185     * This behavior is completely up to app developer though.  Description may contain numerous lines of text.
186     * @param s
187     */
188    @Override
189    public void setDescription(String s) {
190        this.geoContainer.setDescription(s);
191    }
192
193    /**
194     * Gets the description of the Container.
195     * @return
196     */
197    @Override
198    public String getDescription() {
199        return this.geoContainer.getDescription();
200    }
201
202    /**
203     * Returns container properties.
204     * @return
205     */
206    @Override
207    public HashMap<String, String> getProperties() {
208        return this.geoContainer.getProperties();
209    }
210
211    @Override
212    public boolean containsProperty(String propertyName) {
213        return (this.getProperties().containsKey(propertyName.toUpperCase()));
214    }
215
216    @Override
217    public String getProperty(String propertyName) {
218        if (!this.containsProperty(propertyName)) {
219            return null;
220        }
221
222        return this.getProperties().get(propertyName.toUpperCase());
223    }
224
225    @Override
226    public void setProperty(String propertyName, String propertyValue) {
227        this.getProperties().put(propertyName.toUpperCase(), propertyValue);
228    }
229
230    @Override
231    public void removeProperty(String propertyName) {
232        if (this.containsProperty(propertyName.toUpperCase())) {
233            this.getProperties().remove(propertyName.toUpperCase());
234        }
235    }
236
237    @Override
238    public boolean getBooleanProperty(String propertyName) {
239        if (!this.containsProperty(propertyName)) {
240            throw new InvalidParameterException("Property does not exists.");
241        }
242
243        String strValue = this.getProperty(propertyName);
244
245        if ((null == strValue) || strValue.isEmpty() || !(strValue.equalsIgnoreCase("true") || strValue.equalsIgnoreCase("false"))) {
246            throw new InvalidParameterException("Property value is not a boolean.");
247        }
248
249        return Boolean.parseBoolean(strValue);
250    }
251
252    @Override
253    public int getIntegerProperty(String propertyName) {
254        int retValue;
255
256        if (!this.containsProperty(propertyName)) {
257            throw new InvalidParameterException("Property does not exists.");
258        }
259
260        String strValue = this.getProperty(propertyName);
261
262        if ((null == strValue) || strValue.isEmpty()) {
263            throw new InvalidParameterException("Property value is not an integer.");
264        }
265
266        try {
267            retValue = Integer.parseInt(strValue);
268        } catch (NumberFormatException Ex) {
269            throw new InvalidParameterException("Property value is not an integer.");
270        }
271
272        return retValue;
273    }
274
275    @Override
276    public float getFloatProperty(String propertyName) {
277        float retValue;
278
279        if (!this.containsProperty(propertyName)) {
280            throw new InvalidParameterException("Property does not exists.");
281        }
282
283        String strValue = this.getProperty(propertyName);
284
285        if ((null == strValue) || strValue.isEmpty()) {
286            throw new InvalidParameterException("Property value is not a float.");
287        }
288
289        try {
290            retValue = Float.parseFloat(strValue);
291        } catch (NumberFormatException Ex) {
292            throw new InvalidParameterException("Property value is not an float.");
293        }
294
295        return retValue;
296    }
297
298    @Override
299    public double getDoubleProperty(String propertyName) {
300        double retValue;
301
302        if (!this.containsProperty(propertyName)) {
303            throw new InvalidParameterException("Property does not exists.");
304        }
305
306        String strValue = this.getProperty(propertyName);
307
308        if ((null == strValue) || strValue.isEmpty()) {
309            throw new InvalidParameterException("Property value is not a double.");
310        }
311
312        try {
313            retValue = Double.parseDouble(strValue);
314        } catch (NumberFormatException Ex) {
315            throw new InvalidParameterException("Property value is not an double.");
316        }
317
318        return retValue;
319    }
