001package mil.emp3.api.events;
002
003import mil.emp3.api.interfaces.ICamera;
004import mil.emp3.api.interfaces.IEmpBoundingArea;
005import mil.emp3.api.interfaces.ILookAt;
006import mil.emp3.api.interfaces.IMap;
007import mil.emp3.api.enums.MapViewEventEnum;
008import org.cmapi.primitives.IGeoBounds;
009
010/**
011 * This event class is generated when a maps view is  changes.
012 */
013public abstract class MapViewChangeEvent extends Event<MapViewEventEnum, IMap> {
014    protected MapViewChangeEvent(MapViewEventEnum eventEnum, IMap oMap) {
015        super(eventEnum, oMap);
016    }
017
018    /**
019     * This method return the new view parameters.
020     * @return @See ICamera
021     */
022    public abstract ICamera getCamera();
023
024    /**
025     * Gets the map bounds after the event. Returned value could be null based on current camera settings.
026     * If event is VIEW_IN_MOTION then getBounds will return null.
027     * @return bounds if available else null
028     */
029    public abstract IGeoBounds getBounds();
030
031    /**
032     * gets the map bounding area, if available, after Map motion has stopped.
033     * If event is VIEW_IN_MOTION then getBoundingArea will be null.
034     * @return boundingArea if available else null
035     */
036    public abstract IEmpBoundingArea getBoundingArea();
037
038    /**
039     * Gets the updated lookAt object after the event.
040     * @return @See ILookAt
041     */
042    public abstract ILookAt getLookAt();
043}
