001package mil.emp3.api.enums;
002
003/**
004 * Types of MilStd label settings. use setMilStdLabels method on {@link mil.emp3.api.abstracts.Map} set the label settings
005 */
006public enum MilStdLabelSettingEnum {
007    /**
008     * Map engine shall display required MilStd labels only.
009     * (V, L, S, AA, AB, AC)
010     */
011    REQUIRED_LABELS,
012    /**
013     * Map engine shall display the common MilStd labels in addition to the required labels.
014     * (H, M, T, T1, CN)
015     */
016    COMMON_LABELS,
017    /**
018     * Map engine shall display all MilStd labels in addition to the common labels.
019     * (C, F, G, H1, H2, J, K, N, P, W, W1, X, Y, Z)
020     */
021    ALL_LABELS
022}
