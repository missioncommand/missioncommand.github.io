001package mil.emp3.api;
002
003import mil.emp3.api.abstracts.Feature;
004import mil.emp3.api.enums.FeatureTypeEnum;
005
006import org.cmapi.primitives.GeoText;
007import org.cmapi.primitives.IGeoText;
008
009/**
010 * This class positions text on the map at a specific coordinate.
011 */
012public class Text extends Feature<IGeoText> implements IGeoText {
013
014    /**
015     * This constructor creates a Text feature with all default values.
016     */
017    public Text() {
018        super(new GeoText(), FeatureTypeEnum.GEO_TEXT);
019    }
020
021    /**
022     * Thsi constructor create a Text feature and sets the Text to the value provided.
023     * @param text
024     */
025    public Text(String text) {
026        super(new GeoText(), FeatureTypeEnum.GEO_TEXT);
027        this.setName(text);
028    }
029
030    /**
031     * This constructor createa a Text feature based on the GeoText object provided.
032     * @param geoText
033     */
034    public Text(IGeoText geoText) {
035        super(geoText, FeatureTypeEnum.GEO_TEXT);
036    }
037
038    /**
039     * This method retrieves the features text. It is equivalent to calling the object getName method.
040     * @return String
041     */
042    public String getText()
043    {
044        return this.getName();
045    }
046
047    /**
048     * This method sets the features text. It is equivalent to calling the object setName method.
049     * @param value the Text to set.
050     */
051    public void setText(String value) {
052        this.setName(value);
053    }
054
055    /**
056     * This method retrieves the text rotation angle.
057     * @return The angle in degrees.
058     */
059    public double getRotationAngle() {
060        return this.getAzimuth();
061    }
062
063    /**
064     * This method set the text rotation angle. The text will be rotated about the geographic coordinate
065     * point by the angle provided. A value of 0 degrees renders the text parallel to the latitudinal lines on the globe.
066     * A value of 90 degrees renders the text parallel to the longitudinal lines on the globe. the rotation
067     * is performed clockwise.
068     * @param value The rotation angle in degrees.
069     */
070    public void setRotationAngle(double value) {
071        this.setAzimuth(value);
072    }
073}
