004import org.cmapi.primitives.GeoBounds;
005import org.cmapi.primitives.IGeoBounds;

007import java.net.MalformedURLException;

016
017/**
018 * This class implements an image map service layer. It imposes an image on the map at a specified location
019 * bounded by the provided bounding box.
020 */
021
022public class ImageLayer extends MapService implements IImageLayer {
023    private final IGeoBounds boundingBox;
024
025    /**
026     * This constructor creates an Image Layer.
027     * @param urlString the URL the image is located at.
028     * @param bBox The bounding box where to locate the image on the map.
029     * @throws MalformedURLException If the URL is invalid.
030     */
031    public ImageLayer(String urlString, IGeoBounds bBox)
032            throws MalformedURLException {
033        super(urlString);
034        this.boundingBox = new GeoBounds();
035        this.boundingBox.setNorth(bBox.getNorth());
036        this.boundingBox.setSouth(bBox.getSouth());
037        this.boundingBox.setEast(bBox.getEast());
038        this.boundingBox.setWest(bBox.getWest());
039    }
040
041    @Override
042    public IGeoBounds getBoundingBox() {
043        return this.boundingBox;
044    }
045}
