001package mil.emp3.api.utils;
002
003import org.cmapi.primitives.IGeoColor;
004
005import mil.emp3.api.enums.MirrorCacheModeEnum;
006import mil.emp3.api.enums.Property;
007import mil.emp3.api.interfaces.IEmpPropertyList;
008
009/**
010 * This class implements a list of key, value pairs where the key is a string and the value is an Object.
011 */
012// java.utils.Properties only supports string values.
013public class EmpPropertyList extends java.util.HashMap<String, Object> implements IEmpPropertyList {
014
015    @Override
016    public boolean genExceptionIfNotPresent(String key) {
017        if (!this.containsKey(key)) {
018            throw new IllegalArgumentException("Property " + key + " is required.");
019        }
020        return true;
021    }
022
023    @Override
024    public String getStringValue(String key) {
025        if (this.containsKey(key)) {
026            if (!(this.get(key) instanceof String)) {
027                throw new IllegalArgumentException("Property " + key + " value must be a string.");
028            }
029            return (String) this.get(key);
030        }
031        return null;
032    }
033
034    @Override
035    public IGeoColor getColorValue(String key) {
036        if (this.containsKey(key)) {
037            if (!(this.get(key) instanceof IGeoColor)) {
038                throw new IllegalArgumentException("Property " + key + " value must be an IGeoColor object.");
039            }
040            return (IGeoColor) this.get(key);
041        }
042        return null;
043    }
044
045    @Override
046    public double getDoubleValue(String key) {
047        if (this.containsKey(key)) {
048            if (!(this.get(key) instanceof Double)) {
049                throw new IllegalArgumentException("Property " + key + " value must be a double.");
050            }
051            return (double) this.get(key);
052        }
053        return Double.NaN;
054    }
055
056    @Override
057    public mil.emp3.api.enums.MirrorCacheModeEnum getMirrorCacheModeEnum(String key) {
058        if (this.containsKey(key)) {
059            if (!(this.get(key) instanceof mil.emp3.api.enums.MirrorCacheModeEnum)) {
060                throw new IllegalArgumentException("Property " + key + " value must be a mil.emp3.api.enums.MirrorCacheModeEnum value.");
061            }
062            return (mil.emp3.api.enums.MirrorCacheModeEnum) this.get(key);
063        }
064        return null;
065    }
066
067    @Override
068    public android.content.Context getContext(String key) {
069        if (this.containsKey(key)) {
070            if (!(this.get(key) instanceof android.content.Context)) {
071                throw new IllegalArgumentException("Property " + key + " value must be a android.content.Context value.");
072            }
073            return (android.content.Context) this.get(key);
074        }
075        return null;
076    }
077}
