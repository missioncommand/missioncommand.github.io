001package mil.emp3.api.listeners;
002
003import java.util.List;
004
005import mil.emp3.api.interfaces.IEditUpdateData;
006import mil.emp3.api.interfaces.IFeature;
007import mil.emp3.api.interfaces.IMap;
008
009/**
010 * A class must implement this interface to process edit events.
011 */
012public interface IEditEventListener {
013    /**
014     * This method is called when the edit session begins.
015     * @param map This parameter indicates the map the event occurred on.
016     */
017    public void onEditStart(IMap map);
018    /**
019     * This method is called each time the feature is changed.
020     * @param map This parameter indicates the map the event occurred on.
021     * @param updateList This parameter contains a list of changes. see {@link IEditUpdateData}
022     */
023    public void onEditUpdate(IMap map, IFeature feature, List<IEditUpdateData> updateList);
024    /**
025     * This method is called when the edit session is finished.
026     * @param map This parameter indicates the map the event occurred on.
027     * @param feature This parameter is the newly edited feature.
028     */
029    public void onEditComplete(IMap map, IFeature feature);
030    /**
031     * This method is called when the edit session is canceled.
032     * @param map This parameter indicates the map the event occurred on.
033     * @param originalFeature This parameter is the feature as it was before the session started.
034     */
035    public void onEditCancel(IMap map, IFeature originalFeature);
036    /**
037     * This method is called if an error is encountered.
038     * @param map This parameter indicates the map the event occurred on.
039     * @param errorMessage A text description of the error.
040     */
041    public void onEditError(IMap map, String errorMessage);
042}
