001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoContainer;
004
005import java.util.List;
006
007import mil.emp3.api.exceptions.EMP_Exception;
008
009/**
010 * This interface defines the the methods associated with an overlay.
011 */
012public interface IOverlay extends IContainer, IGeoContainer {
013    /**
014     * This method retrieves a list of this overlay's child overlays.
015     *
016     * @return An ArrayList of zero or more IOverlay objects.
017     */
018    public List<IOverlay> getOverlays();
019
020    /**
021     * This method adds a child overlay to this overlay.
022     *
023     * @param overlay The child overlay to add.
024     * @param visible True if the overlay is to be made visible, false otherwise.
025     * @throws EMP_Exception The exception is raised if a processing error is encountered.
026     */
027    public void addOverlay(IOverlay overlay, boolean visible)
028            throws EMP_Exception;
029
030    /**
031     * This method adds one or more child overlays to this overlay.
032     *
033     * @param overlays A list of child overlays to add.
034     * @param visible True if the overlays are to be made visible, false otherwise.
035     * @throws EMP_Exception The exception is raised if a processing error is encountered.
036     */
037    public void addOverlays(List<IOverlay> overlays, boolean visible)
038            throws EMP_Exception;
039
040    /**
041     * This method removes a child overlay from this overlay.
042     *
043     * @param overlay The child overlay to remove.
044     * @throws EMP_Exception The exception is raised if a processing error is encountered.
045     */
046    public void removeOverlay(IOverlay overlay)
047            throws EMP_Exception;
048
049    /**
050     * This method removes one or more child overlays from this overlay.
051     *
052     * @param overlays A list of child overlays to remove.
053     * @throws EMP_Exception The exception is raised if a processing error is encountered.
054     */
055    public void removeOverlays(List<IOverlay> overlays)
056            throws EMP_Exception;
057
058    /**
059     * This method adds a child overlay to this overlay.
060     *
061     * @param overlay The child overlay to add.
062     * @param visible True if the overlay is to be made visible, false otherwise.
063     * @param object user defined object
064     * @throws EMP_Exception The exception is raised if a processing error is encountered.
065     */
066    public void addOverlay(IOverlay overlay, boolean visible, Object object)
067            throws EMP_Exception;
068
069    /**
070     * This method adds one or more child overlays to this overlay.
071     *
072     * @param overlays A list of child overlays to add.
073     * @param visible True if the overlays are to be made visible, false otherwise.
074     * @param object user defined object
075     * @throws EMP_Exception The exception is raised if a processing error is encountered.
076     */
077    public void addOverlays(List<IOverlay> overlays, boolean visible, Object object)
078            throws EMP_Exception;
079
080    /**
081     * This method removes a child overlay from this overlay.
082     *
083     * @param overlay The child overlay to remove.
084     * @param object user defined object
085     * @throws EMP_Exception The exception is raised if a processing error is encountered.
086     */
087    public void removeOverlay(IOverlay overlay, Object object)
088            throws EMP_Exception;
089
090    /**
091     * This method removes one or more child overlays from this overlay.
092     *
093     * @param overlays A list of child overlays to remove.
094     * @param object user defined object
095     * @throws EMP_Exception The exception is raised if a processing error is encountered.
096     */
097    public void removeOverlays(List<IOverlay> overlays, Object object)
098            throws EMP_Exception;
099
100    /**
101     * This method retrieves all child features of this overlay.
102     * @return An ArrayList of IFeatures.
103     */
104    public List<IFeature> getFeatures();
105
106    /**
107     * This method adds a feature to this overlay.
108     * @param feature The feature to add
109     * @param visible True if the feature is to be made visible, false otherwise.
110     * @param object user defined object
111     * @throws EMP_Exception The exception is raised if a processing error is encountered.
112     */
113    public void addFeature(IFeature feature, boolean visible, Object object)
114            throws EMP_Exception;
115
116    /**
117     * This method adds one or more features to this overlay.
118     * @param features ArrayList of IFeatures to add.
119     * @param visible True if the features are to be made visible, false otherwise.
120     * @param object user defined object
121     * @throws EMP_Exception The exception is raised if a processing error is encountered.
122     */
123    public void addFeatures(List<IFeature> features, boolean visible, Object object)
124            throws EMP_Exception;
125
126    /**
127     * This method removes a child feature from this overlay.
128     *
129     * @param feature The child feature to remove.
130     * @param object user defined object
131     * @throws EMP_Exception The exception is raised if a processing error is encountered.
132     */
133    public void removeFeature(IFeature feature, Object object)
134            throws EMP_Exception;
135
136    /**
137     * This method removes one or more child features from this overlay.
138     *
139     * @param features A list of child features to remove.
140     * @param object user defined object
141     * @throws EMP_Exception The exception is raised if a processing error is encountered.
142     */
143    public void removeFeatures(List<IFeature> features, Object object)
144            throws EMP_Exception;
145
146    /**
147     * This method adds a feature to this overlay.
148     * @param feature The feature to add
149     * @param visible True if the feature is to be made visible, false otherwise.
150     * @throws EMP_Exception The exception is raised if a processing error is encountered.
151     */
152    public void addFeature(IFeature feature, boolean visible)
153            throws EMP_Exception;
154
155    /**
156     * This method adds one or more features to this overlay.
157     * @param features ArrayList of IFeatures to add.
158     * @param visible True if the features are to be made visible, false otherwise.
159     * @throws EMP_Exception The exception is raised if a processing error is encountered.
160     */
161    public void addFeatures(List<IFeature> features, boolean visible)
162            throws EMP_Exception;
163
164    /**
165     * This method removes a child feature from this overlay.
166     *
167     * @param feature The child feature to remove.
168     * @throws EMP_Exception The exception is raised if a processing error is encountered.
169     */
170    public void removeFeature(IFeature feature)
171            throws EMP_Exception;
172
173    /**
174     * This method removes one or more child features from this overlay.
175     *
176     * @param features A list of child features to remove.
177     * @throws EMP_Exception The exception is raised if a processing error is encountered.
178     */
179    public void removeFeatures(List<IFeature> features)
180            throws EMP_Exception;
181
182    /**
183     * This method triggers an update of the overlay. After changing any properties, this
184     * method MUST be called to force an update.
185     */
186    public void apply();
187}
