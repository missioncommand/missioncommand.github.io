001package mil.emp3.api.events;
002
003import mil.emp3.api.interfaces.IContainer;
004import mil.emp3.api.interfaces.IMap;
005import mil.emp3.api.enums.VisibilityActionEnum;
006
007/**
008 * This event is generated when an IOverlay or IFeature object's visibility is changed on a specific map.
009 * To register for this event you must instantiate an object that implements the IVisibilityEventListener
010 * interface and register for the event on an IMap, IOverlay or IFeature.
011 */
012public abstract class VisibilityEvent extends Event<VisibilityActionEnum, IContainer> {
013    protected VisibilityEvent(VisibilityActionEnum eEvent, IContainer oTarget) {
014        super(eEvent, oTarget);
015    }
016
017    /**
018     * This method retrieves the parent container the target visibility was changed under.
019     * @return 
020     * {@link IContainer} - The parent container if the visibility of the target was set via
021     * a call to {@link IMap#setVisibility(mil.emp3.api.interfaces.IContainer, mil.emp3.api.interfaces.IContainer, mil.emp3.api.enums.VisibilityActionEnum)} or
022     * {@link IMap#setVisibility(java.util.UUID, java.util.UUID, mil.emp3.api.enums.VisibilityActionEnum)}.
023     * <br/>null - otherwise.
024     */
025    public abstract IContainer getParent();
026
027    /**
028     * This method returns the interface of the map the visibility event was triggered on.
029     * @return IMap Interface object.
030     */
031    public abstract IMap getMapEventOccurredOn();
032}
