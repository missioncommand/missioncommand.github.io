016/**
017 * This static class implements many mil std utility functions.
018 */
019public class MilStdUtilities {
020    /**
021     * This method converts a IGeoMilSymbol.SymbolStandard enumerated value to a
022     * MilStd Renderer symbol version value.
023     * @param eStandard see IGeoMilSymbol.SymbolStandard.
024     * @return an integer value indicating the standard version.
025     */
026    public static int geoMilStdVersionToRendererVersion(IGeoMilSymbol.SymbolStandard eStandard) {
027        int iVersion = RendererSettings.Symbology_2525Bch2_USAS_13_14;
028        
029        switch (eStandard) {
030            case MIL_STD_2525C:
031                iVersion = RendererSettings.Symbology_2525C;
032                break;
033            case MIL_STD_2525B :
034                iVersion = RendererSettings.Symbology_2525Bch2_USAS_13_14;
035                break;
036        }
037        
038        return iVersion;
039    }
040
041    /**
042     * This method converts a Geo Altitude mode enum value to a MilStd renderer altitude mode.
043     * @param mode
044     * @return
045     */
046    public static String geoAltitudeModeToString(IGeoAltitudeMode.AltitudeMode mode) {
047        switch (mode) {
048            case CLAMP_TO_GROUND:
049                return "clampToGround";
050            case RELATIVE_TO_GROUND:
051                return "relativeToGround";
052            case ABSOLUTE:
053                return "absolute";
054        }
055        return "";
056    }
057
058    public static String getMilStdSinglePointIconURL(final MilStdSymbol feature,
059            MilStdLabelSettingEnum eLabelSetting, java.util.Set<IGeoMilSymbol.Modifier> labelSet,
060            int iIconSize, boolean selected, IGeoColor selectedStrokeColor, IGeoColor selectedTextColor) {
061        int iKey;
062        String value;
063        String UniqueDesignator1 = null;
064        String params = "";
065        java.util.HashMap<IGeoMilSymbol.Modifier, String> geoModifiers = feature.getModifiers();
066        SparseArray<String> attributes = feature.getAttributes(iIconSize,
067                selected, selectedStrokeColor, selectedTextColor);
068
069        if (null != attributes) {
070            for (int iIndex = 0; iIndex < attributes.size(); iIndex++) {
071                iKey = attributes.keyAt(iIndex);
072                value = attributes.valueAt(iIndex);
073                switch (iKey) {
074                    case MilStdAttributes.SymbologyStandard:
075                        if (!params.isEmpty()) {
076                            params += "&";
077                        }
078                        switch (value) {
079                            case "0":
080                                params += "symStd=2525B";
081                                break;
082                            case "1":
083                            default:
084                                params += "symStd=2525C";
085                                break;
086                        }
087                        break;
088                    case MilStdAttributes.PixelSize:
089                        if (!params.isEmpty()) {
090                            params += "&";
091                        }
092                        params += "Size=" + value;
093                        break;
094                    case MilStdAttributes.FillColor:
095                        if (!params.isEmpty()) {
096                            params += "&";
097                        }
098                        params += "fillColor=" + value;
099                        break;
100                    case MilStdAttributes.LineColor:
101                        if (!params.isEmpty()) {
102                            params += "&";
103                        }
104                        params += "lineColor=" + value;
105                        break;
106                    case MilStdAttributes.IconColor:
107                        if (!params.isEmpty()) {
108                            params += "&";
109                        }
110                        params += "lineColor=" + value;
111                        break;
112                    case MilStdAttributes.TextColor:
113                        if (!params.isEmpty()) {
114                            params += "&";
115                        }
116                        params += "textColor=" + value;
117                        break;
118                    case MilStdAttributes.FontSize:
119                        if (!params.isEmpty()) {
120                            params += "&";
121                        }
122                        params += "fontSize=" + value;
123                        break;
124                }
125            }
126        }
127        if ((geoModifiers != null) && !geoModifiers.isEmpty()) {
128            java.util.Set<IGeoMilSymbol.Modifier> oModifierList = geoModifiers.keySet();
129
130            for (IGeoMilSymbol.Modifier eModifier: oModifierList) {
131                if ((labelSet != null) && !labelSet.contains(eModifier)) {
132                    // Its not on the list.
133                    continue;
134                }
135                switch (eModifier) {
136                    case SYMBOL_ICON:
137                    case ECHELON:
138                    case QUANTITY:
139                    case TASK_FORCE_INDICATOR:
140                    case FRAME_SHAPE_MODIFIER:
141                    case REDUCED_OR_REINFORCED:
142                    case STAFF_COMMENTS:
143                    case ADDITIONAL_INFO_1:
144                    case ADDITIONAL_INFO_2:
145                    case ADDITIONAL_INFO_3:
146                    case EVALUATION_RATING:
147                    case COMBAT_EFFECTIVENESS:
148                    case SIGNATURE_EQUIPMENT:
149                    case HIGHER_FORMATION:
150                    case HOSTILE:
151                    case IFF_SIF:
152                    case DIRECTION_OF_MOVEMENT:
153                    case MOBILITY_INDICATOR:
154                    case SIGINT_MOBILITY_INDICATOR:
155                    case OFFSET_INDICATOR:
156                    case UNIQUE_DESIGNATOR_2:
157                    case EQUIPMENT_TYPE:
158                    case DATE_TIME_GROUP:
159                    case DATE_TIME_GROUP_2:
160                    case ALTITUDE_DEPTH:
161                    case LOCATION:
162                    case SPEED:
163                    case SPECIAL_C2_HEADQUARTERS:
164                    case FEINT_DUMMY_INDICATOR:
165                    case INSTALLATION:
166                    case PLATFORM_TYPE:
167                    case EQUIPMENT_TEARDOWN_TIME:
168                    case COMMON_IDENTIFIER:
169                    case AUXILIARY_EQUIPMENT_INDICATOR:
170                    case AREA_OF_UNCERTAINTY:
171                    case DEAD_RECKONING:
172                    case SPEED_LEADER:
173                    case PAIRING_LINE:
174                    case OPERATIONAL_CONDITION:
175                    case ENGAGEMENT_BAR:
176                    case COUNTRY_CODE:
177                    case SONAR_CLASSIFICATION_CONFIDENCE:
178                        if (!params.isEmpty()) {
179                            params += "&";
180                        }
181                        params += eModifier.valueOf() + "=" + geoModifiers.get(eModifier);
182                        break;
183                    case UNIQUE_DESIGNATOR_1:
184                        UniqueDesignator1 = geoModifiers.get(eModifier);
185                        if (!params.isEmpty()) {
186                            params += "&";
187                        }
188                        params += eModifier.valueOf() + "=" + geoModifiers.get(eModifier);
189                        break;
190                    case DISTANCE:
191                    case AZIMUTH:
192                        break;
193                }
194            }
195        }
196
197        if ((feature.getName() != null) && !feature.getName().isEmpty()) {
198            if (eLabelSetting != null) {
199                switch (eLabelSetting) {
200                    case REQUIRED_LABELS:
201                        break;
202                    case COMMON_LABELS:
203                    case ALL_LABELS:
204                        if ((UniqueDesignator1 == null) || UniqueDesignator1.isEmpty() || !UniqueDesignator1.toUpperCase().equals(feature.getName().toUpperCase())) {
205                            if (!params.isEmpty()) {
206                                params += "&";
207                            }
208                            params += "CN=" + feature.getName();
209                        }
210                        break;
211                }
212            }
213        }
214
215        return "http://localhost:8080/mil-sym-service/renderer/image/" + feature.getSymbolCode() + "?" + params;
216    }
