001package mil.emp3.api.enums;
002
003/**
004 * The font size modifier values that should applied to text during rendering of {@link mil.emp3.api.Text} and Tactical Graphic {@link mil.emp3.api.MilStdSymbol} feature.
005 */
006public enum FontSizeModifierEnum {
007    /**
008     * Text font size is reduced by 40% of the indicated size.
009     */
010    SMALLEST,
011    /**
012     * Text font size is reduced by 20% of the indicated size.
013     */
014    SMALLER,
015    /**
016     * Text is rendered in its indicated size. No size change is applied.
017     */
018    NORMAL,
019    /**
020     * Text font size is increased by 20% of its indicated value.
021     */
022    LARGER,
023    /**
024     * Text font size is increased by 40% of its indicated value.
025     */
026    LARGEST;
027
028    public float getScaleValue() {
029        switch (this) {
030            case SMALLEST:
031                return (float) 0.6;
032            case SMALLER:
033                return (float) 0.8;
034            case NORMAL:
035                return (float) 1.0;
036            case LARGER:
037                return (float) 1.2;
038            case LARGEST:
039                return (float) 1.4;
040        }
041        return (float) 1.0;
042    }
043}
