001package mil.emp3.api.listeners;
002
003import mil.emp3.api.enums.EventListenerTypeEnum;
004
005/**
006 * This abstract class is the handle return by the add event listener method. It
007 * must be used to remove the listeners from the system.
008 */
009public abstract class EventListenerHandle {
010    /**
011     * This method return the type of listener the handle refers to.
012     * @return {@link EventListenerTypeEnum} The type of event listener.
013     */
014    public abstract EventListenerTypeEnum getListenerType();
015
016    /**
017     * This method return the event listener object the handle refers to.
018     * @return {@link IEventListener} An event listener derived from IEventListener.
019     */
020    public abstract IEventListener getListener();
021}
