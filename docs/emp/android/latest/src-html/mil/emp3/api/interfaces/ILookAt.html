001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoLookAt;
004import org.cmapi.primitives.IGeoPosition;
005
006import mil.emp3.api.exceptions.EMP_Exception;
007import mil.emp3.api.listeners.EventListenerHandle;
008import mil.emp3.api.listeners.ILookAtEventListener;
009
010/**
011 * This class defines the interface to a lookAt.
012 */
013public interface ILookAt extends IGeoLookAt{
014    /**
015     * This method copies all settings from the LookAt object provided. The settings copied are
016     * the coordinates, altitude, altitude mode, tilt, roll, range and azimuth.
017     * @param from An object that implements the ILookAt interface.
018     */
019    void copySettingsFrom(ILookAt from);
020
021    /**
022     * This method adds an event listener to the LookAt.
023     * @param listener A reference to a class object that implements the {@link  ILookAtEventListener} interface.
024     * @return The handle to the event registration. See {@link EventListenerHandle}
025     * @throws EMP_Exception if the listener is null.
026     */
027    EventListenerHandle addLookAtEventListener(ILookAtEventListener listener) throws EMP_Exception;
028
029    /**
030     * This method must be used to remove the registration from the system.
031     * @param hListener  The handle returned by the call to addLookAtEventListener. See {@link EventListenerHandle}
032     */
033    void removeEventListener(EventListenerHandle hListener);
034
035    /**
036     * This method causes any changes done to the LookAt settings to be applied to any map or object that it may
037     * be associated with.
038     * @param userContext user defined object
039     * @param animate If set to true then camera movement will be animated.
040     */
041    void apply(boolean animate, Object userContext);
042
043    /**
044     * This method causes any changes done to the LookAt settings to be applied to any map or object that it may
045     * be associated with.
046     * @param animate If set to true then camera movement will be animated.
047     */
048    void apply(boolean animate);
049
050    /**
051     * Set the lookkAt position. If position parameters are invalid an exception is thrown.
052     * If altitudeMode is null then it will then original altitude mode is maintained.
053     * @param latitude
054     * @param longitude
055     * @param altitude
056     * @param altitudeMode
057     */
058    void setPosition(double latitude, double longitude, double altitude, AltitudeMode altitudeMode);
059
060    /**
061     * set the lookAt position. If position is null or position parameters are invalid an exception is thrown.
062     * Note that altitude mode is not affected by this method.
063     * @param position
064     */
065    void setPosition(IGeoPosition position);
066}
