001package mil.emp3.api.utils;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import mil.emp3.api.Overlay;
007import mil.emp3.api.interfaces.IContainer;
008import mil.emp3.api.interfaces.IFeature;
009import mil.emp3.api.interfaces.IOverlay;
010
011/**
012 * This class implements an entry for the hierarchy of EMP.
013 */
014
015public class EmpObjectHierarchyEntry {
016    private final List<EmpObjectHierarchyEntry> childrenList = new ArrayList<>();
017    private final IContainer empObject;
018    private final EmpObjectHierarchyEntry parentEntry;
019
020    public EmpObjectHierarchyEntry(EmpObjectHierarchyEntry parent, IContainer object) {
021        empObject = object;
022        parentEntry = parent;
023
024        if (null != parentEntry) {
025            parentEntry.addChild(this);
026        }
027    }
028
029    public IContainer getEmpObject() {
030        return this.empObject;
031    }
032
033    public EmpObjectHierarchyEntry getParent() {
034        return this.parentEntry;
035    }
036
037    public List<EmpObjectHierarchyEntry> getChildren() {
038        return this.childrenList;
039    }
040
041    public void  addChild(EmpObjectHierarchyEntry child) {
042        this.childrenList.add(child);
043    }
044
045    public boolean isOverlayEntry() {
046        return (this.empObject instanceof IOverlay);
047    }
048
049    public boolean isFeatureEntry() {
050        return (this.empObject instanceof IFeature);
051    }
052}
