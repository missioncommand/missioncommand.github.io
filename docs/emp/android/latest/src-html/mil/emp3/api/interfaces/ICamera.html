001package mil.emp3.api.interfaces;
002
003import org.cmapi.primitives.IGeoCamera;
004
005import mil.emp3.api.exceptions.EMP_Exception;
006import mil.emp3.api.listeners.EventListenerHandle;
007import mil.emp3.api.listeners.ICameraEventListener;
008
009/**
010 * This class defines the interface to a camera.
011 * @author ish.rivera
012 */
013public interface ICamera extends IGeoCamera {
014    /**
015     * This method copies all settings from the camera object provided. The settings copied are
016     * the coordinates, altitude, altitude mode, tilt, roll, range and azimuth.
017     * @param from An object that implements the ICamera interface.
018     */
019    void copySettingsFrom(ICamera from);
020    
021    /**
022     * This method adds an event listener to the camera.
023     * @param listener A reference to a class object that implements the {@link  ICameraEventListener} interface.
024     * @return The handle to the event registration. See {@link EventListenerHandle}
025     */
026    EventListenerHandle addCameraEventListener(ICameraEventListener listener) throws EMP_Exception;
027    
028    /**
029     * This method must be used to remove the registration from the system.
030     * @param hListener  The handle returned by the call to addCameraEventListener. See {@link EventListenerHandle}
031     */
032    void removeEventListener(EventListenerHandle hListener);
033
034    /**
035     * This method causes any changes done to the camera settings to be applied to any map or object that it may
036     * be associated with.
037     * @param animate If set to true then camera movement will be animated.
038     */
039    void apply(boolean animate);
040}
