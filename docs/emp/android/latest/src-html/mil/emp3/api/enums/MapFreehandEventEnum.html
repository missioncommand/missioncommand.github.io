001package mil.emp3.api.enums;
002
003import mil.emp3.api.interfaces.IEventEnum;
004
005/**
006 * Types of events generated during a freehand draw operation. Map enters FREEHAND_DRAW {@link EditorMode} when application issues drawFreehand action.
007 */
008public enum MapFreehandEventEnum implements IEventEnum{
009    /**
010     * Map entered freehand draw mode as a result of drawFreehand issued by the application
011     */
012    MAP_ENTERED_FREEHAND_DRAW_MODE,
013    /**
014     * User started to draw on the map.
015     */
016    MAP_FREEHAND_LINE_DRAW_START,
017    /**
018     * Line was been updated by the user, i.e. user continued with the DRAG gesture
019     */
020    MAP_FREEHAND_LINE_DRAW_UPDATE,
021    /**
022     * User finished drawing a line (lifted its finger) i.e DRAG is complete.
023     */
024    MAP_FREEHAND_LINE_DRAW_END,
025    /**
026     * Map exited freehand draw mode as a result of drawFreehandExit issues by the  application
027     */
028    MAP_EXIT_FREEHAND_DRAW_MODE
029}
