001package mil.emp3.api.enums;
002
003/**
004 * This class enumerates the map grid line types that can be set on a map.
005 */
006
007public enum MapGridTypeEnum {
008    /**
009     * This enumerated value indicates that the map will display no grid lines. It is the default value.
010     */
011    NONE,
012    /**
013     * This enumerated value indicates that the map will display MGRS grid lines.
014     */
015    MGRS,
016    /**
017     * This enumerated value indicates that the map will display UTM grid lines.
018     */
019    UTM,
020    /**
021     * This enumerated value indicates that the map will display Degree, Minute and seconds grid lines.
022     */
023    DMS,
024    /**
025     * This enumerated value indicates that the map will display decimal degrees grid lines.
026     */
027    DD
028}
