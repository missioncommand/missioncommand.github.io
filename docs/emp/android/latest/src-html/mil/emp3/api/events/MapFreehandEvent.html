001package mil.emp3.api.events;
002
003import mil.emp3.api.enums.MapFreehandEventEnum;
004import mil.emp3.api.interfaces.IMap;
005import org.cmapi.primitives.IGeoPositionGroup;
006import org.cmapi.primitives.IGeoStrokeStyle;
007
008/**
009 * This class contains the map freehand event data.
010 */
011public abstract class MapFreehandEvent extends Event<MapFreehandEventEnum, IMap> {
012    protected MapFreehandEvent(
013            MapFreehandEventEnum eEvent,
014            IMap map) {
015        super(eEvent, map);
016    }
017
018    /**
019     * Gets the style that was used to draw the line.
020     * @return style of line
021     */
022    public abstract IGeoStrokeStyle getStyle();
023
024    /**
025     * Gets the points from the drawing
026     * @return points from drawing
027     */
028    public abstract IGeoPositionGroup getPositionGroup();
029}
