001package mil.emp3.api.interfaces;
002
003/**
004 * This interface defines the screen capture callback methods. A call to IMap.getScreenCapture must
005 * provide an implementation of this interface. The methods defined are called to provide the outcome
006 * of the screen capture operation.
007 */
008
009public interface IScreenCaptureCallback {
010    /**
011     * This method is called when the screen capture has been performed and available in the capture object.
012     * @param capture The object that implements the ICapture interface.
013     */
014    void captureSuccess(ICapture capture);
015
016    /**
017     * This method is called when the screen capture fails.
018     * @param Ex The exception that caused the screen capture to fail.
019     */
020    void captureFailed(Exception Ex);
021}
