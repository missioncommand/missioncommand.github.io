001package mil.emp3.api.interfaces;
002
003import java.util.List;
004
005import mil.emp3.api.enums.WMSVersionEnum;
006import mil.emp3.api.exceptions.EMP_Exception;
007
008/**
009 *
010 */
011public interface IWMS extends IMapService {
012    /**
013     * This method retrieves the wms version.
014     * @return WMSVersionEnum
015     */    
016    WMSVersionEnum getWMSVersion();
017
018    /**
019     * This method retrieves the title format.
020     * @return String
021     */    
022    String getTileFormat();
023
024    /**
025     * This method retrieves the transparency property.
026     * @return String
027     */    
028    boolean getTransaparent();
029
030    /**
031     * This method sets the coordinate system.
032     * @param coordinateSystem
033     */
034
035    void setCoordinateSystem(String coordinateSystem) throws EMP_Exception;
036
037    /** This method returns the coordinate system.
038     * @return String
039     */
040
041    String getCoordinateSystem();
042
043    /**
044     *  This method sets the styles for the map.
045     */
046
047    void setStyles(List<String> styles);
048
049    /**
050     *  This method gets the styles for the map.
051     */
052
053    List<String> getStyles();
054
055    /**
056     * This method sets the time string for the map.
057     */
058    void setTimeString(String timeString);
059
060    /**
061     * This method gets the time string for the map
062     */
063    String getTimeString();
064
065    /**
066     * This sets the resolution of the map data in meters per pixel.
067     * @param resolution
068     */
069
070    void setLayerResolution(double resolution);
071
072    /**
073     * This gets the resolution of the map data in meters per pixel.
074     * @return
075     */
076
077    double getLayerResolution();
078
079    /**
080     * This method sets the layer list. The updated list is sent to all maps utilizing
081     * this service.
082     * @param newLayers A list of string. The list is ignored if its null or empty.
083     * @throws EMP_Exception
084     */
085    void setLayers(List<String> newLayers) throws EMP_Exception;
086
087    /**
088     * This method retrieves the current layer list.
089     * @return A list of String values.
090     */
091    List<String> getLayers();
092}
