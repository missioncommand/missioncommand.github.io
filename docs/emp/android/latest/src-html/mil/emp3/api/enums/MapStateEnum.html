001package mil.emp3.api.enums;
002
003/**
004 * Type of usability state of the Map. Map must be in MAP_READY state before applications can use the API to update the map or issue any actions.
005 */
006public enum MapStateEnum {
007  /**
008   * Map instance has NOT been initialized and therefore inoperable. It is the initial state of all map instances.
009   */
010  MAP_NEW,
011  /**
012   * Map instance is initializing.
013   */
014  INIT_IN_PROGRESS,
015  /**
016   * Map instance is currently being swapped.
017   */
018  MAP_SWAP_IN_PROGRESS,
019  /**
020   * Map instance is ready to process request.
021   */
022  MAP_READY,
023  /**
024   * Map instances is shutting down.
025   */
026  SHUTDOWN_IN_PROGRESS,
027  /**
028   * Map instance was shutdown.
029   */
030  SHUTDOWN
031}
