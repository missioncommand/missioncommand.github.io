577    /**
578     * This constructor creates MilStdSymbol from the the IGeoMilSymbol object provided.
579     * If the standard version is not set, it is set to the default value. If the modifiers are
580     * null it is set to an empty modifier list.
581     * @param oRenderable an object that implements the IGeoMilSymbol interface.
582     * @throws mil.emp3.api.exceptions.EMP_Exception This exception is raised if the geo renderable
583     * provided has an invalid symbol code.
584     */
585    public MilStdSymbol(IGeoMilSymbol oRenderable) throws EMP_Exception {
586        super(oRenderable, FeatureTypeEnum.GEO_MIL_SYMBOL);
587
588        if (this.getRenderable().getSymbolStandard() == null) {
589            this.setSymbolStandard(SymbolStandard.MIL_STD_2525C);
590        }
591
592        if (this.getRenderable().getModifiers() == null) {
593            this.getRenderable().setModifiers(new java.util.HashMap<IGeoMilSymbol.Modifier, String>());
594        }
595        
596        if ((oRenderable.getSymbolCode() == null) || (oRenderable.getSymbolCode().length() != 15)) {
597            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "The geo Renderable has an invalid symbol code.");
598        }
599
600        if (null == this.getAffiliation()) {
601            this.setAffiliation(Affiliation.FRIEND);
602        }
603    }

711    private ArrayList<String> convertToArrayList(String sCommaDelimitedString) {
712        ArrayList<String> aValueList = new ArrayList<>(java.util.Arrays.asList(sCommaDelimitedString.split("\\s*,\\s*")));
713
714        if (sCommaDelimitedString.isEmpty()) {
715            aValueList = new ArrayList<>();
716        } else {
717            aValueList = new ArrayList<>(java.util.Arrays.asList(sCommaDelimitedString.split("\\s*,\\s*")));
718        }
719
720        return aValueList;
721    }

723    /**
724     * This method inserts the numeric value of a comma delimited modifier at the specified index.
725     * @param eModifier See IGeoMilModifier.milModifierName. Not all modifier values are comma delimited values.
726     * @param iIndex The index of the value to insert. The first value is 0.
727     * @param fValue The numeric value to set. or Float.NaN to remove the value.
728     */
729    public void setModifier(IGeoMilSymbol.Modifier eModifier, int iIndex , float fValue) {
730        java.util.HashMap<IGeoMilSymbol.Modifier, String> oModifierList =
731                this.getRenderable().getModifiers();
732        String sValue = "";
733        ArrayList<String> aValueList;
734
735        if (oModifierList.containsKey(eModifier)) {
736            sValue = oModifierList.get(eModifier);
737        }
738
739        aValueList = this.convertToArrayList(sValue);
740
741        if ((iIndex >= 0) && (iIndex < aValueList.size())) {
742            if (Float.isNaN(fValue)) {
743                aValueList.remove(iIndex);
744            } else {
745                aValueList.add(iIndex, Float.toString(fValue));
746            }
747        } else if (!Float.isNaN(fValue)) {
748            // Add it to the end.
749            aValueList.add(Float.toString(fValue));
750        }
751
752        sValue = aValueList.toString(); // This operation encloses the string in "[]" we need to remove them.
753        if (sValue.length() > 0) {
754            sValue = sValue.substring(1, sValue.length() - 1);
755        }
756        this.setModifier(eModifier, sValue);
757    }

760     * This method retrieves the value of the specified modifier.
761     * @param eModifier See IGeoMilModifier.milModifierName
762     * @return A String value or null if it does not exists.
763     */
764    public String getStringModifier(IGeoMilSymbol.Modifier eModifier) {
765        String sValue = null;
766        java.util.HashMap<IGeoMilSymbol.Modifier, String> oModifierList =
767                this.getModifiers();

1360    /**
1361     * This method returns a list of attribute which can be used to call the MilStd renderer.
1362     * @param iIconSize              The size of the icon
1363     * @param selected               True if the feature is currenlt selected.
1364     * @param selectedStrokeColor    The stroke color for selected features.
1365     * @param selectedTextColor      The text color for selected features.
1366     * @return SparseArray of attributes.
1367     */
1368    public SparseArray<String> getAttributes(int iIconSize, boolean selected, IGeoColor selectedStrokeColor, IGeoColor selectedTextColor) {
1369        IGeoColor strokeColor = null;
1370        IGeoColor textColor = null;
1371        SparseArray<String> oArray = new SparseArray<>();
1372        IGeoFillStyle oFillStyle = getFillStyle();
1373        IGeoStrokeStyle oStrokeStyle = getStrokeStyle();
1374        IGeoLabelStyle labelStyle = getLabelStyle();
1375
1376
1377        oArray.put(MilStdAttributes.SymbologyStandard, "" + geoMilStdVersionToRendererVersion());
1378        oArray.put(MilStdAttributes.PixelSize, "" + iIconSize);
1379        oArray.put(MilStdAttributes.KeepUnitRatio, "true");
1380        oArray.put(MilStdAttributes.UseDashArray, "true");
1381
1382        if (selected) {
1383            strokeColor = selectedStrokeColor;
1384            textColor = selectedTextColor;
1385        } else {
1386            if (oStrokeStyle != null) {
1387                strokeColor = oStrokeStyle.getStrokeColor();
1388            }
1389            if (labelStyle != null) {
1390                textColor = labelStyle.getColor();
1391            }
1392        }
1393
1394        if (oFillStyle != null) {
1395            oArray.put(MilStdAttributes.FillColor, "#" + ColorUtils.colorToString(oFillStyle.getFillColor()));
1396        }
1397
1398        if (oStrokeStyle != null) {
1399            oArray.put(MilStdAttributes.LineColor, "#" + ColorUtils.colorToString(oStrokeStyle.getStrokeColor()));
1400            oArray.put(MilStdAttributes.LineWidth, "" + (int) oStrokeStyle.getStrokeWidth());
1401        }
1402
1403        if (strokeColor != null) {
1404            oArray.put(MilStdAttributes.LineColor, "#" + ColorUtils.colorToString(strokeColor));
1405        }
1406
1407        if (textColor != null) {
1408            oArray.put(MilStdAttributes.TextColor, "#" + ColorUtils.colorToString(textColor));
1409            // There is currently no way to change the font.
1410        }
1411
1412        if (isSinglePoint()) {
1413            oArray.put(MilStdAttributes.FontSize, "" + FontUtilities.getTextPixelSize(labelStyle, FontSizeModifierEnum.NORMAL));
1414        }
1415
1416        return oArray;
1417    }

1450    public IEmpBoundingBox getFeatureBoundingBox() {
1451        if (!this.isTacticalGraphic()) {
1452            return null;
1453        }
1454
1455        IEmpBoundingBox bBox = new EmpBoundingBox();
1456        armyc2.c2sd.renderer.utilities.SymbolDef symbolDefinition = getTacticalGraphicSymbolDefinition();
1457
1458        for (IGeoPosition pos: getPositions()) {
1459            bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1460        }
1461
1462        if (getPositions().size() >= symbolDefinition.getMinPoints()) {
1463            // We need the minimum number of points.
1464            List<IGeoPosition> posList = getPositions();
1465
1466            switch (symbolDefinition.getDrawCategory()) {
1467                /**
1468                 * A polyline, a line with n number of points.
1469                 * 0 control points
1470                 */
1471                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_LINE:
1472                    /**
1473                     * A polyline with n points (entered in reverse order)
1474                     * 0 control points
1475                     */
1476                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_ARROW:
1477                    break;
1478                /**
1479                 * TG such as:
1480                 * TACGRP.TSK.ISL,
1481                 * TACGRP.TSK.OCC,
1482                 * TACGRP.TSK.RTN,
1483                 * TACGRP.TSK.SCE
1484                 */
1485                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_AUTOSHAPE: {
1486                    double dist = GeoLibrary.computeDistanceBetween(posList.get(0), posList.get(1));
1487                    IGeoPosition pos = new GeoPosition();
1488
1489                    // Compute north.
1490                    GeoLibrary.computePositionAt(0.0, dist, posList.get(0), pos);
1491                    bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1492                    // Compute east.
1493                    GeoLibrary.computePositionAt(90.0, dist, posList.get(0), pos);
1494                    bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1495                    // Compute south.
1496                    GeoLibrary.computePositionAt(180.0, dist, posList.get(0), pos);
1497                    bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1498                    // Compute west.
1499                    GeoLibrary.computePositionAt(270.0, dist, posList.get(0), pos);
1500                    bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1501                    break;
1502                }
1503                /**
1504                 * An enclosed polygon with n points
1505                 * 0 control points
1506                 */
1507                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_POLYGON:
1508                    break;
1509                /**
1510                 * A graphic with n points whose last point defines the width of the graphic.
1511                 * 1 control point
1512                 */
1513                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_ROUTE:
1514                    break;
1515                /**
1516                 * A line defined only by 2 points, and cannot have more.
1517                 * 0 control points
1518                 */
1519                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_TWOPOINTLINE:
1520                    /**
1521                     * A polyline with 2 points (entered in reverse order).
1522                     * 0 control points
1523                     */
1524                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_TWOPOINTARROW:
1525                    break;
1526                /**
1527                 * Shape is defined by a single point
1528                 * 0 control points
1529                 */
1530                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_POINT:
1531                    break;
1532                /**
1533                 * An animated shape, uses the animate function to draw. Super Autoshape draw
1534                 * in 2 phases, usually one to define length, and one to define width.
1535                 * 0 control points (every point shapes symbol)
1536                 *
1537                 */
1538                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_SUPERAUTOSHAPE:
1539                    break;
1540                /**
1541                 * Circle that requires 1 AM modifier value.
1542                 * See ModifiersTG.js for modifier descriptions and constant key strings.
1543                 */
1544                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_CIRCULAR_PARAMETERED_AUTOSHAPE:
1545                    break;
1546                /**
1547                 * Rectangle that requires 2 AM modifier values and 1 AN value.";
1548                 * See ModifiersTG.js for modifier descriptions and constant key strings.
1549                 */
1550                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_RECTANGULAR_PARAMETERED_AUTOSHAPE:
1551                    break;
1552                /**
1553                 * Requires 2 AM values and 2 AN values per sector.
1554                 * The first sector can have just one AM value although it is recommended
1555                 * to always use 2 values for each sector.  X values are not required
1556                 * as our rendering is only 2D for the Sector Range Fan symbol.
1557                 * See ModifiersTG.js for modifier descriptions and constant key strings.
1558                 */
1559                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_SECTOR_PARAMETERED_AUTOSHAPE:
1560                    break;
1561                /**
1562                 *  Requires at least 1 distance/AM value"
1563                 *  See ModifiersTG.js for modifier descriptions and constant key strings.
1564                 */
1565                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_CIRCULAR_RANGEFAN_AUTOSHAPE: {
1566                    double dist = getNumericModifier(Modifier.DISTANCE, 0);
1567
1568                    if (dist != Double.NaN) {
1569                        IGeoPosition pos = new GeoPosition();
1570
1571                        // Compute north.
1572                        GeoLibrary.computePositionAt(0.0, dist, posList.get(0), pos);
1573                        bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1574                        // Compute east.
1575                        GeoLibrary.computePositionAt(90.0, dist, posList.get(0), pos);
1576                        bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1577                        // Compute south.
1578                        GeoLibrary.computePositionAt(180.0, dist, posList.get(0), pos);
1579                        bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1580                        // Compute west.
1581                        GeoLibrary.computePositionAt(270.0, dist, posList.get(0), pos);
1582                        bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1583                    }
1584                    break;
1585                }
1586                /**
1587                 * Requires 1 AM value.
1588                 * See ModifiersTG.js for modifier descriptions and constant key strings.
1589                 */
1590                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_TWO_POINT_RECT_PARAMETERED_AUTOSHAPE:
1591                    double dist = getNumericModifier(Modifier.DISTANCE, 0);
1592
1593                    if (dist != Double.NaN) {
1594                        IGeoPosition pos = new GeoPosition();
1595
1596                        // Compute north.
1597                        GeoLibrary.computePositionAt(0.0, dist, posList.get(0), pos);
1598                        bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1599                        // Compute south.
1600                        GeoLibrary.computePositionAt(180.0, dist, posList.get(0), pos);
1601                        bBox.includePosition(pos.getLatitude(), pos.getLongitude());
1602                    }
1603                    break;
1604                /**
1605                 * 3D airspace, not a milstd graphic.
1606                 */
1607                case armyc2.c2sd.renderer.utilities.SymbolDef.DRAW_CATEGORY_3D_AIRSPACE:
1608                    break;
1609            }
1610
1611            // Now we need to extend the box by ~ 10%.
1612            double deltaLat = bBox.deltaLatitude();
1613            double deltaLong = bBox.deltaLongitude();
1614
1615            if (deltaLat == 0.0) {
1616                deltaLat = 0.05;
1617            }
1618            if (deltaLong == 0.0) {
1619                deltaLong = 0.05;
1620            }
1621
1622            deltaLat *= 0.05;
1623            deltaLong *= 0.05;
1624
1625            bBox.includePosition(bBox.getNorth() + deltaLat, bBox.getWest());
1626            bBox.includePosition(bBox.getSouth() - deltaLat, bBox.getWest());
1627            bBox.includePosition(bBox.getNorth(), bBox.getWest() - deltaLong);
1628            bBox.includePosition(bBox.getNorth(), bBox.getEast() + deltaLong);
1629        }
1630
1631        return bBox;
1632    }
