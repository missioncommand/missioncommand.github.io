001package mil.emp3.api.enums;
002
003/**
004 * Types of locks that can be placed on the map's motion. In a normal scenario map is in UNLOCKED state i.e. user can pan/zoom/rotate the map.
005 */
006public enum MapMotionLockEnum {
007
008    /**
009     * Map view is free to change i.e. pan, zoom, rotate, tilt
010     * based on Gestures. Note that pan, zoom, rotate, tilt is not necessarily an exhaustive list of motions
011     */
012    UNLOCKED,
013    /**
014     * Map will not pan, zoom or rotate, tilt based on Gestures. The only
015     * motion allowed is panning if the user is dragging an object at the edge of the viewing area.
016     * This is currently used by the Freehand Editor. drawFeature and editFeature actions are not allowed when map
017     * is in this lock mode.
018     */
019    SMART_LOCK,
020    /**
021     * Map will not pan, zoom, rotate, tilt based on Gestures.
022     * Client application can still use Camera and LookAt to change the map view.
023     *  drawFeature and editFeature actions are not allowed when map is in this lock mode.
024     */
025    LOCKED
026}
