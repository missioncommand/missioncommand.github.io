001package mil.emp3.api.enums;
002
003/**
004 *
005 * Visibility states of an object; the visibility
006 * state of an object is specific to a map. Same object can have different
007 * visibility states on different maps.
008 */
009public enum VisibilityStateEnum {
010    /**
011     * Object is visible on the map; the visibility of any object contained within the object will depend on its own
012     * visibility state on the same map.
013     */
014    VISIBLE,
015
016    /**
017     * Visibility state of an object in context to a specific parent on a specific map.
018     * It indicates that the object is marked visible under the parent in question but
019     * is hidden due to the parents ancestors (that are on the same map) visibility 
020     * setting. The object would not be visible if it had no other parents on the 
021     * same map. If the object had other parents on the same map the objects 
022     * visibility on the map would depend on its visibility state with those other parents.
023     */
024    VISIBLE_ANCESTOR_HIDDEN,
025
026    /**
027     * Not visible in relation to parent.
028     * In the case of Map.getVisible(target) then HIDDEN means it is not visible to user on map engine at all.
029     * HIDDEN in relation to a specific parent does not guarantee that it is not visible to the user under a different parent
030     */
031    HIDDEN
032}
