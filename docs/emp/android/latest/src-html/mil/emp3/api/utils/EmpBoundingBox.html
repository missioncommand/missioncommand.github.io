026    /**
027     * This constructor defines an area bounded by the north, south, east, and west parameters provided.
028     * @param northValue The most northern latitude of the area. If this value is less than southValue, the values are swapped.
029     * @param southValue The most southern latitude of the area. If this value is greater than northValue, the values are swapped.
030     * @param eastValue The east longitude of the area. If the east value is greater than the west value the area contains the international date line.
031     * @param westValue The west longitude of the area. If the west value is less than the east value the area contains the international date line.
032     */
033    public EmpBoundingBox(double northValue, double southValue, double eastValue, double westValue) {
034        if (Double.isNaN(northValue) || Double.isNaN(southValue) || Double.isNaN(eastValue) || Double.isNaN(westValue)) {
035            throw new IllegalArgumentException("Invalid value NaN.");
036        }
037
038        if ((northValue < -90) || (northValue > 90) || (southValue < -90) || (southValue > 90)) {
039            throw new IllegalArgumentException("Invalid latitude value.");
040        }
041
042        if ((eastValue < -180) || (eastValue > 180) || (westValue < -180) || (westValue > 180)) {
043            throw new IllegalArgumentException("Invalid longitude value.");
044        }
045
046        if (northValue >= southValue) {
047            this.setNorth(northValue);
048            this.setSouth(southValue);
049        } else {
050            this.setSouth(northValue);
051            this.setNorth(southValue);
052        }
053        this.setEast(eastValue);
054        this.setWest(westValue);
055    }
056
057    /**
058     * This constructor defines an area bounded by the northEast and southWest coordinates.
059     * @param southWest The coordinates of the south west point of the bounding area.
060     * @param northEast The coordinates of the north east point of the bounding area.
061     */
062    public EmpBoundingBox(IGeoPosition southWest, IGeoPosition northEast) {
063        if ((null == southWest) || (null == northEast)) {
064            throw new IllegalArgumentException("Invalid null GeoPosition.");
065        }
066        if (northEast.getLatitude() >= southWest.getLatitude()) {
067            this.setNorth(northEast.getLatitude());
068            this.setSouth(southWest.getLatitude());
069        } else {
070            this.setSouth(northEast.getLatitude());
071            this.setNorth(southWest.getLatitude());
072        }
073
074        this.setEast(northEast.getLongitude());
075        this.setWest(southWest.getLongitude());
076    }

114    @Override

116        if (Double.isNaN(value)) {
117            throw new IllegalArgumentException("value is not a number.");
118        }
119        super.setNorth(((((value + 90.0) % 180.0) + 180.0) % 180.0) - 90.0);

131    public void setWest(double value) {
132        if (Double.isNaN(value)) {
133            throw new IllegalArgumentException("value is not a number.");
134        }
135        super.setWest(((((value + 180.0) % 360.0) + 360.0) % 360.0) - 180.0);
136        if (getWest() == 180.0) {
137            super.setWest(-180.0);
138        }
139    }

147        if (getEast() == -180.0) {
148            super.setEast(180.0);
149        }

152    /**
153     * This method returns the delta of the north and south latitudes.
154     * @return Degrees
155     */
156    @Override
157    public double deltaLatitude() {
158        return Math.abs(this.north() - this.south());
159    }

161    /**
162     * This method returns the delta between the west and east longitudes.
163     * @return degrees.
164     */
165    @Override
166    public double deltaLongitude() {
167
168        if (containsIDL()) {
169            return (180.0 - this.west()) + (this.east() + 180.0);
170        }
171        return this.east() - this.west();
172    }

176     * @return True if the IDL is contained in the bounding box, false if not.

180        return (this.west() > this.east());

199    @Override
200    public boolean intersects(double northDeg, double southDeg, double eastDeg, double westDeg) {
201        if ((northDeg == Double.NaN) || (southDeg == Double.NaN) || (eastDeg == Double.NaN) || (westDeg == Double.NaN)) {
202            throw new IllegalArgumentException("invalid coordinate.");
203        }
204
205        boolean bRet = false;
206
207        // If their north and south don't intersect they don't intersect.
208        if ((this.south() <= northDeg) && (this.north() >= southDeg)) {
209            if (this.containsIDL()) {
210                // This bounding box contains the IDL.
211                if (westDeg > eastDeg) {
212                    // The coordinates provided also contains the IDL;
213                    bRet = true;
214                } else {
215                    // The coordinates provided do not contains the IDL;
216                    bRet = ((this.west() <= eastDeg) && (180 >= westDeg)) || ((-180 <= eastDeg) && (this.east() >= westDeg));
217                }
218            } else {
219                // This bounding box does not contains the IDL.
220                if (westDeg > eastDeg) {
221                    // The coordinates provided contains the IDL;
222                    bRet = ((this.getWest() <= eastDeg) && (this.getEast() <= westDeg)) || ((this.getEast() >= westDeg) && (this.getWest() >= eastDeg));
223                } else {
224                    // The coordinates provided do not contains the IDL;
225                    bRet = ((this.getWest() <= eastDeg) && (this.getEast() >= westDeg));
226                }
227            }
228        }
229        return bRet;
230    }

304    /**
305     * This method checks if the the given lat/long is contained in the bounding box.
306     * @param latitude
307     * @param longitude
308     * @return If if it is contained in the bounding box, false otherwise.
309     */
310    @Override
311    public boolean contains(double latitude, double longitude) {
312        if (latitude > this.north()) {
313            return false;
314        }
315        if (latitude < this.south()) {
316            return false;
317        }
318
319        if (this.containsIDL()) {
320            if ((longitude < this.west()) && (longitude > this.east())) {
321                return false;
322            }
323        } else if (longitude > this.east()) {
324            return false;
325        } else if (longitude < this.west()) {
326            return false;
327        }
328
329        return true;
330    }

356    /**
357     * This method returns the latitude of the center of the bounding area.
358     * @return Center latitude in degrees.
359     */
360    @Override
361    public double centerLatitude() {
362        return (this.north() + this.south()) / 2.0;
363    }
364
365    /**
366     * This method returns the longitude of the center of the bounding area.
367     * @return Center longitude in degrees.
368     */
369    @Override
370    public double centerLongitude() {
371        if (this.containsIDL()) {
372            return (((((this.west() + (this.deltaLongitude() / 2.0)) + 180.0) % 360.0) + 360.0) % 360.0) - 180.0;
373        }
374
375        return (this.east() + this.west()) / 2.0;
376    }

378    /**
379     * This method returns the bounding box that is the intersection of this box and the box defined
380     * by the with parameter. The resulting box is returned in the result parameter.
381     * @param with The bounding box to intersect with.
382     * @param result A bounding box to place the result.
383     * @return The result parameter with the intersecting bounds or null if the two bounding boxes do not intersect.
384     */
385    @Override
386    public IEmpBoundingBox intersection(IEmpBoundingBox with, IEmpBoundingBox result) {
387        if (null == with) {
388            throw new IllegalArgumentException("The with parameters can not be null.");
389        }
390        if (null == result) {
391            throw new IllegalArgumentException("The result parameters can not be null.");
392        }
393
394        if (!this.intersects(with)) {
395            return null;
396        }
397
398        result.setNorth(Math.min(this.getNorth(), with.getNorth()));
399        result.setSouth(Math.max(this.getSouth(), with.getSouth()));
400
401        if (this.containsIDL()) {
402            // This box contains the IDL
403            if (with.containsIDL()) {
404                // They both contain the IDL.
405                result.setWest(Math.max(this.getWest(), with.getWest()));
406                result.setEast(Math.min(this.getEast(), with.getEast()));
407            } else if (with.getWest() > 0) {
408                // With does not contain the IDL and it is on the left side of the IDL.
409                result.setWest(Math.max(this.getWest(), with.getWest()));
410                result.setEast(with.getEast());
411            } else {
412                // With does not contain the IDL and it is on the right side of the IDL.
413                result.setWest(with.getWest());
414                result.setEast(Math.min(this.getEast(), with.getEast()));
415            }
416        } else if (with.containsIDL()) {
417            // This does not contain the IDL but With does.
418            if (this.getWest() > 0) {
419                // This is on the left side of the IDL
420                result.setWest(Math.max(this.getWest(), with.getWest()));
421                result.setEast(this.getEast());
422            } else {
423                // This is on the right side of the IDL
424                result.setWest(this.getWest());
425                result.setEast(Math.min(this.getEast(), with.getEast()));
426            }
427        } else {
428            // This nor with contain the IDL.
429            result.setWest(Math.max(this.getWest(), with.getWest()));
430            result.setEast(Math.min(this.getEast(), with.getEast()));
431        }
432
433        return result;
434    }

447    @Override
448    public void includePosition(double latitude, double longitude) {
449        if (Double.isNaN(latitude)) {
450            throw new IllegalArgumentException("Ivalid latitude.");
451        }
452        if (Double.isNaN(longitude)) {
453            throw new IllegalArgumentException("Ivalid longitude.");
454        }
455
456        if (Double.isNaN(getEast()) || Double.isNaN(getNorth()) || Double.isNaN(getSouth()) || Double.isNaN(getWest())) {
457            // The bounding box is new.
458            setNorth(latitude);
459            setSouth(latitude);
460            setWest(longitude);
461            setEast(longitude);
462        } else {
463            if (latitude > getNorth()) {
464                // Expand the north
465                setNorth(latitude);
466            }
467            if (latitude < getSouth()) {
468                // Expand the south.
469                setSouth(latitude);
470            }
471
472            if (containsIDL()) {
473                // the current box contains the IDL.
474                if ((longitude < getWest()) && (longitude > getEast())) {
475                    // Its outside of the current box.
476                    if (longitude > 0.0) {
477                        // Its closer to the west edge.
478                        setWest(longitude);
479                    } else {
480                        // Its closer to the east edge.
481                        setEast(longitude);
482                    }
483                }
484            } else {
485                if (longitude < getWest()) {
486                    if (Math.abs(getEast() - longitude) > 180.0) {
487                        // This point forces it to cross the IDL.
488                        setEast(longitude);
489                    } else {
490                        setWest(longitude);
491                    }
492                }
493                if (longitude > getEast()) {
494                    if (Math.abs(getWest() - longitude) > 180.0) {
495                        // This point forces it to cross the IDL.
496                        setWest(longitude);
497                    } else {
498                        setEast(longitude);
499                    }
500                }
501            }
502        }
503    }
