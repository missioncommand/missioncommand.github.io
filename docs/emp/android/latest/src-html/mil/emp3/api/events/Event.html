001package mil.emp3.api.events;
002
003import mil.emp3.api.interfaces.IEvent;
004import mil.emp3.api.interfaces.IEventEnum;
005
006/**
007 * This class is the base class for all event classes.
008 * layers of the EMP3 java map
009 * @param <T> A enumerated class derived from {@link IEventEnum}.
010 * @param <TargetType> Any java class.
011 */
012public class Event<T extends IEventEnum, TargetType extends Object> implements IEvent<T, TargetType> {
013
014    private final T eEvent; // if we change bitmap to enum this will become a searchable list of long, Easier to implement but way less performant
015    private final TargetType oTarget;
016
017    protected Event(T eEvent, TargetType oTarget) {
018        this.eEvent = eEvent;
019        this.oTarget = oTarget;
020    }
021
022    /**
023     *
024     * @return returns the enumerated value of the event generated.
025     */
026    @Override
027    public T getEvent() {
028        return this.eEvent;
029    }
030
031    /**
032     *
033     * @return returns the events target container.
034     */
035    @Override
036    public TargetType getTarget() {
037        return this.oTarget;
038    }
039}
