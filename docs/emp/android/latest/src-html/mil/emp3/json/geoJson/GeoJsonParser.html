036    /**
037     * Parses GeoJSON formatted string and returns a feature list
038     * List may contain, zero, one or multiple features
039     * @param is inputstream
040     * @return List of IFeature
041     */
042
043    public static List<IFeature> parse (InputStream is) throws EMP_Exception {
044        try (Reader reader = new InputStreamReader(is)) {
045            JsonValue value = Json.parse(reader);
046            return parseJsonObject(value);
047        } catch (Exception ioe) {
048            ioe.printStackTrace();
049            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "GeoJSON input error");
050        }
051    }

053    /**
054     * Parses GeoJSON formatted string and returns a feature list
055     * List may contain, zero, one or multiple features
056     * @param input string
057     * @return List of IFeature
058     */
059
060    public static List<IFeature> parse(String input) throws EMP_Exception {
061        JsonValue value = Json.parse(input);
062        return parseJsonObject(value);
063    }
064
065    private static List<IFeature> parseJsonObject(JsonValue value) throws EMP_Exception {
066        List<IFeature> featureList = new ArrayList<>();
067        // At this point input must be feature or feature collection
068        if (value.isObject()) {
069            try {
070                JsonObject object = value.asObject();
071                String objectType = object.get("type").asString();
072                switch (objectType) {
073                    case "FeatureCollection":
074                        JsonArray featureArray = object.get("features").asArray();
075                        for (JsonValue feature : featureArray) {
076                            parseFeature(featureList, feature.asObject());
077                        }
078                        break;
079                    case "Feature":
080                        parseFeature(featureList, object);
081                        break;
082                    case "Point":
083                    case "LineString":
084                    case "Polygon":
085                    case "MultiPoint":
086                    case "MultiLineString":
087                    case "MultiPolygon":
088                        parseGeometry(featureList, object, null);
089                        break;
090                    case "GeometryCollection":
091                        JsonArray geometryArray = object.get("geometries").asArray();
092                        for (JsonValue geometry : geometryArray) {
093                            parseGeometry(featureList, geometry.asObject(), null);
094                        }
095                        break;
096                }
097            } catch (ParseException | java.text.ParseException e) {
098                throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "GeoJSON parse error");
099            }
100        } else {
101            throw new EMP_Exception(EMP_Exception.ErrorDetail.INVALID_PARAMETER, "GeoJSON string is not valid object");
102        }
103        return featureList;
104    }
105
106    private static void parseFeature(List<IFeature> featureList, JsonObject object) throws ParseException,
107            java.text.ParseException {
108        JsonObject geometry = object.get("geometry").asObject();
109        // ignore bounding box object.get("bbox").asObject();
110        JsonObject properties = object.get("properties").asObject();
111        parseGeometry(featureList, geometry, properties);
112    }
113
114    private static void parseGeometry(List<IFeature> featureList, JsonObject geometry, JsonObject properties)
115            throws ParseException, java.text.ParseException {
116        String geometryType = geometry.get("type").asString();
117        boolean multi = geometryType.startsWith("Multi");
118        if (multi) {
119            String singleGeometryType = geometryType.substring(5);// Length of "Multi"
120            JsonArray coordinateListList = geometry.get("coordinates").asArray();
121            // Polygons can have "holes" so we have nested lists
122            if (singleGeometryType.equals("Polygon")) {
123                for (int i = 0; i < coordinateListList.size(); i++) {
124                    JsonArray coordinateListList2 = coordinateListList.get(i).asArray();
125                    parseCoordinateList(featureList, coordinateListList2, geometryType, properties);
126                }
127            } else {
128                parseCoordinateList(featureList, coordinateListList, geometryType, properties);
129            }
130        } else if (geometryType.equals("Polygon")) {
131            JsonArray coordinateListList = geometry.get("coordinates").asArray();
132            parseCoordinateList(featureList, coordinateListList, geometryType, properties);
133        } else {
134            IFeature feature = createFeature(geometryType);
135            if (feature != null) {
136                JsonArray coordinateList = geometry.get("coordinates").asArray();
137                parseCoordinates(feature, coordinateList);
138                parseProperties(feature, properties);
139                featureList.add(feature);
140            }
141        }
142    }
143
144    private static IFeature createFeature(String geometryType) {
145        IFeature newFeature = null;
146        switch (geometryType) {
147            case "Point":
148            case "MultiPoint":
149                newFeature = new Point();
150                break;
151            case "LineString":
152            case "MultiLineString":
153                newFeature = new Path();
154                break;
155            case "Polygon":
156            case "MultiPolygon":
157                newFeature = new Polygon();
158                break;
159            default:
160                Log.d(TAG, "Received unknown feature type " + geometryType);
161        }
162        if (null != newFeature) {
163            newFeature.setAltitudeMode(IGeoAltitudeMode.AltitudeMode.CLAMP_TO_GROUND);
164            newFeature.setName(DEFAULT_NAME_PREFIX + geometryType);
165            newFeature.getStrokeStyle().setStrokeColor(DEFAULT_STROKE_COLOR);
166            newFeature.getFillStyle().setFillColor(DEFAULT_FILL_COLOR);
167        }
168        return newFeature;
169    }
170
171    private static void parseCoordinateList(List<IFeature> featureList, JsonArray coordinateListList,
172                                            String geometryType, JsonObject properties)
173            throws ParseException, java.text.ParseException {
174        for (int i = 0; i < coordinateListList.size(); i++) {
175            IFeature feature = createFeature(geometryType);
176            if (feature != null) {
177                JsonArray coordinateList = coordinateListList.get(i).asArray();
178                parseCoordinates(feature, coordinateList);
179                parseProperties(feature, properties);
180                featureList.add(feature);
181            }
182        }
183    }
184
185    private static void parseCoordinates(IFeature feature, JsonArray coordinateList) throws ParseException {
186        if (!(feature instanceof Point)) {
187            for (JsonValue pair : coordinateList) {
188                JsonArray twoDoubles = pair.asArray();
189                IGeoPosition coordinates = new GeoPosition();
190                coordinates.setLatitude(twoDoubles.get(1).asDouble());
191                coordinates.setLongitude(twoDoubles.get(0).asDouble());
192                coordinates.setAltitude(0);
193                feature.getPositions().add(coordinates);
194            }
195        } else {
196            IGeoPosition coordinates = new GeoPosition();
197            coordinates.setLatitude(coordinateList.get(1).asDouble());
198            coordinates.setLongitude(coordinateList.get(0).asDouble());
199            coordinates.setAltitude(0);
200            ((Point)feature).setPosition(coordinates);
201        }
202    }

204    /**
205     * All non CMAPI properties are ignored
206     * @param feature
207     * @param properties
208     * @throws ParseException
209     * @throws java.text.ParseException
210     */
211    private static void parseProperties(IFeature feature, JsonObject properties) throws ParseException,
212                                                        java.text.ParseException {
213        if (properties == null) {
214            return;
215        }
216        JsonValue optional = null;
217        optional = properties.get("name");
218        if (optional != null) {
219            String name = optional.asString();
220            if (name != null && !name.isEmpty()) {
221                feature.setName(name);
222            }
223        }
224        optional = properties.get("id");
225        // Ignore
226//        if (optional != null) {
227//            String id = optional.asString();
228//            if (id != null && !id.isEmpty()) {
229//                feature.setId(id);
230//            }
231//        }
232        optional = properties.get("description");
233        if (optional != null) {
234            String description = optional.asString();
235            if (description != null && !description.isEmpty()) {
236                feature.setDescription(description);
237            }
238        }
239        optional = properties.get("timePrimitive");
240        if (optional != null) {
241            JsonObject timePrimitive = optional.asObject();
242            if (timePrimitive != null) {
243                JsonObject timeSpan = timePrimitive.get("timeSpan").asObject();
244                IGeoTimeSpan geoTimeSpan = new GeoTimeSpan();
245                if (timeSpan != null) {
246                    String timeSpanBegin = timeSpan.get("begin").asString();
247                    geoTimeSpan.setBegin(zonedDate.parse(timeSpanBegin));
248                    String timeSpanEnd = timeSpan.get("end").asString();
249                    geoTimeSpan.setEnd(zonedDate.parse(timeSpanEnd));
250                    feature.getTimeSpans().add(geoTimeSpan);
251                }
252                String timeStamp = timePrimitive.get("timeStamp").asString();
253                if (timeStamp != null && !timeStamp.isEmpty()) {
254                    Date ts = zonedDate.parse(timeStamp);
255                    feature.setTimeStamp(ts);
256                }
257            }
258        }
259        optional = properties.get("style");
260        if (optional != null) {
261            JsonObject style = optional.asObject();
262            if (style != null) {
263                if (feature instanceof Point) {
264                    optional = style.get("iconStyle");
265                    if (optional != null) {
266                        JsonObject iconStyle = optional.asObject();
267                        if (iconStyle != null) {
268                            String url = iconStyle.get("url").asString();
269                            ((Point) feature).setIconURI(url);
270                        }
271                    }
272                } else {
273                    optional = style.get("lineStyle");
274                    if (optional != null) {
275                        JsonObject lineStyle = optional.asObject();
276                        if (lineStyle != null) {
277                            JsonObject lineColor = lineStyle.get("color").asObject();
278                            feature.getStrokeStyle().setStrokeColor(parseColor(lineColor));
279                        }
280                    }
281                    optional = style.get("polyStyle");
282                    if (optional != null && feature instanceof Polygon) {
283                        JsonObject polyStyle = optional.asObject();
284                        if (polyStyle != null) {
285                            JsonObject polyColor = polyStyle.get("color").asObject();
286                            feature.getFillStyle().setFillColor(parseColor(polyColor));
287                        }
288                    }
289                }
290            }
291        }
292    }

294    private static EmpGeoColor parseColor(JsonObject color) throws ParseException {
295        int r = color.get("r").asInt();
296        int g = color.get("g").asInt();
297        int b = color.get("b").asInt();
298        double a = color.get("a").asDouble();
299        EmpGeoColor empGeoColor = new EmpGeoColor(a, r, g, b);
300        return empGeoColor;
301    }
