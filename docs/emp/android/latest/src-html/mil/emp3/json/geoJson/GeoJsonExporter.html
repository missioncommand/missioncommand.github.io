084    public SparseArray<String> getAttributes(final IFeature feature, boolean selected, IGeoColor selectedStrokeColor, IGeoColor selectedTextColor) {
085        IGeoColor strokeColor = null;
086        IGeoColor textColor = null;
087        SparseArray<String> oArray = new SparseArray<>();
088        IGeoFillStyle oFillStyle = feature.getFillStyle();
089        IGeoStrokeStyle oStrokeStyle = feature.getStrokeStyle();
090        IGeoLabelStyle labelStyle = feature.getLabelStyle();
091
092        oArray.put(MilStdAttributes.KeepUnitRatio, "true");
093        oArray.put(MilStdAttributes.UseDashArray, "false");
094
095        if (selected) {
096            strokeColor = selectedStrokeColor;
097            textColor = selectedTextColor;
098        } else {
099            if (oStrokeStyle != null) {
100                strokeColor = oStrokeStyle.getStrokeColor();
101            }
102            if (labelStyle != null) {
103                textColor = labelStyle.getColor();
104            }
105        }
106
107        if (oFillStyle != null) {
108            oArray.put(MilStdAttributes.FillColor, "#" + ColorUtils.colorToString(oFillStyle.getFillColor()));
109        }
110
111        if (oStrokeStyle != null) {
112            oArray.put(MilStdAttributes.LineColor, "#" + ColorUtils.colorToString(oStrokeStyle.getStrokeColor()));
113            oArray.put(MilStdAttributes.LineWidth, "" + (int) oStrokeStyle.getStrokeWidth());
114        }
115
116        if (strokeColor != null) {
117            oArray.put(MilStdAttributes.LineColor, "#" + ColorUtils.colorToString(strokeColor));
118        }
119
120        if (textColor != null) {
121            oArray.put(MilStdAttributes.TextColor, "#" + ColorUtils.colorToString(textColor));
122            // There is currently no way to change the font.
123        }
124
125        return oArray;
126    }

128    private String convertPositionsToString(List<IGeoPosition> posList) {
129        String temp = "";
130
131        if (posList.isEmpty()) {
132            return temp;
133        }
134
135        for (IGeoPosition pos : posList) {
136            if (!temp.isEmpty()) {
137                temp += " ";
138            }
139            temp += pos.getLongitude() + "," + pos.getLatitude() + "," + pos.getAltitude();
140        }
141
142        return temp;
143    }

145    private void appendCircle(final Circle feature, StringBuffer buffer) throws IOException {
146        IEmpBoundingBox bBox = feature.getFeatureBoundingBox();
147        String boundingBoxStr = bBox.getWest() + "," + bBox.getSouth() + "," + bBox.getEast() + "," + bBox.getNorth();
148        String coordinateStr = convertPositionsToString(feature.getPositions());
149
150        double scale = 636000.0;
151
152        SparseArray<String> modifiers = new SparseArray<>();
153        SparseArray<String> attributes = getAttributes(feature,
154                this.map.isSelected(feature), this.map.getSelectedStrokeStyle().getStrokeColor(),
155                this.map.getSelectedLabelStyle().getColor());
156        String altitudeModeStr = MilStdUtilities.geoAltitudeModeToString(feature.getAltitudeMode());
157
158        modifiers.put(ModifiersTG.AM_DISTANCE, feature.getRadius() + "");
159
160        String geoJSON = SECWebRenderer.RenderSymbol(
161                feature.getGeoId().toString(), feature.getName(), feature.getDescription(),
162                "PBS_CIRCLE-----", coordinateStr, altitudeModeStr, scale, boundingBoxStr,
163                modifiers, attributes, RENDER_JSON, 0);
164        buffer.append(geoJSON);
165    }

237    private void appendGeoJSONPositions(IFeature feature, StringBuffer buffer) {
238        boolean addComma = false;
239        for (IGeoPosition position : feature.getPositions()) {
240            if (addComma) {
241                buffer.append(",\n");
242            } else {
243                addComma = true;
244            }
245            buffer.append("[");
246            buffer.append(position.getLongitude());
247            buffer.append(", ");
248            buffer.append(position.getLatitude());
249            buffer.append("]");
250        }
251    }

253    private void appendGeoJSONOtherProperties(IFeature feature, StringBuffer buffer) {
254        boolean haveTimeSpan = false;
255        if (feature.getTimeSpans().size() > 0) {
256            haveTimeSpan = true;
257            buffer.append(",\n\"timePrimitive\": {");
258            buffer.append("\"timeSpan\": {");
259            IGeoTimeSpan timeSpan = feature.getTimeSpans().get(0);
260            buffer.append("\"begin\": {");
261            buffer.append(zonedDate.format(timeSpan.getBegin()));
262            buffer.append("}");
263            buffer.append("\"end\": {");
264            buffer.append(zonedDate.format(timeSpan.getEnd()));
265            buffer.append("}");
266            buffer.append("}");// time span
267        }
268        if (feature.getTimeStamp() != null) {
269            if (!haveTimeSpan) {
270                buffer.append(",\"timePrimitive\": {");
271                haveTimeSpan = true;
272            }
273            buffer.append("\"timeStamp\": {");
274            buffer.append(zonedDate.format(feature.getTimeStamp()));
275            buffer.append("}");// time stamp
276        }
277        if (haveTimeSpan) {
278            buffer.append("}");// time primitive
279        }
280        buffer.append(",\n\"name\":");
281        buffer.append("\"" + feature.getName() + "\"");
282        buffer.append(",\n\"id\":");
283        buffer.append("\"" + feature.getGeoId() + "\"");
284        buffer.append(",\n\"description\":");
285        buffer.append("\"" + feature.getDescription() + "\"");
286        buffer.append("}");
287    }

289    private void appendGeoJSONPolygon(IFeature feature, StringBuffer buffer) {
290        buffer.append("\"geometry\":  {\"type\": \"Polygon\",");
291        buffer.append("\"coordinates\":  [[");
292        appendGeoJSONPositions(feature, buffer);
293        buffer.append("]]");// end of coordinates
294        buffer.append("}");// end of geometry
295        buffer.append(",\n\"properties\": {");
296        buffer.append("\"style\": {");
297        buffer.append("\"lineStyle\": {");
298        IGeoColor color = feature.getStrokeStyle().getStrokeColor();
299        appendGeoJSONColor(color, buffer);
300        buffer.append("}"); // end of lineStyle
301        if (feature.getFillStyle() != null) {
302            buffer.append(",\"polyStyle\": {");
303            color = feature.getFillStyle().getFillColor();
304            appendGeoJSONColor(color, buffer);
305            buffer.append("}"); // end of polyStyle
306        }
307        buffer.append("}"); // end of style
308        appendGeoJSONOtherProperties(feature, buffer);
309        buffer.append("}");
310    }

329    private void appendDataURL(MilStdSymbol feature, StringBuffer buffer) {
330        //String iconURL = "[MIL_SYM_SERVICE_URL]/mil-sym-service/renderer/image/" + feature.getSymbolCode();
331        String iconURL;
332
333        ImageInfo oImageInfo = null;
334        int iconSize = 35; //(int) (this.map.getIconPixelSize() * ratioStdScreenToDeviceDPI);
335        SparseArray<String> saModifiers = feature.getUnitModifiers(this.map.getMilStdLabels());
336        SparseArray<String> saAttr = feature.getAttributes(iconSize,
337                this.map.isSelected(feature),
338                this.map.getSelectedStrokeStyle().getStrokeColor(),
339                this.map.getSelectedLabelStyle().getColor());
340
341        saAttr.put(MilStdAttributes.UseDashArray, "false");
342
343        iconURL = MilStdUtilities.getMilStdSinglePointIconURL(feature, eLabelSetting, oLabels, saAttr);
344
345        oImageInfo = this.oIconRenderer.RenderIcon(feature.getSymbolCode(), ((saModifiers == null) ? this.emptyArray : saModifiers), ((saAttr == null) ? this.emptyArray : saAttr));
346
347        if (null != oImageInfo) {
348            if (this.addExtendedData) {
349                Bitmap iconBitmap = oImageInfo.getImage();
350                ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
351                iconBitmap.compress(Bitmap.CompressFormat.PNG, 100, byteArrayOutputStream);
352                byte[] byteArray = byteArrayOutputStream.toByteArray();
353
354                String encoded = "data:image/png;base64," + Base64.encodeToString(byteArray, Base64.DEFAULT);
355
356                feature.setProperty(TEMP_DATAURL_STRING, encoded);
357            }
358
359            buffer.append("\"url\": {");
360            buffer.append(iconURL);
361            buffer.append("}"); // url
362            buffer.append(",");
363            tempIconStyle.setOffSetX(oImageInfo.getCenterPoint().x);
364            tempIconStyle.setOffSetY(oImageInfo.getImageBounds().height() - oImageInfo.getCenterPoint().y);
365            buffer.append("\"offsetX\": {");
366            buffer.append(tempIconStyle.getOffSetX());
367            buffer.append("}");
368            buffer.append(",");
369            buffer.append("\"offsetY\": {");
370            buffer.append(tempIconStyle.getOffSetY());
371            buffer.append("}");
372        }
373    }

375    private void appendGeoJSONPoint(IFeature feature, StringBuffer buffer) {
376        buffer.append("\"geometry\":  {\"type\": \"Point\",");
377        buffer.append("\"coordinates\":  ");
378        IGeoPosition position = feature.getPositions().get(0);
379        buffer.append("[");
380        buffer.append(position.getLatitude());
381        buffer.append(", ");
382        buffer.append(position.getLatitude());
383        buffer.append("]");
384        buffer.append("}");// end of geometry
385        buffer.append(",\n\"properties\": {");
386        buffer.append("\"style\": {");
387        buffer.append("\"iconStyle\": {");
388        if (feature.getFeatureType() == GEO_POINT) {
389            buffer.append("\"url\": {");
390            buffer.append("\"" + ((Point) feature).getIconURI() + "\"");
391            buffer.append("}"); // url
392        } else {
393            // must be single point milstd symbol
394            appendDataURL((MilStdSymbol)feature, buffer);
395        }
396        buffer.append("}"); // iconStyle
397        buffer.append("}"); // style
398        appendGeoJSONOtherProperties(feature, buffer);
399        buffer.append("}");
400    }

403     * Converts this object to a geoJSON string
404     * Recursive method goes down through child features till
405     * it finds a node

408    public void appendFeature(IFeature feature, StringBuffer buffer) throws IOException {
409
410        buffer.append("{\"type\":  \"Feature\",\n");
411        switch (feature.getFeatureType()) {
412            case GEO_RECTANGLE:
413                appendRectangle((Rectangle)feature, buffer);
414                break;
415            case GEO_SQUARE:
416                appendSquare((Square)feature, buffer);
417                break;
418            case GEO_CIRCLE:
419                appendCircle((Circle)feature, buffer);
420                break;
421            case GEO_ELLIPSE:
422                appendEllipse((Ellipse)feature, buffer);
423                break;
424            case GEO_POLYGON:
425                appendGeoJSONPolygon(feature, buffer);
426                break;
427            case GEO_PATH:
428                appendGeoJSONPath(feature, buffer);
429                break;
430            case GEO_POINT:
431                appendGeoJSONPoint(feature, buffer);
432                break;
433            case GEOJSON:
434            case KML:
435                Log.i(TAG, "Child feature can't be GEOJSON type");
436                break;
437            case GEO_ACM:
438            case GEO_TEXT:
439            case GEO_MIL_SYMBOL:
440                if(((MilStdSymbol)feature).isSinglePoint()) {
441                    appendGeoJSONPoint(feature, buffer);
442                }
443                break;
444            default:
445                buffer.append("}");
446        }
447    }

449    private void appendFeatureList(List<IFeature> featureList, StringBuffer buffer) throws IOException {
450        if (featureList.isEmpty()) {
451            return;
452        }
453        buffer.append("{\"type\":  \"FeatureCollection\",\n");
454        buffer.append("\"features\":[\n");
455        boolean addComma = false;
456        for (IFeature feature : featureList) {
457            if (addComma) {
458                buffer.append(",\n");
459            } else {
460                addComma = true;
461            }
462            appendFeature(feature, buffer);
463        }
464        buffer.append("]\n}\n");
465    }

467    private void getAllFeatures(IOverlay overlay, List<IFeature> featureList) {
468        for (IOverlay ovl : overlay.getOverlays()) {
469            getAllFeatures(ovl, featureList);  // recursive
470            featureList.addAll(ovl.getFeatures());
471        }
472    }

474    private void export(IFeature feature, StringBuffer buffer) throws IOException {
475        List<IFeature> featureList = null;
476        if (feature.getFeatureType() == GEOJSON) {
477            featureList = ((GeoJSON) feature).getFeatureList();
478            appendFeatureList(featureList, buffer);
479        } else if (feature.getFeatureType() == KML) {
480            featureList = ((mil.emp3.api.KML) feature).getFeatureList();
481            appendFeatureList(featureList, buffer);
482        } else if (feature.getChildFeatures().size() > 0) {
483            featureList = feature.getChildFeatures();
484            appendFeatureList(featureList, buffer);
485        } else {
486            appendFeature(feature, buffer);
487        }
488    }
489
490    private void export(List<IFeature> featureList, StringBuffer buffer) throws IOException {
491        appendFeatureList(featureList, buffer);
492    }
493
494    private void export(IOverlay overlay, StringBuffer buffer) throws IOException {
495        List<IFeature> featureList = new ArrayList<>();
496        getAllFeatures(overlay, featureList); // flatten it out
497        appendFeatureList(featureList, buffer);
498    }

500    private void export(IMap map, StringBuffer buffer) throws IOException {
501        List<IFeature> featureList = map.getAllFeatures();
502        appendFeatureList(featureList, buffer);
503    }

505    @Override
506    public void run() {
507        try {
508            StringBuffer buffer = new StringBuffer();
509            if (feature != null) {
510                export(feature, buffer);
511            } else if (featureList != null){
512                export(featureList, buffer);
513            } else if (overlay != null) {
514                export(overlay, buffer);
515            } else {
516                export(map, buffer);
517            }
518            this.callback.exportSuccess(buffer.toString());
519        } catch (Exception Ex) {
520            this.callback.exportFailed(Ex);
521        }
522    }
