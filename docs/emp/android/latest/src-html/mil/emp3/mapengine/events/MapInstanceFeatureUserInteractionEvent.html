001package mil.emp3.mapengine.events;
002
003import org.cmapi.primitives.IGeoPosition;
004
005import java.util.EnumSet;
006import java.util.List;
007
008import mil.emp3.api.enums.UserInteractionKeyEnum;
009import mil.emp3.api.enums.UserInteractionMouseButtonEnum;
010import mil.emp3.api.interfaces.IFeature;
011import mil.emp3.api.enums.UserInteractionEventEnum;
012import mil.emp3.api.events.Event;
013import mil.emp3.mapengine.interfaces.IMapInstance;
014
015/**
016 * This class implements the Map Instance Feature User Interaction Event. It must be generated by all map engine
017 * instances when the user interacts with a feature. See {@link UserInteractionEventEnum}.
018 * All events of this type are handled by the EMP core code.
019 */
020public class MapInstanceFeatureUserInteractionEvent extends Event<UserInteractionEventEnum, IMapInstance> {
021    private final android.graphics.Point oPoint;
022    private final IGeoPosition oCoordinate;
023    private final IGeoPosition oStartCoordinate;
024    private final List<IFeature> oFeatureList;
025    private UserInteractionMouseButtonEnum oButton;
026    private final EnumSet<UserInteractionKeyEnum> oKeys;
027
028    // eventConsumed is used internally by MapInstanceEventHandler. It is looked at by the GestureHandler (WW PickNavigateController)
029    // for DRAG event only. Remember that DRAG and DRAG_COMPLETE events never propagate to applications. Applications should not
030    // use this member.
031    private boolean eventConsumed;
032    /**
033     * This constructor must be called by the map engines to creates a MapInstanceFeatureUserInteractionEvent event
034     * @param oMapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
035     * @param eEvent The enumerated value of the event that occurred. See {@link UserInteractionEventEnum}.
036     * @param featureList A list of references to the features on which the event occurred.
037     * @param oPointCoordinate The geographic coordinate of the location the event occurred.
038     * @param oPosition The cartesian coordinate relative to the origin of the view container.
039     * @param button Mouse button, left, right or middle
040     * @param keys Keyboard keys, shift, ctrl and/or alt
041     */
042    public MapInstanceFeatureUserInteractionEvent(IMapInstance oMapInstance,
043            UserInteractionEventEnum eEvent,
044            EnumSet<UserInteractionKeyEnum> keys,
045            UserInteractionMouseButtonEnum button,
046            List<IFeature> featureList,
047            android.graphics.Point oPointCoordinate,
048            IGeoPosition oPosition) {
049        super(eEvent, oMapInstance);
050        this.oPoint = oPointCoordinate;
051        this.oCoordinate = oPosition;
052        this.oStartCoordinate = null;
053        this.oFeatureList = featureList;
054        this.oButton = button;
055        this.oKeys = keys;
056    }
057
058    /**
059     * This constructor must be called by the map engines to create a MapInstanceFeatureUserInteractionEvent event
060     * @param oMapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
061     * @param eEvent The enumerated value of the event that occurred. See {@link UserInteractionEventEnum}.
062     * @param featureList A list of references to the features on which the event occurred.
063     * @param oPointCoordinate The geographic coordinate of the location the event occurred.
064     * @param oPosition The cartesian coordinate relative to the origin of the view container.
065     * @param button Mouse button, left, right or middle
066     * @param keys Keyboard keys, shift, ctrl and/or alt
067     */
068    public MapInstanceFeatureUserInteractionEvent(IMapInstance oMapInstance,
069            UserInteractionEventEnum eEvent,
070            EnumSet<UserInteractionKeyEnum> keys,
071            UserInteractionMouseButtonEnum button,
072            List<IFeature> featureList,
073            android.graphics.Point oPointCoordinate,
074            IGeoPosition oPosition,
075            IGeoPosition oStartPosition) {
076        super(eEvent, oMapInstance);
077        this.oPoint = oPointCoordinate;
078        this.oCoordinate = oPosition;
079        this.oStartCoordinate = oStartPosition;
080        this.oFeatureList = featureList;
081        this.oButton = button;
082        this.oKeys = keys;
083    }
084
085    /**
086     * The geographic coordinate of the location the event occurred.
087     * @return location
088     */
089
090    public android.graphics.Point getLocation() {
091        return this.oPoint;
092    }
093
094    /**
095     * Returns geo-position of the location of the event.
096     * @return coordinates
097     */
098    public IGeoPosition getCoordinate() {
099        return this.oCoordinate;
100    }
101
102    /**
103     * Returns geo-position of the location where DRAG started
104     * @return coordinates
105     */
106    public IGeoPosition getStartCoordinate() {
107        return this.oStartCoordinate;
108    }
109
110    /**
111     * A list of references to the features on which the event occurred.
112     * @return list of references to features
113     */
114
115    public List<IFeature> getFeatures() {
116        return this.oFeatureList;
117    }
118
119    /**
120     * Returns mouse button, left, right or middle
121     * @return button
122     */
123
124    public UserInteractionMouseButtonEnum getButton() {
125        return this.oButton;
126    }
127
128    /**
129     * Returns keyboard keys, shift, ctrl and/or alt
130     * @return enumeration of keys
131     */
132
133    public EnumSet<UserInteractionKeyEnum> getKeys() {
134        return this.oKeys;
135    }
136
137    /**
138     * eventConsumed is used internally by MapInstanceEventHandler. It is looked at by the GestureHandler (WW PickNavigateController)
139     * for DRAG event only. Remember that DRAG and DRAG_COMPLETE events never propagate to applications. Applications should not
140     * use this method.
141     * @return true if event was consumed
142     */
143    public boolean isEventConsumed() {
144        return eventConsumed;
145    }
146
147    /**
148     * eventConsumed is used internally by MapInstanceEventHandler. It is looked at by the GestureHandler (WW PickNavigateController)
149     * for DRAG event only. Remember that DRAG and DRAG_COMPLETE events never propagate to applications. Applications should not
150     * use this method.
151     * @param eventConsumed
152     */
153    public void setEventConsumed(boolean eventConsumed) {
154        this.eventConsumed = eventConsumed;
155    }
156}
