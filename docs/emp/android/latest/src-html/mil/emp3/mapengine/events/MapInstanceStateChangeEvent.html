001package mil.emp3.mapengine.events;
002
003import mil.emp3.api.enums.MapEventEnum;
004import mil.emp3.api.enums.MapStateEnum;
005import mil.emp3.api.events.Event;
006import mil.emp3.mapengine.interfaces.IMapInstance;
007
008/**
009 * This class implements the Map Instance State Change Event. It must be generated by all map engine
010 * instances when the state of the map changes. See {@link MapStateEnum}.
011 * All events of this type are handled by the EMP core code.
012 */
013public class MapInstanceStateChangeEvent extends Event<MapEventEnum, IMapInstance> {
014    private final MapStateEnum eNewState;
015
016    /**
017     * This constructor must be called by the map engines to create a MapInstanceUserInteractionEvent event
018     * @param oMapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
019     * @param eState The enumerated value of the new state of the map instance. See {@link MapStateEnum}.
020     */
021    public MapInstanceStateChangeEvent(IMapInstance oMapInstance,
022            MapStateEnum eState) {
023        super(MapEventEnum.STATE_CHANGE, oMapInstance);
024        this.eNewState = eState;
025    }
026
027
028    /**
029     * The enumerated value of the new state of the map instance
030     * @return one of the following values
031     *   MAP_NEW, INIT_IN_PROGRESS, MAP_SWAP_IN_PROGRESS, MAP_READY, SHUTDOWN_IN_PROGRESS, SHUTDOWN
032     */
033
034    public MapStateEnum getState() {
035        return this.eNewState;
036    }
037}
