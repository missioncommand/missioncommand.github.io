001package mil.emp3.mapengine.events;
002
003import mil.emp3.api.enums.MapFeatureEventEnum;
004import mil.emp3.api.events.Event;
005import mil.emp3.api.interfaces.IFeature;
006import mil.emp3.mapengine.interfaces.IMapInstance;
007
008/**
009 * MapInstanceFeatureRemoved event must be must be generated by all map engines when a feature is removed from the map. See {@link MapFeatureEventEnum}
010 * All events of this type are handled by the EMP core code.
011 */
012public class MapInstanceFeatureRemovedEvent extends Event<MapFeatureEventEnum, IMapInstance> {
013    final private IFeature feature;
014
015    /**
016     * This constructor must be called by the map engines to create a MapInstanceUserInteractionEvent event
017     * @param mapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
018     * @param event The feature event generated. See {@link MapFeatureEventEnum}.
019     * @param feature The feature this event is associated with. See {@link IFeature}.
020     */
021    public MapInstanceFeatureRemovedEvent(IMapInstance mapInstance, MapFeatureEventEnum event, IFeature feature) {
022        super(event, mapInstance);
023        this.feature = feature;
024    }
025
026    /**
027     * Returns the feature for which the event was generated.
028     * @return the feature removed
029     */
030    public IFeature getFeature() {
031        return feature;
032    }
033}
