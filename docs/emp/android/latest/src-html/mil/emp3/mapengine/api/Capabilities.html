001package mil.emp3.mapengine.api;
002
003import java.util.Arrays;
004import mil.emp3.api.enums.FeatureTypeEnum;
005import mil.emp3.api.enums.WMSVersionEnum;
006import mil.emp3.api.enums.WMTSVersionEnum;
007import mil.emp3.mapengine.interfaces.IMapEngineCapabilities;
008
009/**
010 * EMP3 Core queries this class to determine the capabilities supported by a specific map engine.
011 * Currently this includes: shapes and symbols that can be plotted, version of supported WMS.
012 */
013public class Capabilities implements IMapEngineCapabilities {
014    private final java.util.Set<FeatureTypeEnum> canPlotSet;
015    private final java.util.Set<WMSVersionEnum> supportedWMSVersion;
016    private final java.util.Set<WMTSVersionEnum> supportedWMTSVersion;
017    
018    public Capabilities(FeatureTypeEnum[] aFeatureTypes, WMSVersionEnum[] aVersions,
019                        WMTSVersionEnum[] aWMTSVersions) {
020        this.canPlotSet = new java.util.HashSet<>();
021        this.supportedWMSVersion = new java.util.HashSet<>();
022        this.supportedWMTSVersion = new java.util.HashSet<>();
023        
024        if (aFeatureTypes != null) {
025            this.canPlotSet.addAll(Arrays.asList(aFeatureTypes));
026        }
027        
028        if (aVersions != null) {
029            this.supportedWMSVersion.addAll(Arrays.asList(aVersions));
030        }
031
032        if (aWMTSVersions != null) {
033            this.supportedWMTSVersion.addAll(Arrays.asList((aWMTSVersions)));
034        }
035    }
036
037    @Override
038    public boolean canPlot(FeatureTypeEnum eFeatureType) {
039        return this.canPlotSet.contains(eFeatureType);
040    }
041
042    @Override
043    public boolean wmsVersionSupported(WMSVersionEnum eVersion) {
044        return this.supportedWMSVersion.contains(eVersion);
045    }
046
047    @Override
048    public boolean wmtsVersionSupported(WMTSVersionEnum eVersion) {
049        return this.supportedWMTSVersion.contains(eVersion);
050    }
051
052}
