001package mil.emp3.mapengine.interfaces;
002
003/**
004 * An object that holds capabilities required by a specific map engine.
005 */
006public interface IMapEngineRequirements {
007    /**
008     * Returns true if map engine requires WMS capabilities.
009     * @return
010     */
011    boolean requiresWMSCapabilities();
012    /**
013     * Returns true if map engine requires WMTS capabilities.
014     * @return
015     */
016    public boolean requiresWMTSCapabilities();
017}
