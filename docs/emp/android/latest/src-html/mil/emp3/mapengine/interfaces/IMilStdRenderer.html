001package mil.emp3.mapengine.interfaces;
002
003import android.util.SparseArray;
004
005import org.cmapi.primitives.IGeoBounds;
006
007import java.util.List;
008
009import mil.emp3.api.MilStdSymbol;
010import mil.emp3.api.interfaces.IFeature;
011import mil.emp3.api.interfaces.core.ICoreManager;
012import mil.emp3.api.interfaces.core.IStorageManager;
013
014/**
015 * This object provides interface to Mild Std renderer. Ir is constructed in the EMP Core and registered with the Map Instance for its usage.
016 */
017public interface IMilStdRenderer {
018    void setStorageManager(IStorageManager storageManager);
019    void setCoreManager(ICoreManager coreManager);
020
021    /**
022     * Sets the Cache Directory used by renderer and also sets the available heap memory for the application
023     * @param sCacheDir
024     * @param getMemoryClass
025     */
026    void setRendererCacheDir(String sCacheDir, int getMemoryClass);
027
028    /**
029     * Gets a list of renderables that compose the MilStd tactical graphic symbol.
030     * @param mapInstance
031     * @param symbol
032     * @param selected
033     * @return A list of basic IFeature objects. See {@link IFeature).
034     */
035    List<IFeature> getTGRenderableShapes(IMapInstance mapInstance, MilStdSymbol symbol, boolean selected);
036    
037    /**
038     * Gets the bitmap that represents the MilStd icon symbol.
039     * @param sSymbolCode The symbol code that identifies the 2525 symbol.
040     * @param oModifiers An array of modifiers to apply to the icon.
041     * @param oAttr An array of attributes to use for the icon.
042     * @return {@link IEmpImageInfo}.
043     */
044    IEmpImageInfo getMilStdIcon(String sSymbolCode, SparseArray oModifiers, SparseArray oAttr);
045
046    /**
047     * Gets the 2525 modifiers defined in the symbol.
048     * @param mapInstance The map instance making the call.
049     * @param symbol the feature.
050     * @return SparseArray as per MilStd 2525 renderer
051     */
052    SparseArray<String> getUnitModifiers(IMapInstance mapInstance, MilStdSymbol symbol);
053
054    /**
055     * Gets the 2525 modifiers defined in the tactical graphic symbol.
056     * @param mapInstance The map instance making the call.
057     * @param symbol the feature.
058     * @return SparseArray as per MilStd 2525 renderer
059     */
060    SparseArray<String> getTGModifiers(IMapInstance mapInstance, MilStdSymbol symbol);
061
062    /**
063     * Gets the attributes defines in the feature.
064     * @param mapInstance The map instance making the call.
065     * @param feature the feature.
066     * @return SparseArray as per MilStd 2525 renderer
067     */
068    SparseArray<String> getAttributes(IMapInstance mapInstance, IFeature feature, boolean selected);
069
070    /**
071     * Gets the far distance threshold for the map. All single point MilStd icons at or
072     * beyond the far distances threshold (from the camera) are displayed as dots.
073     * @param mapInstance The map instance for request.
074     * @return The far distance threshold in meters.
075     */
076    double getFarDistanceThreshold(IMapInstance mapInstance);
077
078    /**
079     * Gets the mid distance threshold for the map. All single point MiloStd icons at
080     * a distance less than the mid distance threshold are displayed as fully qualified MilStd icons. The
081     * labels displayed will depend on the maps MilStd label setting.
082     * the distances
083     * @param mapInstance The map instance for request.
084     * @return The mid distance threshold in meters.
085     */
086    double getMidDistanceThreshold(IMapInstance mapInstance);
087
088    /**
089     * Returns the name of the cache used. In the current implementation if we want to use the RenderCache implemented
090     * by NASA WW then MilStdRenderer must use "NoBitmapCache".
091     * If we want to use the EMP cache then "BitmapCache" or "AdaptiveBitmapCache" should be used.
092     * "BitmapCache" and "AdaptiveBitmapCache" does not work well with NAS WW but is kept in place if we move to or add a new map engine.
093     * @return
094     */
095    String getBitmapCacheName();
096
097    /**
098     * Returns the selected icon scale factor for the map instance.
099     * @param mapInstance
100     * @return
101     */
102    double getSelectedIconScale(IMapInstance mapInstance);
103
104    /**
105     * Gets the list of renderables that represents the feature.
106     * @param mapInstance
107     * @param feature
108     * @param selected
109     * @return A list of basic IFeature objects. See {@link IFeature).
110     */
111    List<IFeature> getFeatureRenderableShapes(IMapInstance mapInstance, IFeature feature, boolean selected);
112}
