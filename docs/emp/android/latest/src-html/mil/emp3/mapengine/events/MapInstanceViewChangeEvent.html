001package mil.emp3.mapengine.events;
002
003import mil.emp3.api.events.Event;
004import mil.emp3.api.interfaces.ICamera;
005import mil.emp3.api.enums.MapEventEnum;
006import mil.emp3.api.enums.MapViewEventEnum;
007import mil.emp3.api.interfaces.ILookAt;
008import mil.emp3.mapengine.interfaces.IMapInstance;
009import org.cmapi.primitives.IGeoBounds;
010
011/**
012 * This class implements the Map Instance View Change Event. It must be generated by all map engine
013 * instances when the viewing area of the map changes.
014 * All events of this type are handled by the EMP core code.
015 */
016public class MapInstanceViewChangeEvent extends Event<MapViewEventEnum, IMapInstance> {
017    private final ICamera oCamera;
018    private final ILookAt oLookAt;
019    private final IGeoBounds oBounds;
020    private final int iViewWidth;
021    private final int iViewHeight;
022
023    /**
024     * This constructor must be called by the map engines to create a MapInstanceViewChangeEvent event
025     * @param oMapInstance The actual map instance. The this property of the object that implements the IMapInstance interface.
026     * @param viewEventEnum The view event generated. See {@link MapViewEventEnum}
027     * @param camera The new camera position of the view of the map. See {@link ICamera}.
028     * @param bounds The new view bounds.
029     */
030    public MapInstanceViewChangeEvent(IMapInstance oMapInstance,
031            MapViewEventEnum viewEventEnum,
032            ICamera camera,
033            ILookAt lookAt,
034            IGeoBounds bounds,
035            int iWidth,
036            int iHeight) {
037        super(viewEventEnum, oMapInstance);
038        this.oCamera = camera;
039        this.oBounds = bounds;
040        this.iViewHeight = iHeight;
041        this.iViewWidth = iWidth;
042        this.oLookAt = lookAt;
043    }
044
045    /**
046     * Returns the updated Camera object after the event.
047     * @return @See ICamera
048     */
049    public ICamera getCamera() {
050        return this.oCamera;
051    }
052
053    /**
054     * Returns the updated lookAt object after the event.
055     * @return @See ILookAt
056     */
057    public ILookAt getLookAt() { return this.oLookAt; }
058
059    /**
060     * Returns the map bounds after the event, it could be null based camera settings.
061     * @return bounds
062     */
063
064    public IGeoBounds getBounds() {
065        return this.oBounds;
066    }
067
068    /**
069     * Returns width of the map in pixels
070     * @return width
071     */
072    public int getMapViewWidth() {
073        return this.iViewWidth;
074    }
075
076    /**
077     * Returns height of the map in pixels
078     * @return height
079     */
080
081    public int getMapViewHeight() {
082        return this.iViewHeight;
083    }
084}
