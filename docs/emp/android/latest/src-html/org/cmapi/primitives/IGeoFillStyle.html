001package org.cmapi.primitives;
002
003
004  /**
005   *  Style properties for the fill of a feature.  This can either be the interior
006   *  of a shape such as a polygon or circle, or can also be applied to the
007   *  fill of an a MIL-STD-2525 Icon to override the default affiliation color
008   */
009public interface IGeoFillStyle {
010
011  /**
012
013   */
014    public void setDescription(String description);
015
016    public String getDescription();
017
018
019  /**
020
021   */
022    public void setFillColor(IGeoColor fillColor);
023
024    public IGeoColor getFillColor();
025
026
027  public enum FillPattern {
028    solid, 
029    hatched, 
030    crossHatched
031  }
032
033
034  /**
035   *  Fill patterns allow for alternatives to a solid fill color
036   */
037    public void setFillPattern(IGeoFillStyle.FillPattern fillPattern);
038
039    public IGeoFillStyle.FillPattern getFillPattern();
040
041}
