001package org.cmapi.primitives;
002
003
004  /**
005   *  Style attributes associated with icons
006   */
007  public class GeoIconStyle implements IGeoIconStyle  {
008
009    private double size = 32;
010    private double offSetX = 0;
011    private double offSetY = 0;
012
013  /**
014   * @param size        Size in pixels at 96 ppi.  If the system rendering the text is running at a screen resolution other than 96 ppi, a translation should be done to make the icon larger of smaller to represent the same amount of physical space as if the screen were at 96 ppi. If the icon is not square, the size will represent the larger value of the rectangle
015   */
016    public void setSize(double size){
017      this.size = size;
018    }
019
020
021  /**
022   * @return size        Size in pixels at 96 ppi.  If the system rendering the text is running at a screen resolution other than 96 ppi, a translation should be done to make the icon larger of smaller to represent the same amount of physical space as if the screen were at 96 ppi. If the icon is not square, the size will represent the larger value of the rectangle
023   */
024    public double getSize(){
025      return this.size;
026    }
027
028
029  /**
030   * @param offSetX        X Offest of icon in pixels derived from the lower left point of the icon
031   */
032    public void setOffSetX(double offSetX){
033      this.offSetX = offSetX;
034    }
035
036
037  /**
038   * @return offSetX        X Offest of icon in pixels derived from the lower left point of the icon
039   */
040    public double getOffSetX(){
041      return this.offSetX;
042    }
043
044
045  /**
046   * @param offSetY        X Offest of icon in pixels derived from the lower left point of the icon
047   */
048    public void setOffSetY(double offSetY){
049      this.offSetY = offSetY;
050    }
051
052
053  /**
054   * @return offSetY        X Offest of icon in pixels derived from the lower left point of the icon
055   */
056    public double getOffSetY(){
057      return this.offSetY;
058    }
059
060  }
