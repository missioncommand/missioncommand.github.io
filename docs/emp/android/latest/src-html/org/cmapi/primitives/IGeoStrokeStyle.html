001package org.cmapi.primitives;
002
003
004  /**
005   *  Style attributes for strokes
006   */
007public interface IGeoStrokeStyle {
008
009  /**
010
011   */
012    public void setStrokeColor(IGeoColor strokeColor);
013
014    public IGeoColor getStrokeColor();
015
016
017  /**
018   *  specifies a number whose lower 16 bits define a pattern of which pixels
019   *  in the image are white and which are transparent. Each bit corresponds to
020   *  a pixel, and the pattern repeats after every n*16 pixels, where n is
021   *  the factor. For example, if the factor is 3, each bit in the pattern is
022   *  repeated three times before using the next bit
023   */
024    public void setStipplingPattern(short stipplingPattern);
025
026    public short getStipplingPattern();
027
028
029  /**
030   *  specifies the number of times each bit in the pattern is repeated before
031   *  the next bit is used. For example, if the factor is 3, each bit is repeated
032   *  three times before using the next bit. The specified factor must be
033   *  either 0 or an integer greater than 0. A factor of 0 indicates no stippling
034   */
035    public void setStipplingFactor(int stipplingFactor);
036
037    public int getStipplingFactor();
038
039
040  /**
041   *  Width of the stroke on the screen in pixels
042   */
043    public void setStrokeWidth(double strokeWidth);
044
045    public double getStrokeWidth();
046
047}
