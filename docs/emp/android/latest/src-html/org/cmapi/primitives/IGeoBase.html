001package org.cmapi.primitives;
002
003
004  /**
005   *  This is the base object all containers and features are derived from
006   */
007public interface IGeoBase {
008
009  /**
010   *  Name is used to display text labels alongside to a feature and can be
011   *  used in user interfaces that manage the data on the map
012   */
013    public void setName(String name);
014
015    public String getName();
016
017
018  /**
019   *  The id value should be a UUID (see https://www.ietf.org/rfc/rfc4122.txt) to
020   *  avoid conflict of data from multiple sources.  All id values SHALL be
021   *  unique regardless of type and conflicting id values will be treated as
022   *  the same element.  Once this ID is set, it should not be changed to
023   *  avoid refernces by ID to break.  The setter is provided so that an item
024   *  can be deserialized from an object with an exisitng ID or provided an
025   *  exisiting ID immediately after instantiation.
026   */
027    public void setGeoId(java.util.UUID geoId);
028
029    public java.util.UUID getGeoId();
030
031
032  /**
033   *  This is a free form string that can represent an ID specific to this piece
034   *  of data as it is tracked uniquely by the system providing this data. 
035   *  This is different from GeoId as the GeoId is of type UUID.  Some systems
036   *  may use IDs formatted in a way that it cannot be stored in a UUID. 
037   *  In this case, this property can be used by the system to track this psece
038   *  of data by the ID they use internally.  The GeoID proerty will be used
039   *  by CMAPI as the unique ID and this ID is for the benefit of the system providing
040   *  the data.
041   */
042    public void setDataProviderId(String dataProviderId);
043
044    public String getDataProviderId();
045
046
047  /**
048   *  A simple string, or HTML formatted string that can be displayed to describe
049   *  the IGeoBase.  In the case of a feature the description property can
050   *  be used to store the content that should display in a pop up window after
051   *  clicking on a feature.
052   */
053    public void setDescription(String description);
054
055    public String getDescription();
056
057
058  /**
059   *  A key value pair where both the key and value where the key is a unique
060   *  string and the value is an object that can be stored as and read from
061   *  a string.  In cases where an IGeoBase will be serialized all of the values
062   *  will be serialized using the objects toString() method.  It is the resposibility
063   *  of the applications accessing values stored in the properties hash map
064   *  to determine if the value is in a string format and convert back to
065   *  whatever object type the string was derived from.  The object should not
066   *  contain any references to other objects where that relationship is expected
067   *  to exist beyond copying a current value of a simple type as the relationship will
068   *  not be properly restored when deserialization occurs.  CMAPI will not
069   *  use the properties hash for any internal purpose, it will simply allow
070   *  3rd party data to pass additional attributes that will stay associated with
071   *  the CMAPI object as it moves between applications and systems.
072   */
073    public java.util.HashMap<String, String> getProperties();
074
075}
