001package org.cmapi.primitives;

004  /**
005   *  Feature to represent MIL-STD-2525 B and C symbols for the entire symbology
006   *  standard
007   */
008public interface IGeoMilSymbol extends IGeoRenderable {
009
010  public enum SymbolStandard {
011    MIL_STD_2525C, 
012    MIL_STD_2525B
013  }
014
015
016  /**
017   *  Version of the MIL-STD-2525 standard to use
018   */
019    public void setSymbolStandard(IGeoMilSymbol.SymbolStandard symbolStandard);
020
021    public IGeoMilSymbol.SymbolStandard getSymbolStandard();
022
023
024  /**
025   *  Symbol code as defined by MIL-STD-2525
026   */
027    public void setSymbolCode(String symbolCode);
028
029    public String getSymbolCode();
030
031
032  public enum Modifier {
033    SYMBOL_ICON("A"), 
034    ECHELON("B"), 
035    QUANTITY("C"), 
036    TASK_FORCE_INDICATOR("D"), 
037    FRAME_SHAPE_MODIFIER("E"), 
038    REDUCED_OR_REINFORCED("F"), 
039    STAFF_COMMENTS("G"), 
040    ADDITIONAL_INFO_1("H"), 
041    ADDITIONAL_INFO_2("H1"), 
042    ADDITIONAL_INFO_3("H2"), 
043    EVALUATION_RATING("J"), 
044    COMBAT_EFFECTIVENESS("K"), 
045    SIGNATURE_EQUIPMENT("L"), 
046    HIGHER_FORMATION("M"), 
047    HOSTILE("N"), 
048    IFF_SIF("P"), 
049    DIRECTION_OF_MOVEMENT("Q"), 
050    MOBILITY_INDICATOR("R"), 
051    SIGINT_MOBILITY_INDICATOR("R2"), 
052    OFFSET_INDICATOR("S"), 
053    UNIQUE_DESIGNATOR_1("T"), 
054    UNIQUE_DESIGNATOR_2("T1"), 
055    EQUIPMENT_TYPE("V"), 
056    DATE_TIME_GROUP("W"), 
057    DATE_TIME_GROUP_2("W1"), 
058    ALTITUDE_DEPTH("X"), 
059    LOCATION("Y"), 
060    SPEED("Z"), 
061    SPECIAL_C2_HEADQUARTERS("AA"), 
062    FEINT_DUMMY_INDICATOR("AB"), 
063    INSTALLATION("AC"), 
064    PLATFORM_TYPE("AD"), 
065    EQUIPMENT_TEARDOWN_TIME("AE"), 
066    COMMON_IDENTIFIER("AF"), 
067    AUXILIARY_EQUIPMENT_INDICATOR("AG"),

074    AZIMUTH("AN"), 
075    ENGAGEMENT_BAR("AO"), 
076    COUNTRY_CODE("CC"), 
077    SONAR_CLASSIFICATION_CONFIDENCE("SCC");
078
079    private String value;
080
081    private Modifier(String value) {
082      this.value = value;
083    }
084
085    public String valueOf() {
086      return this.value;
087    }
088  }
089

096    public java.util.HashMap<IGeoMilSymbol.Modifier, String> getModifiers();
097
098}
