001package org.cmapi.primitives;
002
003
004  /**
005   *  Style attributes for strokes
006   */
007  public class GeoStrokeStyle implements IGeoStrokeStyle  {
008
009    private IGeoColor strokeColor = new GeoColor();
010    private short stipplingPattern = 0;
011    private int stipplingFactor = 0;
012    private double strokeWidth = 3;
013
014  /**
015   * @param strokeColor        
016   */
017    public void setStrokeColor(IGeoColor strokeColor){
018      this.strokeColor = strokeColor;
019    }
020
021
022  /**
023   * @return strokeColor        
024   */
025    public IGeoColor getStrokeColor(){
026      return this.strokeColor;
027    }
028
029
030  /**
031   * @param stipplingPattern        specifies a number whose lower 16 bits define a pattern of which pixels in the image are white and which are transparent. Each bit corresponds to a pixel, and the pattern repeats after every n*16 pixels, where n is the factor. For example, if the factor is 3, each bit in the pattern is repeated three times before using the next bit
032   */
033    public void setStipplingPattern(short stipplingPattern){
034      this.stipplingPattern = stipplingPattern;
035    }
036
037
038  /**
039   * @return stipplingPattern        specifies a number whose lower 16 bits define a pattern of which pixels in the image are white and which are transparent. Each bit corresponds to a pixel, and the pattern repeats after every n*16 pixels, where n is the factor. For example, if the factor is 3, each bit in the pattern is repeated three times before using the next bit
040   */
041    public short getStipplingPattern(){
042      return this.stipplingPattern;
043    }
044
045
046  /**
047   * @param stipplingFactor        specifies the number of times each bit in the pattern is repeated before the next bit is used. For example, if the factor is 3, each bit is repeated three times before using the next bit. The specified factor must be either 0 or an integer greater than 0. A factor of 0 indicates no stippling
048   */
049    public void setStipplingFactor(int stipplingFactor){
050      this.stipplingFactor = stipplingFactor;
051    }
052
053
054  /**
055   * @return stipplingFactor        specifies the number of times each bit in the pattern is repeated before the next bit is used. For example, if the factor is 3, each bit is repeated three times before using the next bit. The specified factor must be either 0 or an integer greater than 0. A factor of 0 indicates no stippling
056   */
057    public int getStipplingFactor(){
058      return this.stipplingFactor;
059    }
060
061
062  /**
063   * @param strokeWidth        Width of the stroke on the screen in pixels
064   */
065    public void setStrokeWidth(double strokeWidth){
066      this.strokeWidth = strokeWidth;
067    }
068
069
070  /**
071   * @return strokeWidth        Width of the stroke on the screen in pixels
072   */
073    public double getStrokeWidth(){
074      return this.strokeWidth;
075    }
076
077  }
