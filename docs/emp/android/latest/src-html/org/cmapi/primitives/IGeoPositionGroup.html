001package org.cmapi.primitives;
002
003
004  /**
005   *  An ordered list of positions representing a 1 or more positions with an
006   *  associated IGeoAltitudeMode to interpret the altitude values.  In the
007   *  case of a point, a single position will create a single icon, wheras mulitple
008   *  positions will create the same icon at multiple positions to be interpreted as
009   *  a composite feature. For consistency, and IGeoRenderables use an IGeoPositionGroup
010   *  even when only containing a single position
011   */
012public interface IGeoPositionGroup extends IGeoAltitudeMode {
013
014  /**
015   *  An ordered list of IGeoPosition objects representing a 1 or more positions
016   *  with an associated IGeoAltitudeMode to interpret the altitude values. 
017   *  In the case of a point, a single position will create a single icon, wheras
018   *  mulitple positions will create the same icon at multiple positions to
019   *  be interpreted as a composite feature. For consistency, and IGeoRenderables use
020   *  an IGeoPositionGroup even when only containing a single positions
021   */
022    public void setPositions(java.util.List<IGeoPosition> positions);
023
024    public java.util.List<IGeoPosition> getPositions();
025
026
027  /**
028   *  Defines a point in time that something occurred, was created, or was
029   *  last updated time value as defined by http://tools.ietf.org/html/rfc3339
030   */
031    public void setTimeStamp(java.util.Date timeStamp);
032
033    public java.util.Date getTimeStamp();
034
035
036  /**
037   *  Defines one or more periods of time something occured, or was active.
038   */
039    public void setTimeSpans(java.util.List<IGeoTimeSpan> timeSpans);
040
041    public java.util.List<IGeoTimeSpan> getTimeSpans();
042
043}
