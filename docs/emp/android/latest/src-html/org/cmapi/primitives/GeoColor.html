001package org.cmapi.primitives;
002
003
004  /**
005   *  Object containing three color channels for red, green, and blue as well
006   *  as an alpha channel for opacity
007   */
008  public class GeoColor implements IGeoColor  {
009
010    private int red = 0;
011    private int green = 0;
012    private int blue = 0;
013    private double alpha = 0.8;
014
015  /**
016   * @param red        Value for red color channel. Integer ranging between 0 and 255
017   */
018    public void setRed(int red){
019      this.red = red;
020    }
021
022
023  /**
024   * @return red        Value for red color channel. Integer ranging between 0 and 255
025   */
026    public int getRed(){
027      return this.red;
028    }
029
030
031  /**
032   * @param green        Value for green color channel. Integer ranging between 0 and 255
033   */
034    public void setGreen(int green){
035      this.green = green;
036    }
037
038
039  /**
040   * @return green        Value for green color channel. Integer ranging between 0 and 255
041   */
042    public int getGreen(){
043      return this.green;
044    }
045
046
047  /**
048   * @param blue        Value for blue color channel. Integer ranging between 0 and 255
049   */
050    public void setBlue(int blue){
051      this.blue = blue;
052    }
053
054
055  /**
056   * @return blue        Value for blue color channel. Integer ranging between 0 and 255
057   */
058    public int getBlue(){
059      return this.blue;
060    }
061
062
063  /**
064   * @param alpha        Value for alpha channel to control opacity. Decimal ranging from 0 to 1
065   */
066    public void setAlpha(double alpha){
067      this.alpha = alpha;
068    }
069
070
071  /**
072   * @return alpha        Value for alpha channel to control opacity. Decimal ranging from 0 to 1
073   */
074    public double getAlpha(){
075      return this.alpha;
076    }
077
078  }
