001package org.cmapi.primitives;
002
003
004  /**
005   *  File containing geospatial feature data in a known format such as KML
006   *  or GeoJSON
007   */
008public interface IGeoDocument extends IGeoBase {
009
010  /**
011   *  URL the document should be loaded from, or a dataURI encoding of the
012   *  resource embedded as the value defined by RFC 2397 (see https://tools.ietf.org/html/rfc2397)
013   */
014    public void setDocumentURI(String documentURI);
015
016    public String getDocumentURI();
017
018
019  /**
020   *  MIME Type of document.  Common Types are KML (application/vnd.google-earth.kml+xml), and
021   *  GeoJSON (application/vnd.geo+json)
022   */
023    public void setDocumentMIMEType(String documentMIMEType);
024
025    public String getDocumentMIMEType();
026
027}
