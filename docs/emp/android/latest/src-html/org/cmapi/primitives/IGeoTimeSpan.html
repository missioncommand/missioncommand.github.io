001package org.cmapi.primitives;
002
003
004  /**
005   *  Defines a range in time that represents a beginning and end.  This can
006   *  be used to describe things such as availability, visibility, active/inactive for
007   *  provided periods of time.  From a filtering point of view, if a time filter
008   *  is applied a feature would only be visible on a map view within the defined
009   *  timespan(s)
010   */
011public interface IGeoTimeSpan {
012
013  /**
014   *  Starting date time value as defined by http://tools.ietf.org/html/rfc3339
015   */
016    public void setBegin(java.util.Date begin);
017
018    public java.util.Date getBegin();
019
020
021  /**
022   *  Ending date time value as defined by http://tools.ietf.org/html/rfc3339
023   */
024    public void setEnd(java.util.Date end);
025
026    public java.util.Date getEnd();
027
028}
