004  /**
005   *  Air control measures (ACM) represent a three dimensional volume in the
006   *  air above earth used to indicate where aircarfat should stay within, or
007   *  stay out of
008   */
009public interface IGeoAirControlMeasure extends IGeoRenderable {
010
011  public enum AcmType {
012    ROUTE, 
013    CYLINDER, 
014    ORBIT, 
015    POLYGON, 
016    RADARC, 
017    POLYARC, 
018    TRACK, 
019    CURTAIN
020  }
021
022
023  /**
024   *  Type of air control measure to create
025   */

027
028    public IGeoAirControlMeasure.AcmType getAcmType();
029
030

032    RADIUS, 
033    INNER_RADIUS, 
034    TURN, 
035    MIN_ALT, 
036    MAX_ALT, 
037    LEFT_AZIMUTH, 
038    RIGHT_AZIMUTH, 
039    WIDTH, 
040    LEFT_WIDTH, 
041    RIGHT_WIDTH;
042
043  }
044
045
046  /**
047
048   */
049    public void setAttributes(java.util.HashMap<IGeoAirControlMeasure.Attribute, String> attributes);
050
051    public java.util.HashMap<IGeoAirControlMeasure.Attribute, String> getAttributes();
052
053}
