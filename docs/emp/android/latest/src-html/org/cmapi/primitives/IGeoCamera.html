001package org.cmapi.primitives;
002
003
004  /**
005   *  Defines the virtual camera that views the scene. This element defines the
006   *  position of the camera relative to the Earth's surface as well as the
007   *  viewing direction of the camera. The camera position is defined by <longitude>,
008   *  <latitude>, <altitude>, and either <altitudeMode>. The viewing direction of
009   *  the camera is defined by <azimuth>, <tilt>, and <roll>. An IGeoCamera provides
010   *  full six-degrees-of-freedom control over the view, so you can position the
011   *  Camera in space and then rotate it around the X, Y, and Z axes. Most importantly,
012   *  you can tilt the camera view so that you're looking above the horizon into
013   *  the sky. The order of rotation is important. By default, the camera is
014   *  looking straight down the −Z axis toward the Earth. Before rotations are
015   *  performed, the camera is translated along the Z axis to <altitude>. The
016   *  order of transformations is as follows: <altitude> - translate along the
017   *  Z axis to <altitude><azimuth> - rotate around the Z axis.<tilt> - rotate
018   *  around the X axis.<roll> - rotate around the Z axis (again). Note that
019   *  each time a rotation is applied, two of the camera axes change their orientation.
020   */
021public interface IGeoCamera extends IGeoBase, IGeoAltitudeMode, IGeoPosition {
022
023  /**
024   *  Rotation, in degrees, of the camera around the X axis. A value of 0
025   *  indicates that the view is aimed straight down toward the earth (the most
026   *  common case). A value for 90 for <tilt> indicates that the view is aimed
027   *  toward the horizon. Values greater than 90 indicate that the view is
028   *  pointed up into the sky. Values for <tilt> are clamped at +180 degrees
029   */
030    public void setTilt(double tilt);
031
032    public double getTilt();
033
034
035  /**
036   *  Rotation, in degrees, of the camera around the Z axis. Values range from
037   *  −180 to +180 degrees
038   */
039    public void setRoll(double roll);
040
041    public double getRoll();
042
043
044  /**
045   *  Direction (that is, North, South, East, West), in degrees. Default=0 (North).
046   *  Values range from 0 to 360 degrees
047   */
048    public void setHeading(double heading);
049
050    public double getHeading();
051
052}
