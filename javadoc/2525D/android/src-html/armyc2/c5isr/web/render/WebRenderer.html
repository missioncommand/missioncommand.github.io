130     * Set minimum level at which an item can be logged.
131     * In descending order:
132     * OFF = Integer.MAX_VALUE
133     * Severe = 1000
134     * Warning = 900
135     * Info = 800
136     * Config = 700
137     * Fine = 500
138     * Finer = 400 
139     * Finest = 300

141     * Use like WebRenderer.setLoggingLevel(Level.INFO);
142     * or
143     * Use like WebRenderer.setLoggingLevel(800);
144     * @param level int
145     */
146    public static void setLoggingLevel(int level)
147    {
148        try
149        {
150            if(level > 1000)
151                  ErrorLogger.setLevel(Level.OFF,true);
152            else if(level > 900)
153                  ErrorLogger.setLevel(Level.SEVERE,true);
154            else if(level > 800)
155                  ErrorLogger.setLevel(Level.WARNING,true);
156            else if(level > 700)
157                  ErrorLogger.setLevel(Level.INFO,true);
158            else if(level > 500)
159                  ErrorLogger.setLevel(Level.CONFIG,true);
160            else if(level > 400)
161                  ErrorLogger.setLevel(Level.FINE,true);
162            else if(level > 300)
163                  ErrorLogger.setLevel(Level.FINER,true);
164            else if(level > Integer.MIN_VALUE)
165                  ErrorLogger.setLevel(Level.FINEST,true);
166            else
167                ErrorLogger.setLevel(Level.ALL,true);
168            
169            ErrorLogger.LogMessage("WebRenderer", "setLoggingLevel(int)",
170                    "Logging level set to: " + ErrorLogger.getLevel().getName(), 
171                    Level.CONFIG);
172        }
173        catch(Exception exc)
174        {
175            ErrorLogger.LogException("WebRenderer", "setLoggingLevel(int)", exc, Level.INFO);
176        }
177    }

220    /**
221     * Renders all multi-point symbols, creating KML that can be used to draw
222     * it on a Google map.  Multipoint symbols cannot be draw the same 
223     * at different scales. For instance, graphics with arrow heads will need to 
224     * redraw arrowheads when you zoom in on it.  Similarly, graphics like a 
225     * Forward Line of Troops drawn with half circles can improve performance if 
226     * clipped when the parts of the graphic that aren't on the screen.  To help 
227     * readjust graphics and increase performance, this function requires the 
228     * scale and bounding box to help calculate the new locations.
229     * @param id A unique identifier used to identify the symbol by Google map. 
230     * The id will be the folder name that contains the graphic.
231     * @param name a string used to display to the user as the name of the 
232     * graphic being created.
233     * @param description a brief description about the graphic being made and 
234     * what it represents.
235     * @param symbolCode A 20-30 digit symbolID corresponding to one of the
236     * graphics in the MIL-STD-2525D
237     * @param controlPoints The vertices of the graphics that make up the
238     * graphic.  Passed in the format of a string, using decimal degrees 
239     * separating lat and lon by a comma, separating coordinates by a space.  
240     * The following format shall be used "x1,y1[,z1] [xn,yn[,zn]]..."
241     * @param altitudeMode Indicates whether the symbol should interpret 
242     * altitudes as above sea level or above ground level. Options are 
243     * "clampToGround", "relativeToGround" (from surface of earth), "absolute" 
244     * (sea level), "relativeToSeaFloor" (from the bottom of major bodies of 
245     * water).
246     * @param scale A number corresponding to how many meters one meter of our 
247     * map represents. A value "50000" would mean 1:50K which means for every 
248     * meter of our map it represents 50000 meters of real world distance.
249     * @param bbox The viewable area of the map.  Passed in the format of a
250     * string "lowerLeftX,lowerLeftY,upperRightX,upperRightY." Not required
251     * but can speed up rendering in some cases.
252     * example: "-50.4,23.6,-42.2,24.2"
253     * @param modifiers keyed using constants from Modifiers.
254     * Pass in comma delimited String for modifiers with multiple values like AM, AN &amp; X
255     * @param attributes keyed using constants from MilStdAttributes.
256     * @param format An enumeration: 2 for GeoJSON.
257     * @return A JSON string representation of the graphic.
258     */
259    public static String RenderSymbol(String id, String name, String description,
260                                      String symbolCode, String controlPoints, String altitudeMode,
261                                      double scale, String bbox, Map<String,String> modifiers, Map<String,String> attributes, int format) {
262        String output = "";
263        try {         
264                
265                JavaRendererUtilities.addAltModeToModifiersString(attributes,altitudeMode);
266        
267
268            output = MultiPointHandler.RenderSymbol(id, name, description, symbolCode, controlPoints,
269                    scale, bbox, modifiers, attributes, format);
270
271            //DEBUGGING
272            if(ErrorLogger.getLevel().intValue() <= Level.FINER.intValue())
273            {
274                System.out.println("");
275                StringBuilder sb = new StringBuilder();
276                sb.append("\nID: " + id + "\n");
277                sb.append("Name: " + name + "\n");
278                sb.append("Description: " + description + "\n");
279                sb.append("SymbolID: " + symbolCode + "\n");
280                sb.append("Scale: " + String.valueOf(scale) + "\n");
281                sb.append("BBox: " + bbox + "\n");
282                sb.append("Coords: " + controlPoints + "\n");
283                sb.append("Modifiers: " + modifiers + "\n");
284                ErrorLogger.LogMessage("WebRenderer", "RenderSymbol", sb.toString(),Level.FINER);
285            }
286            if(ErrorLogger.getLevel().intValue() <= Level.FINEST.intValue())
287            {
288                String briefOutput = output.replaceAll("</Placemark>", "</Placemark>\n");
289                briefOutput = output.replaceAll("(?s)<description[^>]*>.*?</description>", "<description></description>");
290                ErrorLogger.LogMessage("WebRenderer", "RenderSymbol", "Output:\n" + briefOutput,Level.FINEST);
291            }
292
293            
294            
295        } catch (Exception ea) {
296            
297            output = "{\"type\":'error',error:'There was an error creating the MilStdSymbol - " + ea.toString() + "'}";
298            ErrorLogger.LogException("WebRenderer", "RenderSymbol", ea, Level.WARNING);
299        }
300        
301        return output;
302    }

308    /**
309     * Renders all multi-point symbols, creating KML or JSON for the user to
310     * parse and render as they like.
311     * This function requires the bounding box to help calculate the new
312     * locations.
313     * @param id A unique identifier used to identify the symbol by Google map.
314     * The id will be the folder name that contains the graphic.
315     * @param name a string used to display to the user as the name of the 
316     * graphic being created.
317     * @param description a brief description about the graphic being made and 
318     * what it represents.
319     * @param symbolCode A 20-30 digit symbolID corresponding to one of the
320     * graphics in the MIL-STD-2525D
321     * @param controlPoints The vertices of the graphics that make up the
322     * graphic.  Passed in the format of a string, using decimal degrees
323     * separating lat and lon by a comma, separating coordinates by a space.
324     * The following format shall be used "x1,y1 [xn,yn]..."
325     * @param pixelWidth pixel dimensions of the viewable map area
326     * @param pixelHeight pixel dimensions of the viewable map area
327     * @param bbox The viewable area of the map.  Passed in the format of a
328     * string "lowerLeftX,lowerLeftY,upperRightX,upperRightY."
329     * example: "-50.4,23.6,-42.2,24.2"
330     * @param modifiers keyed using constants from Modifiers.
331     * Pass in comma delimited String for modifiers with multiple values like AM, AN &amp; X
332     * @param attributes keyed using constants from MilStdAttributes.
333     * @param format An enumeration: 2 for GeoJSON.
334     * @return A JSON (1) or KML (0) string representation of the graphic.
335     */
336    public static String RenderSymbol2D(String id, String name, String description, String symbolCode, String controlPoints,
337            int pixelWidth, int pixelHeight, String bbox, Map<String,String> modifiers, Map<String,String> attributes, int format)
338    {
339        String output = "";
340        try
341        {
342            output = MultiPointHandler.RenderSymbol2D(id, name, description, 
343                    symbolCode, controlPoints, pixelWidth, pixelHeight, bbox, 
344                    modifiers, attributes, format);
345        }
346        catch(Exception exc)
347        {
348            output = "{\"type\":'error',error:'There was an error creating the MilStdSymbol: " + symbolCode + " - " + exc.toString() + "'}";
349        }
350        return output;
351    }
