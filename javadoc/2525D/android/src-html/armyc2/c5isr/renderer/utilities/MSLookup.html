049    public final synchronized void init(Context context)
050    {
051        if (_initCalled == false)
052        {
053            _MSLookupD = new HashMap<>();
054            _MSLookupE = new HashMap<>();
055            _IDListD = new ArrayList<>();
056            _IDListE = new ArrayList<>();
057
058            try
059            {
060                InputStream isD = context.getResources().openRawResource(R.raw.msd);
061                InputStream isE = context.getResources().openRawResource(R.raw.mse);
062
063                BufferedReader brD = new BufferedReader(new InputStreamReader(isD));
064                populateLookup(brD, SymbolID.Version_2525Dch1);
065                brD.close();
066
067                BufferedReader brE = new BufferedReader(new InputStreamReader(isE));
068                populateLookup(brE, SymbolID.Version_2525E);
069                brE.close();
070
071                _initCalled = true;
072            }
073            catch (Exception e)
074            {
075                System.out.println(e.getMessage());
076            }
077        }
078    }

080    private void populateLookup(BufferedReader br, int version)
081    {
082        Map<String, MSInfo> lookup;
083        List<String> list;
084        String[] temp = null;
085        String delimiter = "\t";
086
087        try
088        {
089            if(version == SymbolID.Version_2525E) {
090                lookup = _MSLookupE;
091                list = _IDListE;
092            }
093            else//2525Dch1
094            {
095                lookup = _MSLookupD;
096                list = _IDListD;
097            }
098
099            String id = null;
100            String ss = null;
101            String e = null;
102            String et = null;
103            String est = null;
104            String ec = null;
105            String g = null;
106            String dr = null;
107            String m = null;
108            String[] modifiers = null;
109
110            String line = br.readLine();
111            while (line != null) {
112                //parse first line
113                temp = line.split(delimiter);
114
115                if(temp.length < 5)
116                    ec = "000000";
117                else
118                    ec = temp[4];
119                if(ec.equals("0"))
120                    ec = "000000";
121
122                if(temp.length < 4)
123                    est = "";
124                else
125                    est = temp[3];
126                if(temp.length < 3)
127                    et = "";
128                else if(est.equals(""))
129                    et = temp[2];
130                if(temp.length < 2)
131                    e = "";
132                else if(et.equals(""))
133                    e = temp[1];
134
135                if(!temp[0].equals(""))
136                    ss = temp[0];
137
138                id = ss + ec;
139
140                if(!ec.equals("000000"))
141                {
142                    if (temp.length >= 7)
143                    {//Control Measures and METOCS
144                        if(temp.length >= 8)
145                        {
146                            m = temp[7];//modifiers
147                            if(m != null && !m.equals(""))
148                                //m = m.replace("\"","");
149                                modifiers = m.split(",");
150                            else
151                                modifiers = null;
152                        }
153                        g = temp[5];//geometry
154                        dr = temp[6];//draw rule
155                        lookup.put(id, new MSInfo(version, ss, e, et, est, ec, g, dr, populateModifierList(modifiers)));
156                    }
157                    else
158                    {//Everything else
159                        //_MSLookupD.put(id, new MSInfo(ss, e, et, est, ec));
160                        lookup.put(id, new MSInfo(version, ss, e, et, est, ec,populateModifierList(ss,ec, version)));
161                    }
162                    list.add(id);
163                }
164
165                modifiers = null;
166
167                //read next line for next loop
168                line = br.readLine();
169            }
170
171        }
172        catch(Exception exc)
173        {
174            System.out.println(exc.getMessage());
175        }
176
177    }

179    private ArrayList<String> populateModifierList(String[] modifiers)
180    {
181        ArrayList<String> mods = new ArrayList<String>();
182        if(modifiers != null && modifiers.length > 0)
183        {
184            for (String mod : modifiers)
185            {
186                mods.add(Modifiers.getModifierKey(mod));
187            }
188        }
189        return mods;
190    }

663    /**
664     *
665     * @param  symbolID Full 20-30 digits from the symbol code
666     * @return
667     */
668    public MSInfo getMSLInfo(String symbolID)
669    {
670        int length = symbolID.length();
671
672        if(length >= 20 && length <= 30)
673        {
674            int version = SymbolID.getVersion(symbolID);
675            return getMSLInfo(SymbolUtilities.getBasicSymbolID(symbolID), version);
676        }
677        else
678            return null;
679    }

681    /**
682     *
683     * @param basicID id SymbolSet + Entity code like 50110100
684     * @param version like SymbolID.Version_2525Dch1
685     * @return
686     */
687    public MSInfo getMSLInfo(String basicID, int version)
688    {
689        int length = basicID.length();
690        if(length == 8)
691        {
692            if(version < SymbolID.Version_2525E)
693                return _MSLookupD.getOrDefault(basicID, null);
694            else if(version == SymbolID.Version_2525E)
695                return _MSLookupE.getOrDefault(basicID, null);
696            else
697                return _MSLookupD.getOrDefault(basicID, null);
698        }
699        else if(length >= 20 && length <= 30)//probably got a full id instead of a basic ID.
700        {
701            if(version < SymbolID.Version_2525E)
702                return _MSLookupD.getOrDefault(SymbolUtilities.getBasicSymbolID(basicID), null);
703            else if(version == SymbolID.Version_2525E)
704                return _MSLookupE.getOrDefault(SymbolUtilities.getBasicSymbolID(basicID), null);
705            else
706                return _MSLookupD.getOrDefault(SymbolUtilities.getBasicSymbolID(basicID), null);
707        }
708        else
709            return null;
710    }

726    /*
727     * For use only by MilStdIconRenderer.addCustomSymbol()
728     * @param msInfo
729     * @return
730     */
731    public boolean addCustomSymbol(MSInfo msInfo)
732    {
733        boolean success = false;
734        try
735        {
736            int version = msInfo.getVersion();
737            if (version < SymbolID.Version_2525E)
738            {
739                if(this._IDListD.indexOf(msInfo.getBasicSymbolID()) == -1)
740                {
741                    this._IDListD.add(msInfo.getBasicSymbolID());
742                    MSLookup._MSLookupD.put(msInfo.getBasicSymbolID(), msInfo);
743                    success = true;
744                }
745                else
746                    ErrorLogger.LogMessage("Symbol Set and Entity Code combination already exist: " + msInfo.getBasicSymbolID(), Level.INFO,false);
747            }
748            else if (version == SymbolID.Version_2525E)
749            {
750                if(this._IDListE.indexOf(msInfo.getBasicSymbolID()) == -1)
751                {
752                    this._IDListE.add(msInfo.getBasicSymbolID());
753                    MSLookup._MSLookupE.put(msInfo.getBasicSymbolID(), msInfo);
754                    success = true;
755                }
756                else
757                    ErrorLogger.LogMessage("Symbol Set and Entity Code combination already exist: " + msInfo.getBasicSymbolID(), Level.INFO,false);
758            }
759        }
760        catch(Exception e)
761        {
762            ErrorLogger.LogException("MSLookup", "addCustomSymbol",e);
763        }
764        return success;
765    }
